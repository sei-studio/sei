```typescript
async function runModerationGate(
  characterId: string,
): Promise<import('./cloud/moderationGate').PublishResult> {
  const { isCloudWriteAllowed } = await import('./auth/authState');
  if (!(await isCloudWriteAllowed())) {
    return {
      ok: false,
      code: 'CLOUD_MODERATION_PROVIDER_UNAVAILABLE',
      friendlyMessage: 'Please sign in and accept the Terms of Service before publishing.',
    } as const;
  }
  const { getClient } = await import('./auth/supabaseClient');
  const { data: { session } } = await getClient().auth.getSession();
  if (!session?.access_token) {
    return {
      ok: false,
      code: 'CLOUD_MODERATION_PROVIDER_UNAVAILABLE',
      friendlyMessage: 'Please sign in before publishing.',
    } as const;
  }
  const accessToken = session.access_token;

  const [
    { publishWithModeration },
    { callEdgeFunction },
    { upsertCharacter, downloadCharacter },
    { getSupabaseUrl },
  ] = await Promise.all([
    import('./cloud/moderationGate'),
    import('./auth/edgeFunctionClient'),
    import('./cloud/cloudCharacterClient'),
    import('./env'),
  ]);

  // Adapter bag is identical to the one in the old browse.publishWithModeration
  // handler at the prior ipc.ts:821-924 location. Copy it verbatim.
  return await publishWithModeration(characterId, {
    callEdgeFunction: async <T,>(name, opts) => {
      const res = await callEdgeFunction(name, {
        jwt: opts.jwt,
        body: opts.body,
        timeoutMs: opts.timeoutMs,
      });
      if (!res.ok) {
        throw new Error(`edge-${name}-failed: status=${res.status} ${res.message}`);

        playtime_ms: Number(char.playtime_ms ?? 0),
        portrait_image: (char.portrait_image as string | null) ?? null,

          png_sha256: (char.skin_png_sha256 as string | null) ?? null,
          applied_at: (char.skin_applied_at as string | null) ?? null,
        },
        username: (char.username as string | null) ?? null,
        metadata: (char.metadata as Record<string, unknown>) ?? {},

      return {

        owner: session.user.id,

        username: c.username,

      };
    },

}
```

**Part B — modify `chars.save` handler (src/main/ipc.ts:128).**

CURRENT BEHAVIOR: parses Character + optional `{ skipExpansion }`, then either saveCharacter or expandAndSaveCharacter, returns Character.

NEW BEHAVIOR: if `character.shared === true` AND `!character.is_default`, the cloud-side publication MUST be gated by publishWithModeration. The cleanest insertion is:
  1. Run the LLM expansion (or skipExpansion local write) AS TODAY to produce a `persisted: Character` value. This populates persona.expanded which moderation needs.
  2. After persisted.shared===true && !persisted.is_default: BEFORE the saveCharacter mirror enqueues the cloud upsert (which already happens inside saveCharacter for non-defaults), force a non-shared local write first, then call runModerationGate(persisted.id), then on ok flip shared=true locally.
  3. On `ok:false`, the local row STAYS at shared=false and we throw an Error carrying `result.friendlyMessage` so the existing renderer toast surfaces it. The user can edit the persona and re-share via the CharacterPage toggle.

CRITICAL ORDERING — the cloud upsert (which sets shared=true server-side) MUST happen INSIDE publishWithModeration only. Today saveCharacter unconditionally enqueues a cloud mirror via syncQueue. We must avoid the race where the mirror queue uploads shared=true BEFORE the moderation gate runs.

Implementation pattern: when chars.save sees a shared=true non-default character, the handler:
  a. Calls `saveCharacter({ ...character, shared: false })` (or expandAndSaveCharacter for the non-skipExpansion path) → produces `persistedPrivate` with shared=false on disk + queues a cloud upsert with shared=false (safe — the cloud row will land as private).
  b. Calls `runModerationGate(persistedPrivate.id)`.
  c. If `result.ok === true`: moderationGate has already done the cloud upsert with shared=true via its own upsertCharacter adapter. Re-run a local write `saveCharacterRaw({ ...persistedPrivate, shared: true })` (via `import('./characterStore').then(m => m.saveCharacterRaw(...))`) so the local row mirrors cloud truth without re-enqueueing the sync queue (raw = no mirror per the existing comment at characterStore.ts:145-158). Return `{ ...persistedPrivate, shared: true }`.
  d. If `result.ok === false`: leave the local row at shared=false. Throw `new Error(result.friendlyMessage)`. The renderer's existing error toast surfaces it.

For non-shared characters (character.shared === false) OR defaults (character.is_default === true — though defaults can never reach shared=true per existing guards), KEEP THE EXISTING BEHAVIOR (expandAndSaveCharacter or saveCharacter + skipExpansion).

```typescript

  const character = CharacterSchema.parse(charArg);
  const skipExpansion = optsArg != null && typeof optsArg === 'object'
    && (optsArg as { skipExpansion?: unknown }).skipExpansion === true;

  const willShare = character.shared === true && !character.is_default;

  // Phase 1 — local persist as PRIVATE first (defense against the sync-queue
  // race that could mirror shared=true before moderation runs). Defaults and
  // private chars skip this branch and use the existing fast path.
  if (!willShare) {
    if (skipExpansion) { await saveCharacter(character); return character; }
    return await expandAndSaveCharacter({ character });
  }

  const draftPrivate: Character = { ...character, shared: false };
```
