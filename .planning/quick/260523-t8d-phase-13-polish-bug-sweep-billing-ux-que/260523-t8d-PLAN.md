```typescript
/**
 * Default tokens/minute used for the playtime estimate when the server has
 * not yet supplied a recalibrated value (rolling 24h ledger_consumption avg).
 *
 * HARDCODED CONSTANT — derivation (per ITEM 4):
 *   - Haiku 4.5 chat with the Sei persona prompt (~1400 tokens)
 *   - Average user turn ~20 tokens, assistant turn ~80 tokens, tool overhead ~50 tokens
 *   - At ~20 turns/min cadence: persona context churns plus 20 × (20 + 80 + 50)
 *     ≈ ~850 input+output tokens/min absorbed by the chat ring.
 *   - Bias slightly conservative — better to under-estimate playtime than
 *     promise more than the user will actually see.
 *
 * Why no JSON fixture: the rolling-24h client-side recalibration (proxyClient.creditsGet)
 * self-heals from real usage data within the user's first day. A frozen fixture
 * adds a TBD calibration script and locks the constant to assumptions that drift
 * as the persona prompt evolves.
 */
export const DEFAULT_TOKENS_PER_MIN = 850;

export interface PlaytimeEstimate {
  display: string;  // e.g. "~2h left", "~45min left", "0min left", "Calculating…"
  minutes: number;  // raw computed minutes (for testing / debugging)
}

/**
 * Convert remaining tokens to a teenage-friendly playtime string.
 *
 * Rounding rule (per ITEM 4 constraint):
 *   - undefined inputs → "Calculating…"
 *   - 0 tokens → "0min left"
 *   - < 60min → floor to nearest 15min (minimum granularity)
 *   - >= 60min → floor to nearest hour
 *
 * Aggressive DOWN-rounding so we never overpromise.
 */
export function tokensRemainingToPlaytime(
  remainingTokens: number | undefined,
  tokensPerMin: number | undefined = DEFAULT_TOKENS_PER_MIN,
): PlaytimeEstimate {
  if (remainingTokens === undefined || tokensPerMin === undefined || tokensPerMin <= 0) {
    return { display: 'Calculating…', minutes: 0 };
  }
  if (remainingTokens <= 0) return { display: '0min left', minutes: 0 };
  const rawMin = remainingTokens / tokensPerMin;
  if (rawMin >= 60) {
    const hours = Math.floor(rawMin / 60);
    return { display: `~${hours}h left`, minutes: hours * 60 };
  }
  const min15 = Math.floor(rawMin / 15) * 15;
  if (min15 === 0) {
    // Edge case: <15min remaining. Show '~15min' as the floor floor (user
    // said minimum granularity 15min) — but if literally 0min computed, that
    // would lie. Compromise: show '~15min' only when raw >= 1; else 0min.
    return rawMin >= 1
      ? { display: '~15min left', minutes: 15 }
      : { display: '0min left', minutes: 0 };
  }
  return { display: `~${min15}min left`, minutes: min15 };
}
```
