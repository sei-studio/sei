```sql
create or replace view public.ledger_balance as
select u.id as user_id,
  coalesce((select sum(credits_micro) from ledger_grants g
            where g.user_id = u.id), 0)::bigint
  - coalesce((select sum(c.micro) from ledger_consumption c
              where c.user_id = u.id
                and c.reservation_state in ('reserved','settled')), 0)::bigint
  as balance_micro
from auth.users u
where auth.uid() = u.id or auth.role() = 'service_role';
```
