<objective>
Cluster B remediation — proxy critical security fixes. Implement model whitelist on both forward routes, server-side max_tokens clamp on the free route, refund event handlers with ledger compensation, atomic webhook RPC with 5xx-on-DB-failure, conditional subscription revival guard, secondary dedup key on ledger_grants, drop the /health version leak, and capture orphan webhooks into a recovery table.

Purpose: Closes the highest-severity findings from the proxy audit. Today an attacker (or honest bug) can:
- Request Opus on /v1/messages and be billed at Haiku rates (C1)
- Burn arbitrary Anthropic budget on /free/v1/messages via a non-Haiku model or oversized max_tokens (C2/C3)
- Keep their balance after Lemon Squeezy refunds because no refund event is handled (C4)
- Cause silent grant loss when a webhook partially fails (M4)
- Revive a cancelled subscription by replaying an old payment_success (M5)
- Double-grant when LS resends with a fresh webhook_id (M6)
- Probe the proxy version via /health (M29)
- Lose orders forever when custom_data.user_id is missing (M30)

Output: Four migrations under 20260528000400-700, one rewritten lemon-webhook Edge Function, three proxy code changes (forward.ts, rateLimit/gate.ts, forwardFree.ts), and a one-line /health change in app.ts. Plus the test deltas needed to keep `cd proxy && npm test` green.

```sql
  id              uuid primary key default gen_random_uuid(),
  user_id         uuid not null references auth.users(id) on delete cascade,
  kind            text not null check (kind in ('trial','pack','subscription')),
  credits_micro   bigint not null check (credits_micro >= 0),
  source          text,
  lemon_event_id  text unique,
  granted_at      timestamptz not null default now(),
  expires_at      timestamptz
);

create view public.ledger_balance as
select u.id as user_id,
  coalesce((select sum(credits_micro) from public.ledger_grants g where g.user_id = u.id), 0)::bigint
  - coalesce((select sum(c.micro) from public.ledger_consumption c

  as balance_micro
from auth.users u;
```

```sql
create or replace function public.apply_lemon_event(
  p_event_name text,
  p_event_id   text,
  p_user_id    uuid,
  p_payload    jsonb
) returns void
language plpgsql security definer set search_path = public as $$
declare
  v_lemon_sub_id text        := coalesce(p_payload->'data'->>'id', '');
  v_renews_at    timestamptz := nullif(p_payload->'data'->'attributes'->>'renews_at', '')::timestamptz;
  v_ends_at      timestamptz := nullif(p_payload->'data'->'attributes'->>'ends_at', '')::timestamptz;
  v_status_raw   text;
  v_status       text;
  v_period_start timestamptz;
  v_refunded_eid text;
  v_orig_micro   bigint;
begin
  case p_event_name
    when 'order_created' then
      insert into public.ledger_grants
        (user_id, kind, credits_micro, source, lemon_event_id)
      values
        (p_user_id, 'pack', 4750000, 'lemon_squeezy', p_event_id)
      on conflict (lemon_event_id) do nothing;

    when 'subscription_created' then
      insert into public.subscription_status
        (user_id, status, lemon_subscription_id, renews_at, ends_at)
      values
        (p_user_id, 'active', v_lemon_sub_id, v_renews_at, null)
      on conflict (user_id) do update set
        status = excluded.status,
        lemon_subscription_id = excluded.lemon_subscription_id,
        renews_at = excluded.renews_at,
        ends_at = null;

    when 'subscription_payment_success' then
      -- Derive a stable billing-period anchor; date_trunc to a day keeps it
      -- robust against minor LS clock drift. If renews_at is missing, fall
      -- back to today (less precise but still dedups within 24h).
      v_period_start := coalesce(
        date_trunc('day', v_renews_at - interval '1 month'),
        date_trunc('day', now())
      );

      insert into public.ledger_grants
        (user_id, kind, credits_micro, source, lemon_event_id,
         lemon_subscription_id, billing_period_start)
      values
        (p_user_id, 'subscription', 18500000, 'lemon_squeezy', p_event_id,
         v_lemon_sub_id, v_period_start)
      on conflict (lemon_event_id) do nothing;
      -- Secondary dedup (M6): the partial UNIQUE index from migration 500
      -- raises 23505 on the (user_id, source, lemon_subscription_id,
      -- billing_period_start) collision. Caught by the EXCEPTION block below.

      -- M5: do NOT revive a cancelled/expired/refunded subscription.
      update public.subscription_status
      set    status = 'active',
             lemon_subscription_id = v_lemon_sub_id,
             renews_at = v_renews_at,
             ends_at   = null
      where  user_id = p_user_id
        and  status not in ('cancelled','expired','refunded');
      -- First-ever payment_success (no row yet): insert active.
      insert into public.subscription_status
        (user_id, status, lemon_subscription_id, renews_at, ends_at)
      values
        (p_user_id, 'active', v_lemon_sub_id, v_renews_at, null)
      on conflict (user_id) do nothing;

    when 'subscription_updated' then
      v_status_raw := coalesce(p_payload->'data'->'attributes'->>'status', 'active');
      v_status := case
        when v_status_raw in ('active','cancelled','expired','past_due','refunded')
          then v_status_raw
        else 'past_due'
      end;
      insert into public.subscription_status
        (user_id, status, lemon_subscription_id, renews_at, ends_at)
      values
        (p_user_id, v_status, v_lemon_sub_id, v_renews_at, v_ends_at)
      on conflict (user_id) do update set
        status = excluded.status,
        lemon_subscription_id = excluded.lemon_subscription_id,
        renews_at = excluded.renews_at,
        ends_at = excluded.ends_at;

    when 'subscription_cancelled' then
      insert into public.subscription_status
        (user_id, status, lemon_subscription_id, ends_at)
      values
        (p_user_id, 'cancelled', v_lemon_sub_id, v_ends_at)
      on conflict (user_id) do update set
        status = 'cancelled',
        lemon_subscription_id = excluded.lemon_subscription_id,
        ends_at = excluded.ends_at;

    when 'subscription_expired' then
      insert into public.subscription_status
        (user_id, status, lemon_subscription_id, ends_at)
      values
        (p_user_id, 'expired', v_lemon_sub_id, v_ends_at)
      on conflict (user_id) do update set
        status = 'expired',
        lemon_subscription_id = excluded.lemon_subscription_id,
        ends_at = excluded.ends_at;

    when 'order_refunded' then
      v_refunded_eid := coalesce(
        p_payload->'meta'->>'refunded_event_id',
        p_payload->'data'->'attributes'->>'refunded_event_id'
      );
      select credits_micro into v_orig_micro
        from public.ledger_grants
       where lemon_event_id = v_refunded_eid
         and user_id        = p_user_id
       limit 1;
      if v_orig_micro is null then
        raise exception 'order_refunded: original grant not found for refunded_event_id=%', v_refunded_eid
          using errcode = 'P0002';
      end if;
      insert into public.ledger_grants
        (user_id, kind, credits_micro, source, lemon_event_id, refunded_event_id)
      values
        (p_user_id, 'refund', -v_orig_micro, 'lemon_squeezy_refund', p_event_id, v_refunded_eid)
      on conflict (lemon_event_id) do nothing;

    when 'subscription_payment_refunded' then
      v_refunded_eid := coalesce(
        p_payload->'meta'->>'refunded_event_id',
        p_payload->'data'->'attributes'->>'refunded_event_id'
      );
      select credits_micro into v_orig_micro
        from public.ledger_grants
       where lemon_event_id = v_refunded_eid
         and user_id        = p_user_id
       limit 1;
      if v_orig_micro is null then
        raise exception 'subscription_payment_refunded: original grant not found for refunded_event_id=%', v_refunded_eid
          using errcode = 'P0002';
      end if;
      insert into public.ledger_grants
        (user_id, kind, credits_micro, source, lemon_event_id, refunded_event_id)
      values
```
