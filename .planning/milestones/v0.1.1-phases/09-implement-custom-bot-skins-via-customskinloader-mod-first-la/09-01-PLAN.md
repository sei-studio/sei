    Use Node's built-in `zlib` + a tiny hand-rolled PNG encoder (CRC32 + IDAT) so we do NOT add a runtime image-encoding dep. Pattern:

    ```js
    // scripts/build-default-skins.mjs
    import { writeFileSync, mkdirSync } from 'node:fs';
    import { deflateSync } from 'node:zlib';
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';

    const __dirname = path.dirname(fileURLToPath(import.meta.url));
    const OUT_DIR = path.join(__dirname, '..', 'resources', 'skins');
    mkdirSync(OUT_DIR, { recursive: true });

    const PALETTES = {
      // Pick three distinguishable per-persona color schemes. Layout follows
      // the modern (post-1.8) 64×64 skin format: head 8×8 top-left, body
      // strip below. We fill the WHOLE 64×64 with the body color and stamp
      // a 4×4 highlight square for the head front so each skin has a
      // visible distinct face on the 3D preview.
      sui:       { body: [225, 188, 226, 255], head: [255, 215, 0, 255] },     // peach body, gold head
      mochineko: { body: [120, 100, 90, 255],  head: [255, 245, 230, 255] },   // muted brown body, cream head
      clawd:     { body: [80, 90, 110, 255],   head: [110, 130, 150, 255] },   // slate body, lighter slate head
    };

    function buildPng(rgba32x32Pixels /* Buffer length = 64*64*4 */) { /* PNG encoder: signature + IHDR + IDAT (deflate(rows-with-filter-bytes)) + IEND */ }

    for (const [name, palette] of Object.entries(PALETTES)) {
      const pixels = Buffer.alloc(64 * 64 * 4);
      // 1. fill entire image with body color
      for (let i = 0; i < pixels.length; i += 4) pixels.set(palette.body, i);
      // 2. stamp head FRONT face: skin format places it at x=8..15, y=8..15
      for (let y = 8; y < 16; y++) {
        for (let x = 8; x < 16; x++) {
          const off = (y * 64 + x) * 4;
          pixels.set(palette.head, off);
        }
      }
      writeFileSync(path.join(OUT_DIR, `${name}.png`), buildPng(pixels));
    }
    ```
