    **1A — Create `src/renderer/src/lib/stores/useWizardStore.ts`:**

    Zustand store managing wizard state. Public shape (BLOCKER 2 fix — sessionId state + sei.wizardCancel instead of renderer-local AbortController):

    ```ts
    import { create } from 'zustand';
    import { sei } from '../ipcClient';
    import type { McInstall, WizardInstallResult, WizardProgressEvent, WizardState } from '@shared/ipc';

    export type WizardStep = 'welcome' | 'detecting' | 'none-found' | 'pick' | 'installing' | 'one-failed' | 'done';

    interface WizardStoreState {
      open: boolean;
      step: WizardStep;
      isReentry: boolean;          // true when re-running from Settings (controls Back button on welcome step)
      installs: McInstall[];
      selectedIds: Set<string>;
      sessionId: string | null;    // BLOCKER 2 — non-null while runWizardInstall is in flight; lets cancelInstall target the right run
      // Map<installId, latest WizardProgressEvent so progress bars render the current stage>
      progress: Map<string, WizardProgressEvent>;
      results: WizardInstallResult[];
      error: string | null;
      // imperative actions:
      openWizard: (isReentry: boolean) => void;
      closeWizard: () => void;
      gotoStep: (step: WizardStep) => void;
      runDetection: () => Promise<void>;
      toggleSelected: (id: string) => void;
      runInstall: () => Promise<void>;
      cancelInstall: () => Promise<void>;   // BLOCKER 2 — now async because it fires an IPC call
    }

    export const useWizardStore = create<WizardStoreState>((set, get) => {
      let progressUnsub: (() => void) | null = null;

      return {
        open: false,
        step: 'welcome',
        isReentry: false,
        installs: [],
        selectedIds: new Set(),
        sessionId: null,                                     // BLOCKER 2
        progress: new Map(),
        results: [],
        error: null,

        openWizard: (isReentry) => {
          set({ open: true, isReentry, step: 'welcome', installs: [], selectedIds: new Set(), progress: new Map(), results: [], error: null, sessionId: null });
        },
        closeWizard: () => {
          if (progressUnsub) { progressUnsub(); progressUnsub = null; }
          // BLOCKER 2 — if a wizard run is in flight, abort it via IPC (not via local AbortController).
          const sid = get().sessionId;
          if (sid) {
            // Fire-and-forget; main's handler resolves immediately. The in-flight runWizardInstall
            // promise then rejects via its AbortSignal chain.
            void sei.wizardCancel(sid).catch(() => { /* best-effort */ });
          }
          set({ open: false, sessionId: null });
        },
        gotoStep: (step) => set({ step }),

        runDetection: async () => {
          set({ step: 'detecting', error: null });
          try {
            const { installs } = await sei.detectMcInstalls();
            if (installs.length === 0) {
              set({ installs, step: 'none-found' });
              return;
            }
            // Pre-select all detected installs (CONTEXT idempotency: re-runs preserve sei_enabled flags)
            const selected = new Set(installs.filter(i => i.sei_enabled).map(i => i.id));
            // First-run: select all installs by default so the user just clicks Continue
            if (selected.size === 0) installs.forEach(i => selected.add(i.id));
            set({ installs, selectedIds: selected, step: 'pick' });
          } catch (err) {
            set({ error: (err as Error).message ?? 'Detection failed', step: 'pick' });
          }
        },

        toggleSelected: (id) => {
          const next = new Set(get().selectedIds);
          if (next.has(id)) next.delete(id);
          else next.add(id);
          set({ selectedIds: next });
        },

        runInstall: async () => {
          // BLOCKER 2 — generate a fresh sessionId for this install run. The renderer keeps the
          // id in store state so cancelInstall can pass it back through sei.wizardCancel(sessionId).
          const sessionId = crypto.randomUUID();
          set({ step: 'installing', progress: new Map(), results: [], error: null, sessionId });
          const { baseUrl } = await sei.getSkinServerUrl();
          // Subscribe to push channel BEFORE invoking runWizardInstall so we don't miss the 'queued' event for the first install.
          progressUnsub = sei.onWizardProgress((ev) => {
            const m = new Map(get().progress);
            m.set(ev.installId, ev);
            set({ progress: m });
          });
          try {
            const { results } = await sei.runWizardInstall({
              sessionId,                                       // BLOCKER 2 — send sessionId so main's session Map binds it
              installIds: Array.from(get().selectedIds),
              skinServerBaseUrl: baseUrl,
            });
            set({ results });
            // Did ANY install fail?
            const anyFailed = results.some(r => !r.ok);
            set({ step: anyFailed ? 'one-failed' : 'done' });
          } catch (err) {
            set({ error: (err as Error).message ?? 'Install failed', step: 'one-failed' });
          } finally {
            if (progressUnsub) { progressUnsub(); progressUnsub = null; }
            set({ sessionId: null });                          // BLOCKER 2 — session is over; clear the id
          }
        },

        cancelInstall: async () => {
          // BLOCKER 2 — fire the IPC cancel. Main's handler aborts the matching AbortController,
          // which propagates through Plan 04's signal-aware modules (installFabricLoader, etc.).
          // The runWizardInstall promise then rejects and the finally block above clears sessionId.
          const sid = get().sessionId;
          if (!sid) return;
          try {
            await sei.wizardCancel(sid);
          } catch { /* best-effort — main may already have cleaned up the session */ }
          // Don't call closeWizard — user might want to retry. Stay on installing step;
          // results will reflect partial progress when runWizardInstall settles via its catch path.
        },
      };
    });
    ```

| T-09-T8 | T (Tampering) | useWizardStore selectedIds | accept | Selected ids are validated by main's z.array(z.string()) parse before runWizardInstall executes. Renderer cannot inject paths outside scanned installs. |
