    ```ts
    /**
     * Locate the Java runtime that Minecraft bundles inside its game directory.
     * Returns the absolute path to a runnable `java` / `javaw` executable, or null if not present.
     * Probed BEFORE system PATH per BLOCKER 3 — Minecraft installs its own JRE under
     * `<mcDir>/runtime/java-runtime-gamma/<platform-tag>/...`, and we should use that
     * to honor the "zero manual config" goal.
     */
    export async function findBundledJava(mcInstall: McInstall): Promise<string | null> {
      const plat = process.platform;
      const arch = process.arch;  // 'x64' | 'arm64' (mac/win); 'x64' on Linux
      const mcDir = mcInstall.path;
      const runtimeRoot = path.join(mcDir, 'runtime', 'java-runtime-gamma');

      // Each platform/arch pair maps to a single canonical relative path inside runtimeRoot.
      // Names match what the Mojang launcher writes when installing the bundled JRE.
      let relPath: string;
      if (plat === 'darwin') {
        const archTag = arch === 'arm64' ? 'mac-os-arm64' : 'mac-os';
        relPath = path.join(archTag, 'java-runtime-gamma', 'jre.bundle', 'Contents', 'Home', 'bin', 'java');
      } else if (plat === 'win32') {
        const archTag = arch === 'arm64' ? 'windows-arm64' : 'windows-x64';
        // Fabric installer is a CLI JAR — prefer `javaw` (no console window pop) per Windows convention,
        // but fall back to `java` if javaw isn't present.
        const tryFirst = path.join(runtimeRoot, archTag, 'java-runtime-gamma', 'bin', 'javaw.exe');
        try { await fs.access(tryFirst, fs.constants.X_OK); return tryFirst; } catch { /* fall through */ }
        relPath = path.join(archTag, 'java-runtime-gamma', 'bin', 'java.exe');
      } else if (plat === 'linux') {
        relPath = path.join('linux', 'java-runtime-gamma', 'bin', 'java');
      } else {
        return null;  // unsupported OS
      }

      const abs = path.join(runtimeRoot, relPath);
      try {
        await fs.access(abs, fs.constants.X_OK);
        return abs;
      } catch {
        return null;  // not installed (or wrong permissions)
      }
    }
    ```

    Notes for BLOCKER 3:
    - The `runtime/java-runtime-gamma/<platform-tag>/...` layout is what Mojang's launcher writes when the user has launched a vanilla profile at least once. If the user has never launched MC (e.g. fresh install of just the launcher), the bundled Java may be absent.
    - `fs.access(p, fs.constants.X_OK)` checks executable bit on macOS/Linux (Windows X_OK is effectively a presence check — but that's fine; if the file exists we'll attempt to exec it).
    - The fabricInstaller orchestrates the bundled-first → PATH-fallback flow in Task 2. This module only exposes the probe; it doesn't pick.
