    ```js
    // Post-process say() text per D-7: lowercase, strip terminals/em-dashes/quotes/backticks,
    // KEEP apostrophes (contractions), collapse whitespace, cap at 256 chars.
    // Internal `text` (think) is exempt and never passes through this.
    export function postProcessSay(s) {
      return String(s ?? '')
        .toLowerCase()
        .replace(/[.,!?;:—–"`]/g, '')
        .replace(/\s+/g, ' ')
        .trim()
        .slice(0, 256)
    }
    ```

    - `:` colon

    - `"` ASCII double-quote
