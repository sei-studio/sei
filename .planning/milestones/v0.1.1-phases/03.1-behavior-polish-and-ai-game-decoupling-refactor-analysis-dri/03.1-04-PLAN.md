    Step C — Create src/brain/memory/affectLog.js:

    ```js
    import fs from 'node:fs/promises'
    import path from 'node:path'
    import { atomicWrite } from '../storage/atomicWrite.js'

    const HEADER = '# Affect Log\n\nAppend-only record of moments worth remembering across sessions.\nFormat: `- [ISO timestamp] (kind) summary`\n\n'

    export function createAffectLog({ path: filePath }) {
      return {
        path: filePath,
        async append(entry) { return appendAffect(filePath, entry) },
        async readAll()   { return readAffectFull(filePath) },
      }
    }

    export async function appendAffect(filePath, { kind, summary, when }) {
      const t = (when instanceof Date ? when : new Date()).toISOString()
      const safeKind = String(kind ?? 'moment')
      const safeSummary = String(summary ?? '').replace(/\n+/g, ' ').trim()
      const line = `- [${t}] (${safeKind}) ${safeSummary}\n`

      let existing = ''
      try { existing = await fs.readFile(filePath, 'utf8') }
      catch (e) { if (e.code !== 'ENOENT') throw e; existing = HEADER }
      if (!existing.startsWith('# Affect Log')) existing = HEADER + existing
      await atomicWrite(filePath, existing + line)
    }

    export async function readAffectFull(filePath) {
      try { return await fs.readFile(filePath, 'utf8') }
      catch (e) {
        if (e.code !== 'ENOENT') throw e
        await atomicWrite(filePath, HEADER)
        return HEADER
      }
    }
    ```

    export function loopHasAffect(messages) {
      // Walk the loop's messages array; return true if any tool_use with name 'noteToSelf'
      // appears, OR if any owner-chat message text matches affect-keyword regex.
      for (const m of messages) {
        if (m.role === 'assistant' && Array.isArray(m.content)) {
          for (const block of m.content) {
            if (block?.type === 'tool_use' && block.name === 'noteToSelf') return true
          }
        }
        if (m.role === 'user' && Array.isArray(m.content)) {
          for (const block of m.content) {
            if (block?.type === 'text' && typeof block.text === 'string') {
              if (/\b(good job|thanks|thank you|nice work|love|appreciate|sorry|please|name is|call me|i am)\b/i.test(block.text)) return true
            }
          }
        }
      }
      return false
    }
