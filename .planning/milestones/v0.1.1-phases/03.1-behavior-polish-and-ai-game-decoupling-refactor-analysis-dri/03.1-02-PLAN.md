    ```js
    import { createDefaultRegistry } from './registry.js'
    import { createSnapshotComposer } from './observers/snapshot.js'
    import { worldPrimer } from './primer.js'
    import { setInflightProvider } from './behaviors/follow.js'
    import { closeContainerSession } from './behaviors/container.js'
    import { wireBotEvents } from './fsmWires.js'

    export function createMinecraftAdapter({ bot, config }) {
      const registry = createDefaultRegistry(bot)
      let attachedHandlers = null

      return {
        // Action surface
        listActions: () => registry.list(),
        getActionSchema: (name) => registry.getSchema(name),
        getActionDescription: (name) => registry.getDescription(name),
        executeAction: (name, args, ctx) => registry.execute(name, args, bot, ctx),

        // World perception
        createSnapshotComposer: () => createSnapshotComposer({ bot, config }),
        worldPrimer,

        // Session lifecycle
        attach(handlers) {
          attachedHandlers = handlers
          wireBotEvents(bot, handlers, { config })
        },

        // Effects
        chat: (text) => { try { bot.chat(text) } catch {} },
        setInflightProvider,
        closeAnySessions: async () => { try { await closeContainerSession(bot) } catch {} },

        // Capabilities (read from registry / plugin presence)
        get supportsAutoEat() { return Boolean(bot.autoEat) },
        get supportsFollow() { return registry.list().includes('follow') },

        // Identity
        get botUsername() { return bot.username },
        getKnownPlayers: () => bot.players ?? {},
      }
    }
    ```
