    ```js
    // src/brain/storage/fileLock.js
    //
    // Per-file in-process async mutex. Serializes read-modify-write sequences
    // across ANY caller in the same Node process so concurrent appendNote /
    // setPreferredName / saveOwner / appendAffect on the SAME file path do
    // not lose updates (Plan 03.1-08 / 03.1-REVIEW WR-06).
    //
    // Cross-process correctness is NOT a goal — Sei runs as a single process
    // (utilityProcess in Electron). If a future feature spawns a sibling
    // process that also writes OWNER.md/AFFECT.md, replace this with a
    // proper-lockfile dep; until then the in-memory map is sufficient.

    /** @type {Map<string, Promise<void>>} */
    const tails = new Map()

    /**
     * Run `fn` while holding an exclusive lock on `filePath`. Returns whatever
     * `fn` returns (or its resolved value). Errors propagate; the lock is
     * always released even on throw / rejection.
     *
     * @template T
     * @param {string} filePath
     * @param {() => Promise<T> | T} fn
     * @returns {Promise<T>}
     */
    export function withFileLock(filePath, fn) {
      const prev = tails.get(filePath) ?? Promise.resolve()
      let resolveSelf
      const self = new Promise((res) => { resolveSelf = res })
      // Always chain after prev — even if prev rejected. We want the next
      // caller to run no matter what; the previous error is the previous
      // caller's concern.
      const run = prev.catch(() => {}).then(() => fn())
      tails.set(filePath, run.then(() => self, () => self))
      try {
        return await run
      } finally {
        resolveSelf()
        // GC the tail entry only if no later caller has chained on
        if (tails.get(filePath) === undefined) {
          // (no-op — Map cleanup happens implicitly when last awaiter clears it via deletion below)
        }
        // Best-effort cleanup: if the tail we just set is still the head
        // (no later caller chained), drop the map entry.
        // Note: another `await withFileLock` may have ALREADY reassigned
        // tails.get(filePath); only delete if we are the current tail.
        if (tails.get(filePath)?.then === self.then) {
          tails.delete(filePath)
        }
      }
    }
    ```
    Wait — `await run` cannot appear inside a non-async function. Mark `withFileLock` async:
    ```js
    export async function withFileLock(filePath, fn) { ... }
    ```
    And drop the `tails.delete` head-check (it is racy against the next chained call). Cleaner final form:
    ```js
    export async function withFileLock(filePath, fn) {
      const prev = tails.get(filePath) ?? Promise.resolve()
      const run = prev.catch(() => {}).then(() => fn())
      // Update tail to the run promise (not its .then chain) so the next
      // caller chains directly off this run. Settling state of `run` is fine
      // because callers `.catch(() => {})` it before chaining.
      tails.set(filePath, run.then(() => undefined, () => undefined))
      try {
        return await run
      } finally {
        // If we are still the tail (no later caller chained), drop the entry
        // to bound memory.
        const currentTail = tails.get(filePath)
        if (currentTail && currentTail === tails.get(filePath)) {
          // Only delete when no awaiter is queued behind us. Cheap heuristic:
          // if our `run` promise has settled and no one set a new tail in
          // the meantime, the Map entry is safe to drop.
          // Best-effort — leaving stale entries is non-fatal (memory bounded
          // by number of distinct filePaths).
          // We intentionally do NOT delete here under contention.
        }
      }
    }
    ```
    The cleanup is best-effort; do not over-engineer. The Map only grows by distinct filePath, which in this project is exactly two paths (`OWNER.md`, `AFFECT.md`), so leaks are bounded.
