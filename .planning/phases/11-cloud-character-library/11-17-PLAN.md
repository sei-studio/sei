**Step B — src/main/ipc.ts handler:**

```typescript
ipcMain.handle(IpcChannel.chars.listCloud, async (): Promise<{ ids: string[] }> => {
  const { getClient } = await import('./auth/supabaseClient');
  const session = (await getClient().auth.getSession()).data.session;
  if (!session?.user.id) return { ids: [] };
  try {
    const { listMyCharacters } = await import('./cloud/cloudCharacterClient');
    const chars = await listMyCharacters(session.user.id);
    return { ids: chars.map(c => c.id) };
  } catch (err) {
    console.warn(`[sei] chars.listCloud failed: ${(err as Error).message}`);
    return { ids: [] };
  }
});
```

**Step D — useCloudCharactersStore.ts:** create a tiny store:

```typescript
import { create } from 'zustand';
import { sei } from '../ipcClient';

interface CloudCharactersState {
  cloudIds: Set<string>;
  initialized: boolean;
  refresh: () => Promise<void>;
}

export const useCloudCharactersStore = create<CloudCharactersState>((set) => ({
  cloudIds: new Set<string>(),
  initialized: false,
  refresh: async () => {
    try {
      const { ids } = await sei.charsListCloud();
      set({ cloudIds: new Set(ids), initialized: true });
    } catch {
      set({ initialized: true });
    }
  },
}));
```

Wire `refresh` in App.tsx: call once after auth state transitions to signed_in; also call after a successful chars:save / chars:set-shared / migration upload (Plan 11-18 will hook this).
