import { randomUUID } from 'node:crypto';
import { readdir, rename, unlink, readFile, writeFile, mkdir, stat } from 'node:fs/promises';
import { DEFAULT_CHARACTER_UUIDS } from './defaultCharacters';
// path, paths, atomicWrite, withFileLock should already be imported from runFirstLaunchMigration

const UUID_V4_RE = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i;
const DATA_URL_RE = /^data:image\/(png|jpe?g|webp);base64,/i;

async function fileExists(p: string): Promise<boolean> {
  try { await stat(p); return true; } catch { return false; }
}

/**
 * Phase 11 D-23 — Slug→UUID one-shot rename migration.
 *
 * Source: 11-RESEARCH §Pattern 6 (full step sequence) + §Pitfall 7 (boot order).
 *
 * Idempotent via paths.migrationManifestPath() — existence of the manifest
 * file means migration has completed for this install.
 *
 * Runs PURELY LOCAL — no Supabase code paths (grep gate: zero
 * `supabase.from` or `getClient()` calls within this function).
 *
 * Must run BEFORE botSupervisor is wired (Pitfall 7).
 */
export async function runUuidRenameMigration(): Promise<void> {
  const logger = {
    info: (m: string) => console.log(`[sei] uuid-rename: ${m}`),
    warn: (m: string) => console.warn(`[sei] uuid-rename: ${m}`),
  };

  // Idempotency gate
  if (await fileExists(paths.migrationManifestPath())) {
    return;
  }

  const indexPath = paths.characterIndexPath?.() ?? path.join(paths.userData(), 'characters', 'index.json');
  // ^ If the existing characterStore exposes a getter, use it; else use the same literal path it uses.

  let indexData: { version: number; order: string[] };
  try {
    indexData = JSON.parse(await readFile(indexPath, 'utf8'));
  } catch (err) {
    if ((err as NodeJS.ErrnoException).code === 'ENOENT') {
      // Fresh install, no chars yet — write manifest and return.
      logger.info('no characters/index.json found — fresh install, marking manifest');
      await writeManifest({ migratedAt: new Date().toISOString(), entries: [] });
      return;
    }
    throw err;
  }

  const manifest: Array<{ oldSlug: string; newId: string }> = [];

  for (const oldId of indexData.order) {
    if (UUID_V4_RE.test(oldId)) {
      // Already migrated — skip
      manifest.push({ oldSlug: oldId, newId: oldId });
      continue;
    }

    // Resolve target UUID: defaults get frozen UUID, user-created get fresh randomUUID
    let newId: string;
    if (oldId === 'sui' || oldId === 'lyra' || oldId === 'clawd') {
      newId = DEFAULT_CHARACTER_UUIDS[oldId];
    } else {
      newId = randomUUID();
    }

    // Read old JSON
    const oldJsonPath = path.join(paths.userData(), 'characters', `${oldId}.json`);
    const newJsonPath = path.join(paths.userData(), 'characters', `${newId}.json`);

    let charData: Record<string, unknown>;
    try {
      charData = JSON.parse(await readFile(oldJsonPath, 'utf8')) as Record<string, unknown>;
    } catch (err) {
      logger.warn(`skipping ${oldId}: failed to read ${oldJsonPath}: ${(err as Error).message}`);
      continue;
    }

    // Rewrite fields
    charData.id = newId;
    charData.slug = oldId;
    if (charData.shared === undefined) charData.shared = true;
    if (charData.metadata === undefined) charData.metadata = {};

    // Portrait migration: decode data URL → write file → replace field
    const portrait = typeof charData.portrait_image === 'string' ? charData.portrait_image : null;
    if (portrait && DATA_URL_RE.test(portrait)) {
      const commaIdx = portrait.indexOf(',');
      const b64 = portrait.slice(commaIdx + 1);
      const bytes = Buffer.from(b64, 'base64');
      await mkdir(paths.portraitsDir(), { recursive: true });
      await withFileLock(paths.portraitPath(newId), async () => {
        await atomicWrite(paths.portraitPath(newId), bytes);
      });
      charData.portrait_image = `${newId}.png`; // local-cache filename; Plan 11-06 resolves to local file:// or Storage URL
    }

    // Write new JSON
    await mkdir(path.dirname(newJsonPath), { recursive: true });
    await withFileLock(newJsonPath, async () => {
      await atomicWrite(newJsonPath, JSON.stringify(charData, null, 2) + '\n');
    });

    // Rename skin PNG
    const oldSkinPath = path.join(paths.userData(), 'skins', `${oldId}.png`);
    const newSkinPath = path.join(paths.userData(), 'skins', `${newId}.png`);
    if (await fileExists(oldSkinPath)) {
      await mkdir(path.dirname(newSkinPath), { recursive: true });
      await rename(oldSkinPath, newSkinPath);
    }

    // Rename memory dir (LIB-02: memory keys must follow rename but content stays local forever)
    const oldMemoryDir = path.join(paths.userData(), 'memory', oldId);
    const newMemoryDir = path.join(paths.userData(), 'memory', newId);
    if (await fileExists(oldMemoryDir)) {
      await rename(oldMemoryDir, newMemoryDir);
    }

    // Delete old JSON (after new JSON committed)
    try { await unlink(oldJsonPath); } catch { /* swallow ENOENT */ }

    manifest.push({ oldSlug: oldId, newId });
    logger.info(`renamed ${oldId} → ${newId}`);
  }

  // Update index.json order to UUIDs
  const newOrder = manifest.map(e => e.newId);
  await withFileLock(indexPath, async () => {
    await atomicWrite(indexPath, JSON.stringify({ version: indexData.version, order: newOrder }, null, 2) + '\n');
  });

  // Update defaults-seeded.json if present
  const defaultsSeededPath = path.join(paths.userData(), 'defaults-seeded.json');
  if (await fileExists(defaultsSeededPath)) {
    try {
      const seeded = JSON.parse(await readFile(defaultsSeededPath, 'utf8')) as { ids?: string[] };
      const remapped = (seeded.ids ?? []).map(id => {
        const m = manifest.find(e => e.oldSlug === id);
        return m ? m.newId : id;
      });
      await withFileLock(defaultsSeededPath, async () => {
        await atomicWrite(defaultsSeededPath, JSON.stringify({ ids: remapped }, null, 2) + '\n');
      });
    } catch (err) {
      logger.warn(`defaults-seeded rewrite failed: ${(err as Error).message}`);
    }
  }

  // Manifest write — LAST step (idempotency gate)
  await writeManifest({ migratedAt: new Date().toISOString(), entries: manifest });
  logger.info(`completed: ${manifest.length} entries`);
}

async function writeManifest(data: { migratedAt: string; entries: Array<{ oldSlug: string; newId: string }> }): Promise<void> {
  await mkdir(path.dirname(paths.migrationManifestPath()), { recursive: true });
  await withFileLock(paths.migrationManifestPath(), async () => {
    await atomicWrite(paths.migrationManifestPath(), JSON.stringify(data, null, 2) + '\n');
  });
}
