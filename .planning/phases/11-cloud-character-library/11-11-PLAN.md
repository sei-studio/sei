Replace the existing synchronous `buildExport` with the async version:

```typescript
import type { Session } from '@supabase/supabase-js';
import { getClient } from './supabaseClient';

export interface SeiExportV1 {
  schemaVersion: 1;
  exportedAt: string;
  account: {
    email: string;
    createdAt: string;
  };
  characters: unknown[];   // Phase 11: filled by buildExport
  sharing: unknown[];      // Phase 12: still empty here
}

const TIMEOUT_MS = 15_000;

/**
 * Phase 11 — fills characters[] via supabase.from('characters').select('*')
 * .eq('owner', session.user.id). RLS allows the user to read their own
 * rows; the result mirrors the DB row shape (snake_case columns).
 *
 * Source: 10-CONTEXT D-14 (schema reserves characters[]); 11-RESEARCH §Export.
 *
 * Error: throws Error('CLOUD_LIST_FAILED: <msg>') on supabase failure;
 *        authHandlers.exportData maps this to its existing
 *        { ok: false, code: 'write_failed' } envelope.
 */
export async function buildExport(session: Session): Promise<SeiExportV1> {
  const supabase = getClient();
  const controller = new AbortController();
  const handle = setTimeout(() => controller.abort(), TIMEOUT_MS);
  let characters: unknown[] = [];
  try {
    const { data, error } = await supabase
      .from('characters')
      .select('*')
      .eq('owner', session.user.id)
      .abortSignal(controller.signal);
    if (error) throw new Error(`CLOUD_LIST_FAILED: ${error.message}`);
    characters = data ?? [];
  } finally {
    clearTimeout(handle);
  }

  return {
    schemaVersion: 1,
    exportedAt: new Date().toISOString(),
    account: {
      email: session.user.email ?? '',
      createdAt: session.user.created_at,
    },
    characters,
    sharing: [],
  };
}
```
