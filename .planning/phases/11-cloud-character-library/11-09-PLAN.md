**Step C — Push status updates over sync:status:update**

In `src/main/ipc.ts` near the end of `registerIpc()` (or wherever the registration completes), wire the subscriber:

```typescript
// Phase 11 — push sync status updates to all renderer windows whenever queue mutates.
(async () => {
  const { subscribeStatusChange, getStatus } = await import('./cloud/syncQueue');
  subscribeStatusChange(async () => {
    try {
      const { BrowserWindow } = await import('electron');
      const status = await getStatus();
      for (const w of BrowserWindow.getAllWindows()) {
        w.webContents.send(IpcChannel.sync.statusUpdate, {
          pending: status.pending,
          pendingByUuid: status.pendingByUuid,
        });
      }
    } catch (err) {
      console.warn(`[sei] sync status broadcast failed: ${(err as Error).message}`);
    }
  });
})();
```
