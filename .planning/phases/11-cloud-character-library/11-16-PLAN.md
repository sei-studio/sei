```typescript
/**
 * Phase 11 — renderer-side sync queue status.
 *
 * Source: 11-CONTEXT §sync pill + Plan 11-09 IPC surface.
 *
 * Boot wiring (App.tsx):
 *   useSyncStore.getState().init(); // fetches initial status + subscribes
 */

import { create } from 'zustand';
import { sei } from '../ipcClient';

interface SyncState {
  pending: number;
  pendingByUuid: Record<string, 'syncing' | 'failed'>;
  initialized: boolean;
  unsubscribe?: () => void;
}

interface SyncActions {
  init: () => Promise<void>;
  refresh: () => Promise<void>;
  retry: (uuid: string) => Promise<void>;
  getStatus: (uuid: string) => 'syncing' | 'failed' | undefined;
}

export const useSyncStore = create<SyncState & SyncActions>((set, get) => ({
  pending: 0,
  pendingByUuid: {},
  initialized: false,

  init: async () => {
    if (get().initialized) return;
    const unsubscribe = sei.onSyncStatusUpdate((status) => {
      set({ pending: status.pending, pendingByUuid: status.pendingByUuid });
    });
    set({ unsubscribe, initialized: true });
    await get().refresh();
  },

  refresh: async () => {
    try {
      const s = await sei.syncStatus();
      set({ pending: s.pending, pendingByUuid: s.pendingByUuid });
    } catch {
      // swallow — pill defaults to no-pill
    }
  },

  retry: async (uuid: string) => {
    await sei.syncRetry(uuid);
    await get().refresh();
  },

  getStatus: (uuid: string) => get().pendingByUuid[uuid],
}));
```
