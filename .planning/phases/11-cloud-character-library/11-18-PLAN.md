```typescript
ipcMain.handle(IpcChannel.migration.listLocal, async (): Promise<{ characters: Array<{ id: string; name: string; slug: string | null; created: string }> }> => {
  const { listCharacters } = await import('./characterStore'); // adapt name to existing export
  const all = await listCharacters();
  // LOCAL ONLY = signed_in + !is_default + not in cloud. Caller is signed_in by definition (the modal only mounts then).
  const { getClient } = await import('./auth/supabaseClient');
  const session = (await getClient().auth.getSession()).data.session;
  if (!session?.user.id) return { characters: [] };
  const { listMyCharacters } = await import('./cloud/cloudCharacterClient');
  let cloudIds: Set<string>;
  try {
    const cloud = await listMyCharacters(session.user.id);
    cloudIds = new Set(cloud.map(c => c.id));
  } catch {
    cloudIds = new Set();
  }
  const localOnly = all
    .filter(c => !c.is_default && !cloudIds.has(c.id))
    .map(c => ({ id: c.id, name: c.name, slug: c.slug, created: c.created }));
  return { characters: localOnly };
});

ipcMain.handle(IpcChannel.migration.upload, async (_event, uuidsArg: unknown): Promise<{ results: Array<{ id: string; ok: boolean; message?: string }> }> => {
  const uuids = z.array(IdSchema).parse(uuidsArg);

  // ToS gate
  const { isCloudWriteAllowed } = await import('./auth/authState');
  if (!(await isCloudWriteAllowed())) {
    return { results: uuids.map(id => ({ id, ok: false, message: 'Cloud writes not allowed yet (verify email and accept Terms first).' })) };
  }
  const { getClient } = await import('./auth/supabaseClient');
  const session = (await getClient().auth.getSession()).data.session;
  if (!session?.user.id) return { results: uuids.map(id => ({ id, ok: false, message: 'Not signed in.' })) };
  const owner = session.user.id;

  const { getCharacter } = await import('./characterStore');
  const { resolveSkinPng } = await import('./skinStore');
  const { upsertCharacter, uploadSkin, uploadPortrait } = await import('./cloud/cloudCharacterClient');
  const { readFile } = await import('node:fs/promises');
  const { paths } = await import('./paths');

  const results: Array<{ id: string; ok: boolean; message?: string }> = [];
  for (const id of uuids) {
    try {
      const char = await getCharacter(id);
      if (!char) { results.push({ id, ok: false, message: 'Character not found.' }); continue; }
      if (char.is_default) { results.push({ id, ok: false, message: 'Defaults cannot be uploaded.' }); continue; }
      await upsertCharacter(char, owner);
      try {
        const skin = await resolveSkinPng(id);
        if (skin) await uploadSkin(owner, id, skin);
      } catch (e) {
        console.warn(`[sei] migration: skin upload for ${id} failed: ${(e as Error).message}`);
      }
      try {
        const portrait = await readFile(paths.portraitPath(id));
        if (portrait) await uploadPortrait(owner, id, portrait, 'png');
      } catch { /* no portrait file or ENOENT — fine */ }
      results.push({ id, ok: true });
    } catch (err) {
      results.push({ id, ok: false, message: (err as Error).message });
    }
  }
  return { results };
});

ipcMain.handle(IpcChannel.migration.shown, async (_event, actionArg: unknown): Promise<{ shown: boolean }> => {
  const action = z.enum(['get', 'set']).parse(actionArg);
  const { paths } = await import('./paths');
  const { existsSync, mkdirSync, writeFileSync } = await import('node:fs');
  const p = paths.migrationModalShownPath();
  if (action === 'set') {
    try {
      mkdirSync(require('node:path').dirname(p), { recursive: true });
      writeFileSync(p, JSON.stringify({ shown: true, shownAt: new Date().toISOString() }, null, 2) + '\n');
    } catch (err) {
      console.warn(`[sei] migration.shown set failed: ${(err as Error).message}`);
    }
    return { shown: true };
  }
  return { shown: existsSync(p) };
});
```

**Step A — MigrateLocalCharsModal.tsx:**

```typescript
import React, { useEffect, useState } from 'react';
import { sei } from '../lib/ipcClient';
import { Button } from './Button';
import { useCloudCharactersStore } from '../lib/stores/useCloudCharactersStore';
import styles from './MigrateLocalCharsModal.module.css';

export interface MigrateLocalCharsModalProps {
  onClose: () => void;
}

type Char = { id: string; name: string; slug: string | null; created: string };

export function MigrateLocalCharsModal({ onClose }: MigrateLocalCharsModalProps): JSX.Element {
  const [chars, setChars] = useState<Char[]>([]);
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [phase, setPhase] = useState<'idle' | 'submitting' | 'results'>('idle');
  const [results, setResults] = useState<Array<{ id: string; ok: boolean; message?: string }>>([]);

  useEffect(() => {
    void (async () => {
      const r = await sei.migrationListLocal();
      setChars(r.characters);
      setSelected(new Set(r.characters.map(c => c.id))); // default: all selected
    })();
  }, []);

  const handleUpload = async (): Promise<void> => {
    setPhase('submitting');
    const r = await sei.migrationUpload(Array.from(selected));
    setResults(r.results);
    setPhase('results');
    void useCloudCharactersStore.getState().refresh();
  };

  const handleDismiss = (): void => {
    void sei.migrationShown('set'); // suppress auto-mount
    onClose();
  };

  return (
    <div className={styles.scrim}>
      <div className={styles.modal} role="dialog" aria-modal="true">
        <h2 className={styles.title}>Upload local characters?</h2>
        <p className={styles.body}>
          These characters are saved on this machine only. Upload any to your cloud library to access them from other devices.
        </p>
        {phase === 'idle' && (
          <>
            <ul className={styles.list}>
              {chars.map(c => (
                <li key={c.id}>
                  <label>
                    <input
                      type="checkbox"
                      checked={selected.has(c.id)}
                      onChange={e => {
                        const next = new Set(selected);
                        if (e.target.checked) next.add(c.id); else next.delete(c.id);
                        setSelected(next);
                      }}
                    />
                    <span>{c.name}</span>
                  </label>
                </li>
              ))}
              {chars.length === 0 && <li className={styles.empty}>No local-only characters.</li>}
            </ul>
            <div className={styles.actions}>
              <Button kind="secondary" onClick={handleDismiss}>Maybe later</Button>
              <Button kind="primary" onClick={handleUpload} disabled={selected.size === 0}>
                Upload selected
              </Button>
            </div>
          </>
        )}
        {phase === 'submitting' && <p>Uploading…</p>}
        {phase === 'results' && (
          <>
            <ul className={styles.list}>
              {results.map(r => (
                <li key={r.id} className={r.ok ? styles.resultOk : styles.resultFail}>
                  {r.ok ? '✓' : '✗'} {chars.find(c => c.id === r.id)?.name ?? r.id}
                  {!r.ok && r.message && <span className={styles.resultMessage}> — {r.message}</span>}
                </li>
              ))}
            </ul>
            <div className={styles.actions}>
              <Button kind="primary" onClick={() => { void sei.migrationShown('set'); onClose(); }}>Done</Button>
            </div>
          </>
        )}
      </div>
    </div>
  );
}
```
