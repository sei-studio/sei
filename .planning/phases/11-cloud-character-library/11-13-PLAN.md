Create `src/renderer/src/components/AcceptToSModal.tsx`:

```typescript
import React, { useEffect, useState } from 'react';
import { sei } from '../lib/ipcClient';
import { Button } from './Button';
import styles from './AcceptToSModal.module.css';

export interface AcceptToSModalProps {
  onAccepted: () => void;
}

export function AcceptToSModal({ onAccepted }: AcceptToSModalProps): JSX.Element {
  const [checked, setChecked] = useState(false);
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // ESC suppression — this is a BLOCKING modal (D-26)
  useEffect(() => {
    const onKey = (e: KeyboardEvent): void => {
      if (e.key === 'Escape') {
        // No dismissal — blocking modal
        e.preventDefault();
      }
    };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, []);

  const handleSubmit = async (): Promise<void> => {
    setSubmitting(true);
    setError(null);
    try {
      const result = await sei.tosAccept();
      if (result.ok) {
        onAccepted();
      } else {
        setError(result.message);
        setSubmitting(false);
      }
    } catch (err) {
      setError((err as Error).message);
      setSubmitting(false);
    }
  };

  const openTerms = (): void => { void sei.openExternal('https://sei.gg/terms.html'); };
  const openPrivacy = (): void => { void sei.openExternal('https://sei.gg/privacy.html'); };

  return (
    <div className={styles.scrim}>
      <div className={styles.modal} role="dialog" aria-modal="true" aria-labelledby="tos-title">
        <h2 id="tos-title" className={styles.title}>Review Sei's Terms</h2>
        <p className={styles.body}>
          We have published a Privacy Policy and Terms of Service. Please review and accept to continue.
        </p>
        <div className={styles.linkRow}>
          <Button kind="secondary" onClick={openTerms}>Open Terms of Service</Button>
          <Button kind="secondary" onClick={openPrivacy}>Open Privacy Policy</Button>
        </div>
        <label className={styles.checkbox}>
          <input
            type="checkbox"
            checked={checked}
            onChange={e => setChecked(e.target.checked)}
            disabled={submitting}
          />
          <span>I have read and agree to both</span>
        </label>
        {error && <div className={styles.error}>{error}</div>}
        <div className={styles.actions}>
          <Button kind="primary" onClick={handleSubmit} disabled={!checked || submitting}>
            {submitting ? 'Accepting…' : 'Accept and continue'}
          </Button>
        </div>
      </div>
    </div>
  );
}
```
