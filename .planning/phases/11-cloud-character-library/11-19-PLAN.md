/**
 * Phase 11 — Cache-on-demand sync (D-19).
 *
 * On a new machine, the Characters page lists cloud rows. When the user
 * opens a character we don't yet have locally, fetch it + skin + portrait
 * once and write to <userData>/. Subsequent opens are offline-safe.
 *
 * Source: 11-CONTEXT D-19 + 11-RESEARCH §Pitfall 5 (conflict shadow).
 *
 * MAIN PROCESS ONLY.
 */

import { existsSync, mkdir } from 'node:fs';
import { mkdir as mkdirP, writeFile } from 'node:fs/promises';
import path from 'node:path';
import { atomicWrite } from '../../bot/brain/storage/atomicWrite.js';
import { withFileLock } from '../../bot/brain/storage/fileLock.js';
import { paths } from '../paths';

export async function ensureLocallyCached(uuid: string): Promise<void> {
  const localJson = paths.characterPath(uuid);
  const localExists = existsSync(localJson);

  // Conflict detection: if local exists AND there's a pending sync op for this uuid → shadow
  if (localExists) {
    const { getStatus } = await import('./syncQueue');
    const s = await getStatus();
    if (s.pendingByUuid[uuid]) {
      // Local has unflushed pending writes — preserve them. Write the incoming
      // cloud row to <uuid>.json.conflict as a shadow file for inspection.
      try {
        const { downloadCharacter } = await import('./cloudCharacterClient');
        const cloud = await downloadCharacter(uuid);
        if (cloud) {
          const conflictPath = `${localJson}.conflict`;
          await mkdirP(path.dirname(conflictPath), { recursive: true });
          await withFileLock(conflictPath, async () => {
            await atomicWrite(conflictPath, JSON.stringify(cloud, null, 2) + '\n');
          });
          console.warn(`[sei] cache-on-demand: pending local write for ${uuid}; cloud version written to ${conflictPath}.conflict`);
        }
      } catch (err) {
        console.warn(`[sei] cache-on-demand conflict shadow failed for ${uuid}: ${(err as Error).message}`);
      }
      return;
    }
    // No conflict — local cache is good, return.
    return;
  }

  // Cache miss — fetch from cloud
  const { getClient } = await import('../auth/supabaseClient');
  const session = (await getClient().auth.getSession()).data.session;
  if (!session?.user.id) throw new Error('CLOUD_DOWNLOAD_FAILED: not signed in');

  const { downloadCharacter, downloadSkin, downloadPortrait } = await import('./cloudCharacterClient');
  const cloud = await downloadCharacter(uuid);
  if (!cloud) throw new Error('CLOUD_CHARACTER_NOT_FOUND');

  // Write character JSON (raw — bypass cloud mirror)
  const { saveCharacterRaw } = await import('../characterStore');
  await saveCharacterRaw(cloud);

  // Skin
  try {
    const skin = await downloadSkin(session.user.id, uuid);
    if (skin) {
      const skinTarget = paths.skinPngPath(uuid);
      await mkdirP(path.dirname(skinTarget), { recursive: true });
      await withFileLock(skinTarget, async () => {
        await atomicWrite(skinTarget, skin);
      });
    }
  } catch (err) {
    console.warn(`[sei] cache-on-demand skin download failed for ${uuid}: ${(err as Error).message}`);
  }

  // Portrait
  try {
    const portrait = await downloadPortrait(session.user.id, uuid);
    if (portrait) {
      const portraitTarget = paths.portraitPath(uuid);
      await mkdirP(path.dirname(portraitTarget), { recursive: true });
      await withFileLock(portraitTarget, async () => {
        await atomicWrite(portraitTarget, portrait);
      });
    }
  } catch (err) {
    console.warn(`[sei] cache-on-demand portrait download failed for ${uuid}: ${(err as Error).message}`);
  }
}

export interface MergedCharacter {
  id: string;
  source: 'local' | 'cloud' | 'both';
  // Plus character fields — caller decides whether to dedupe by id
}

export async function listMerged(): Promise<{ characters: Array<{ id: string; name: string; is_default: boolean; source: 'local' | 'cloud' | 'both' }> }> {
  const { listCharacters } = await import('../characterStore'); // adapt name
  const local = await listCharacters();

  const { getClient } = await import('../auth/supabaseClient');
  const session = (await getClient().auth.getSession()).data.session;
  let cloud: { id: string; name: string }[] = [];
  if (session?.user.id) {
    try {
      const { listMyCharacters } = await import('./cloudCharacterClient');
      const rows = await listMyCharacters(session.user.id);
      cloud = rows.map(r => ({ id: r.id, name: r.name }));
    } catch (err) {
      console.warn(`[sei] cache-on-demand listMerged cloud fetch failed: ${(err as Error).message}`);
    }
  }
  const localIds = new Set(local.map(c => c.id));
  const cloudIds = new Set(cloud.map(c => c.id));
  const allIds = new Set([...localIds, ...cloudIds]);
  const merged = Array.from(allIds).map(id => {
    const l = local.find(c => c.id === id);
    const c = cloud.find(r => r.id === id);
    const source: 'local' | 'cloud' | 'both' =
      l && c ? 'both' : l ? 'local' : 'cloud';
    return {
      id,
      name: (l?.name ?? c?.name)!,
      is_default: l?.is_default ?? false,
      source,
    };
  });
  return { characters: merged };
}
