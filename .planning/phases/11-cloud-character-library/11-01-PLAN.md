```sql
-- Phase 11 — Extend Phase 10's purge-deletion-queue cron to actually
-- delete storage.objects per storage_paths jsonb entry before marking
-- the row purged. Closes Phase 10's deferred BL-03 follow-on.
-- Source: 10-VERIFICATION (BL-03 deferred to Phase 11); 11-RESEARCH §Pitfall 6.

select cron.unschedule('purge-deletion-queue');

select cron.schedule(
  'purge-deletion-queue',
  '0 3 * * *',
  $$
    with due as (
      select id, user_id, storage_paths
      from public.deletion_queue
      where deletion_requested_at < now() - interval '30 days'
        and purged_at is null
    ),
    paths as (
      select
        d.id,
        d.user_id,
        (jsonb_array_elements_text(d.storage_paths))::text as obj_path
      from due d
    ),
    -- T-11-10-01 path-ownership guard: a malicious user could insert a deletion_queue
    -- row naming another user's '<other_owner>/<uuid>.png' path. The cron runs with
    -- privileged context, so storage RLS is bypassed. Filter each path to ones whose
    -- first segment (storage.foldername(name)[1]) matches the row's user_id.
    safe_paths as (
      select id, obj_path
      from paths
      where split_part(obj_path, '/', 1) = user_id::text
    ),
    deleted as (
      delete from storage.objects o
      using safe_paths p
      where o.bucket_id in ('skins', 'portraits')
        and o.name = p.obj_path
      returning 1
    )
    update public.deletion_queue
    set purged_at = now()
    where id in (select id from due)
  $$
);
```

Note: storage_paths entries are stored as `<owner_uuid>/<character_uuid>.png` (Plan 11-01 storage_buckets nested layout, Plan 11-10 deletionQueueWriter). The cron iterates both buckets via `bucket_id in ('skins', 'portraits')`. The `safe_paths` CTE filters out any entry whose first path segment does NOT match the row's `user_id` — this is the T-11-10-01 mitigation against cross-user storage delete via a crafted deletion_queue row.
