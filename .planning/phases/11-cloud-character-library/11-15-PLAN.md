Handler:

```tsx
const onToggleShared = async (): Promise<void> => {
  const next = !character.shared;
  if (authState.kind === 'signed_in') {
    try {
      await sei.charsSetShared({ id: character.id, shared: next });
      // Local state will refresh via the chars:list re-query or the character refetch path
    } catch (err) {
      // Show error inline
      console.warn(`set-shared failed: ${(err as Error).message}`);
    }
  } else {
    // Signed-out: upgrade-to-sign-in
    setUpgradeFraming('share this character');
    setPendingShareIntent({ characterId: character.id });
    openModal({ kind: 'signin' });
  }
};
```

Add styling for `.sharedToggleRow` / `.sharedToggle` / `.sharedToggleOn` / `.sharedLabel` to the matching CSS module. Match the existing StatusPill / chip palette тАФ no new colors.
