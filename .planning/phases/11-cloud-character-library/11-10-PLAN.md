**Step A — create deletionQueueWriter.ts:**

```typescript
/**
 * Phase 11 — Belt-and-suspenders Storage orphan cleanup writer.
 *
 * Source: 11-RESEARCH §Pitfall 6 (cloud delete succeeds, Storage delete fails).
 *         Inserts a deletion_queue row at the moment of character delete; the
 *         Plan 11-01 storage_purge_extend cron iterates storage_paths nightly
 *         and deletes storage.objects. If the syncQueue.processNext path also
 *         deleted the storage objects directly, the cron finds them already
 *         gone (storage.objects DELETE is idempotent — WHERE clause filters to
 *         nothing).
 *
 * MAIN PROCESS ONLY.
 */

import { getClient } from '../auth/supabaseClient';

const TIMEOUT_MS = 15_000;

export async function enqueueStorageOrphans(
  ownerUuid: string,
  paths: string[],
): Promise<void> {
  if (paths.length === 0) return;
  const supabase = getClient();
  const controller = new AbortController();
  const handle = setTimeout(() => controller.abort(), TIMEOUT_MS);
  try {
    const { error } = await supabase
      .from('deletion_queue')
      .insert({
        user_id: ownerUuid,
        storage_paths: paths,
      })
      .abortSignal(controller.signal);
    if (error) throw new Error(`CLOUD_DELETION_QUEUE_INSERT_FAILED: ${error.message}`);
  } finally {
    clearTimeout(handle);
  }
}
```

if (!character.is_default) {
  void (async () => {
    try {
      const { getClient } = await import('./auth/supabaseClient');
      const session = (await getClient().auth.getSession()).data.session;
      const owner = session?.user.id;
      if (!owner) return; // signed-out delete — no cloud cleanup needed (nothing was uploaded)

      // Belt: write the deletion_queue row first (cheap insert, RLS allows user-self insert).
      try {
        const { enqueueStorageOrphans } = await import('./cloud/deletionQueueWriter');
        await enqueueStorageOrphans(owner, [
          `${owner}/${id}.png`, // applies to both buckets (purge body iterates both)
        ]);
      } catch (err) {
        console.warn(`[sei] deletion_queue insert failed for ${id}: ${(err as Error).message}`);
        // Don't block — the sync queue still attempts direct delete below.
      }

      // Suspenders: enqueue the direct cloud-delete + storage delete attempt.
      const { enqueueDelete } = await import('./cloud/syncQueue');
      await enqueueDelete(id, [
        { bucket: 'skins',     name: `${owner}/${id}.png` },
        { bucket: 'portraits', name: `${owner}/${id}.png` },
      ]);
      const { processNext } = await import('./cloud/syncQueue');
      void processNext();
    } catch (err) {
      console.warn(`[sei] cloud delete enqueue failed for ${id}: ${(err as Error).message}`);
    }
  })();
}

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-10-01 | T | User writes deletion_queue rows targeting another user's storage paths | mitigate | RLS `with check (user_id = auth.uid())` prevents wrong-user insert; the cron iterates only the paths in the row but the storage RLS on the cron's service_role context is bypassed — note: cron operates as the database superuser via cron.schedule's privileged context; the paths are written by the user but the cron deletes via service_role. RISK: a malicious user could insert deletion_queue rows naming another user's `<other_owner>/<uuid>.png` path. Per Plan 11-01 nested layout, the path starts with owner UUID — combined with this RLS, the cron can be tricked into deleting another user's objects. MITIGATION: extend the cron body to verify each path's first segment matches the row's user_id before deleting. THIS MUST BE BACK-PORTED INTO PLAN 11-01's storage_purge_extend migration. |
| T-11-10-02 | D | User spams deletion_queue with millions of rows | mitigate | RLS only constrains content, not rate; for v1.0 accept the risk (no abuse seen in pre-launch alpha). Phase 12 may add a rate-limit |
| T-11-10-03 | I | deletion_queue row leak reveals UUIDs of deleted chars | accept | RLS default-deny on select means users cannot read others' rows; only service_role (cron) reads |
