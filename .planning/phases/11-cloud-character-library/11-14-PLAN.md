```typescript
import { isTosAccepted } from './tosGate';

const TOS_CACHE_TTL_MS = 60_000;
let tosCache: { userId: string; accepted: boolean; cachedAt: number } | null = null;

export function invalidateTosCache(): void {
  tosCache = null;
}

/**
 * Phase 11 — defense-in-depth UX gate for cloud writes.
 *
 * Returns true ONLY when ALL of:
 *   - auth state is 'signed_in'
 *   - user.emailVerified === true (Phase 10 D-04 + Pitfall 8)
 *   - tos_acceptance row exists at current TOS_VERSION + PRIVACY_VERSION (D-26)
 *
 * Fails closed on any error.
 *
 * RLS at the database is the SECURITY gate; this is the UX gate that
 * prevents misleading "syncing…" UI when sync cannot actually succeed.
 * Plan 11-08 syncQueue.processNext uses this as its drain-eligibility check.
 */
export async function isCloudWriteAllowed(): Promise<boolean> {
  if (currentState.kind !== 'signed_in') {
    logCloudWriteDenied('not_signed_in');
    return false;
  }
  // Email verification check (Pitfall 8)
  // The exact field name depends on Phase 10's signed_in user shape — check
  // currentState.user.emailVerified vs currentState.user.email_verified. The
  // Supabase user object exposes `email_confirmed_at` (timestamp or null).
  // Phase 10 normalizes this into the state. Use whichever Phase 10 ships.
  if (!currentState.user.emailVerified) {
    logCloudWriteDenied('email_unverified');
    return false;
  }
  const userId = currentState.user.id;

  // Cache check
  const now = Date.now();
  if (tosCache && tosCache.userId === userId && now - tosCache.cachedAt < TOS_CACHE_TTL_MS) {
    if (!tosCache.accepted) logCloudWriteDenied('tos_not_accepted');
    return tosCache.accepted;
  }

  let accepted: boolean;
  try {
    accepted = await isTosAccepted(userId);
  } catch {
    accepted = false;
  }
  tosCache = { userId, accepted, cachedAt: now };
  if (!accepted) logCloudWriteDenied('tos_not_accepted');
  return accepted;
}

function logCloudWriteDenied(reason: 'not_signed_in' | 'email_unverified' | 'tos_not_accepted'): void {
  // Trace-level logging — keep low-noise. Console is fine for v1.0.
  console.log(`[sei] isCloudWriteAllowed: false (reason: ${reason})`);
}
```
