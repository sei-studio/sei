<objective>
Build the orchestrator that owns ALL LLM-related state per CONTEXT.md and RESEARCH.md "Architectural Responsibility Map":

- Hop counter (LLM-04, D-11/D-12): per-event-chain `EventChain { hops, MAX:5 }`. On overflow: abort signal + emit pre-rendered cap-hit line + log warn.
- Token-bucket rate limiter (LLM-06): 30/min on the personality call site. Idle dropped on empty bucket; chat queued (depth <=5).
- Ollama circuit breaker (LLM-08, D-13/D-14): startup probe with 3 retries x 2s; mid-session trip after 3 consecutive Qwen failures; no half-open state.
- Goal store (D-06): in-memory `owner_goals[]` + `self_goals[]`. Mutated ONLY via `setGoals` registry action (D-07).
- `setGoals` registry action: appended to `createDefaultRegistry()`. Schema: `{ list: 'owner'|'self', op: 'add'|'remove', goal: string }`.
- Debouncer (LLM-05): 500ms per-key coalesce. Wave 3 wires it into bot event ingestion; this plan only builds the primitive.
- Personality call (LLM-02): system prefix = `[systemInstructions, persona, toolDescriptions(personality-tools)]` with cache_control on the last block. Personality tools are ONLY: `setGoals`, `say(text)`, `handOffToMovement(prose)` — NEVER mineflayer actions (CLAUDE.md ADR #2, D-04).
- Movement call (LLM-03): when `handOffToMovement(prose)` is invoked, route prose to OllamaClient (or AnthropicClient when in haiku-fallback) with the FULL Phase 1 registry as tools (D-03, minus `setGoals`). Validate each tool_call via `registry.execute` (which Zod-validates).
- PERS-04 proactive observations: idle ticks invoke personality LLM, which may emit `say` calls; rate-limited via the same 30/min bucket.

Purpose: Wave 3 wires this orchestrator into the FSM `sei:dispatch` event. Building it standalone first means Wave 3 is pure wiring (no new logic).

Output: 5 new files under `src/llm/` + extension of `src/registry.js`.
</objective>

```javascript
/**
 * In-memory goal store (D-06). Phase 3 lifts to SQLite (MEM-04).
 * Mutated ONLY via the `setGoals` registry action (D-07).
 */
export function createGoalStore() {
  const owner_goals = []
  const self_goals = []
  const pickList = (list) =&gt; (list === 'owner' ? owner_goals : list === 'self' ? self_goals : null)
  return {
    get owner_goals() { return [...owner_goals] },
    get self_goals()  { return [...self_goals] },
    add(list, goal) {
      const arr = pickList(list); if (!arr) throw new Error(`unknown list: ${list}`)
      const trimmed = String(goal).trim()
      if (!trimmed) return false
      if (arr.includes(trimmed)) return false
      arr.push(trimmed); return true
    },
    remove(list, goal) {
      const arr = pickList(list); if (!arr) throw new Error(`unknown list: ${list}`)
      const i = arr.indexOf(String(goal).trim())
      if (i &lt; 0) return false
      arr.splice(i, 1); return true
    },
    snapshot() {
      return { owner_goals: [...owner_goals], self_goals: [...self_goals] }
    },
  }
}
```

```javascript
/**
 * Token bucket. capacity tokens, refill `refillPerMin` per minute.
 * @param {{capacity:number, refillPerMin:number}} opts
 */
export function createTokenBucket({ capacity, refillPerMin }) {
  let tokens = capacity
  const refillIntervalMs = 60_000 / refillPerMin
  let lastRefill = Date.now()
  function refill() {
    const elapsed = Date.now() - lastRefill
    const add = Math.floor(elapsed / refillIntervalMs)
    if (add &gt; 0) {
      tokens = Math.min(capacity, tokens + add)
      lastRefill += add * refillIntervalMs
    }
  }
  return {
    /** @returns {boolean} */
    tryAcquire() { refill(); if (tokens &gt;= 1) { tokens -= 1; return true } return false },
    available() { refill(); return tokens },
    /** Wait up to `timeoutMs` for a token. Resolves true if acquired, false on timeout. */
    async awaitAcquire(timeoutMs = 5000) {
      const deadline = Date.now() + timeoutMs
      while (Date.now() &lt; deadline) {
        if (this.tryAcquire()) return true
        const wait = Math.min(refillIntervalMs, deadline - Date.now())
        if (wait &lt;= 0) break
        await new Promise(r =&gt; setTimeout(r, wait))
      }
      return false
    },
  }
}
```

```javascript
/**
 * Per-key 500ms coalescer. Last payload wins; one fire per key per window.
 */
export function createDebouncer(windowMs) {
  /** @type {Map&lt;string,{timer:any,payload:any,fire:Function}&gt;} */
  const pending = new Map()
  return {
    /** Coalesce events with the same `key`; `fire(payload)` is called once after windowMs of quiet. */
    debounce(key, payload, fire) {
      const existing = pending.get(key)
      if (existing) { clearTimeout(existing.timer); existing.payload = payload; existing.fire = fire }
      const entry = existing ?? { payload, fire, timer: null }
      entry.timer = setTimeout(() =&gt; { pending.delete(key); entry.fire(entry.payload) }, windowMs)
      pending.set(key, entry)
    },
    /** Cancel all pending. */
    flushCancel() { for (const e of pending.values()) clearTimeout(e.timer); pending.clear() },
  }
}
```

```javascript
/**
 * Two-state Ollama circuit (D-13/D-14). No half-open: D-14 says no flapping.
 */
export function createOllamaCircuit({ tripAt = 3 } = {}) {
  let state = 'qwen'  // or 'haiku-fallback'
  let consecutiveFailures = 0
  return {
    get state() { return state },
    isOpen() { return state === 'haiku-fallback' },
    recordSuccess() { consecutiveFailures = 0 },
    recordFailure() {
      consecutiveFailures += 1
      if (consecutiveFailures &gt;= tripAt) state = 'haiku-fallback'
      return state
    },
    /** Force open (used by startup-probe failure, D-13). */
    trip(reason) { state = 'haiku-fallback'; return reason },
  }
}
```

```javascript
import { createAnthropicClient } from './anthropicClient.js'
import { createOllamaClient } from './ollamaClient.js'
import { createGoalStore } from './goals.js'
import { createTokenBucket } from './rateLimiter.js'
import { createDebouncer } from './debounce.js'
import { createOllamaCircuit } from './circuit.js'
import { renderPersona, capHitLine } from './persona.js'
import { buildAnthropicTools, buildOllamaTools } from './schemaBridge.js'

const SYSTEM_INSTRUCTIONS = [
  'You are the personality layer of a Minecraft companion bot.',
  'You react to chat, world events, and idle ticks.',
  'You decide WHAT to do at a high level — never mention coordinates, action names, or code.',
  'When you want the body to move or interact with the world, call handOffToMovement with a short natural-language instruction (e.g. "go check what shawn is building over by the water").',
  'When you want to speak in chat, call say with the exact line.',
  'When the owner sets a goal or you decide on a self-goal, call setGoals.',
  'You may call multiple tools in one response. Keep responses brief — under 3 sentences of internal reasoning.',
  'If you have owner_goals, prioritize progressing them. Otherwise pick a self_goal or freely play.',  // D-08
].join('\n')

const MOVEMENT_SYSTEM = [
  'You translate natural-language intent into one or more registered action calls.',
  'You ONLY emit tool_calls — no prose. Pick the action(s) that best fulfill the intent.',
  'If the intent is unclear or no action fits, emit no tool_calls.',
].join('\n')

const ACTION_DESCRIPTIONS = {
  goTo: 'Move the bot to the given (x, y, z) coordinates within `range` blocks.',
  setGoals: 'Add or remove a goal from owner_goals or self_goals.',
  // Personality-only pseudo-tools (not in mineflayer registry):
  say: 'Speak the given text in in-game chat.',
  handOffToMovement: 'Hand off a natural-language movement/interaction intent to the movement layer.',
}

/**
 * @param {object} deps
 * @param {object} deps.bot
 * @param {object} deps.config
 * @param {object} deps.registry  // result of createDefaultRegistry() — already includes setGoals
 * @param {{warn:Function,info:Function,error:Function}} [deps.logger]
 */
export function createOrchestrator({ bot, config, registry, logger = console }) {
  const goals = createGoalStore()
  const anthropic = createAnthropicClient(config)
  const ollama = createOllamaClient(config)
  const circuit = createOllamaCircuit({ tripAt: 3 })
  const personalityBucket = createTokenBucket({
    capacity: config.llm.rate_limit_per_min,
    refillPerMin: config.llm.rate_limit_per_min,
  })
  const ingressDebouncer = createDebouncer(config.llm.debounce_ms)

  // Personality-only tools: setGoals, say, handOffToMovement
  // Movement gets the full mineflayer registry MINUS setGoals (Open Q 2)
  const personalityTools = [
    {
      name: 'say',
      description: ACTION_DESCRIPTIONS.say,
      input_schema: { type: 'object', properties: { text: { type: 'string' } }, required: ['text'] },
    },
    {
      name: 'handOffToMovement',
      description: ACTION_DESCRIPTIONS.handOffToMovement,
      input_schema: { type: 'object', properties: { intent: { type: 'string' } }, required: ['intent'] },
    },
    {
      name: 'setGoals',
      description: ACTION_DESCRIPTIONS.setGoals,
      input_schema: {
        type: 'object',
        properties: {
          list: { type: 'string', enum: ['owner', 'self'] },
          op:   { type: 'string', enum: ['add', 'remove'] },
          goal: { type: 'string', minLength: 1 },
        },
        required: ['list', 'op', 'goal'],
      },
    },
  ]

  // Movement registry tools (exclude setGoals — that's personality-only)
  function movementToolsFor(provider) {
    const subRegistry = {
      list:   () =&gt; registry.list().filter(n =&gt; n !== 'setGoals'),
      schema: (n) =&gt; registry.schema(n),
    }
    return provider === 'anthropic'
      ? buildAnthropicTools(subRegistry, ACTION_DESCRIPTIONS)
      : buildOllamaTools(subRegistry, ACTION_DESCRIPTIONS)
  }

  // Build cached personality system prefix once at start (re-renders only on config change — D-16: no hot-reload)
  let cachedSystemBlocks = null
  function rebuildPersonalitySystem() {
    cachedSystemBlocks = anthropic.buildCachedSystem(
      SYSTEM_INSTRUCTIONS,
      renderPersona(config.persona),
      personalityTools
    )
  }
  rebuildPersonalitySystem()

  // ─── Startup probe (D-13) — 3 retries × 2s ───
  async function probeOllamaWithRetry() {
    for (let i = 0; i &lt; 3; i++) {
      if (await ollama.probe()) return true
      await new Promise(r =&gt; setTimeout(r, 2000))
    }
    return false
  }

  async function start() {
    const ok = await probeOllamaWithRetry()
    if (!ok) {
      circuit.trip('startup probe failed')
      logger.warn('[sei/orch] Ollama unreachable at startup — using Haiku-as-executor for this session.')
    } else {
      logger.info(`[sei/orch] Ollama reachable at ${ollama.host} — model ${ollama.model}`)
    }
  }

  // ─── Movement dispatch (LLM-03 / LLM-08) ───
  async function callMovement(intent, signal) {
    const tools = movementToolsFor(circuit.isOpen() ? 'anthropic' : 'ollama')
    const messages = [
      { role: 'system', content: MOVEMENT_SYSTEM },
      { role: 'user',   content: intent },
    ]
    if (circuit.isOpen()) {
      // Anthropic shape — system goes in `system` not `messages`
      const resp = await anthropic.call({
        systemBlocks: [{ type: 'text', text: MOVEMENT_SYSTEM }],
        tools,  // already in Anthropic shape
        messages: [{ role: 'user', content: intent }],
        signal,
      })
      return { toolCalls: resp.toolUses.map(u =&gt; ({ name: u.name, args: u.input })) }
    }
    try {
      const resp = await ollama.call({ messages, tools, signal })
      circuit.recordSuccess()
      return { toolCalls: resp.toolCalls }
    } catch (err) {
      const newState = circuit.recordFailure()
      logger.warn(`[sei/orch] Ollama call failed (${err.message}); circuit state=${newState}`)
      throw err
    }
  }

  // ─── Personality call (LLM-02 / LLM-06) ───
  async function callPersonality(userBlock, signal) {
    if (!personalityBucket.tryAcquire()) {
      logger.warn('[sei/orch] Rate limit hit — dropping personality call')
      return null
    }
    return await anthropic.call({
      systemBlocks: cachedSystemBlocks,
      tools: personalityTools,
      messages: [{ role: 'user', content: userBlock }],
      signal,
    })
  }

  // ─── Main dispatch loop (LLM-04 hop cap, LLM-07 abort propagation) ───
  /**
   * Single entry point. Wave 3 wires this to the FSM `sei:dispatch` event.
   * @param {string} event   // 'sei:chat_received' | 'sei:idle' | 'sei:attacked' | ...
   * @param {object} data
   * @param {AbortSignal} signal
   */
  async function handleDispatch(event, data, signal) {
    const MAX_HOPS = config.llm.max_hops
    let hops = 0
    let nextUser = renderUserContext(event, data, goals.snapshot())

    try {
      while (true) {
        if (signal.aborted) return
        hops += 1
        if (hops &gt; MAX_HOPS) throw new HopCapHit()

        const personalityResp = await callPersonality(nextUser, signal)
        if (!personalityResp) return  // rate-limited drop

        // Process tool_uses serially: say first, then setGoals, then handOffToMovement (Open Q 5 ordering)
        const sayCalls    = personalityResp.toolUses.filter(u =&gt; u.name === 'say')
        const goalCalls   = personalityResp.toolUses.filter(u =&gt; u.name === 'setGoals')
        const handoffCall = personalityResp.toolUses.find(u =&gt; u.name === 'handOffToMovement')

        for (const c of sayCalls)  bot.chat(String(c.input?.text ?? '').slice(0, 256))
        for (const c of goalCalls) await registry.execute('setGoals', c.input, bot, { ...config, _goalStore: goals })

        if (!handoffCall) return  // no movement requested → chain ends here

        // Movement hop
        if (signal.aborted) return
        hops += 1
        if (hops &gt; MAX_HOPS) throw new HopCapHit()

        const movement = await callMovement(String(handoffCall.input?.intent ?? ''), signal)
        if (signal.aborted) return

        // Dispatch movement tool_calls via Phase 1 registry. Each gets the FSM signal.
        for (const call of movement.toolCalls) {
          if (signal.aborted) return
          try {
            await registry.execute(call.name, call.args, bot, { ...config, _goalStore: goals, signal })
          } catch (err) {
            logger.warn(`[sei/orch] action ${call.name} failed: ${err.message}`)
          }
        }
        // After movement, the FSM will emit a new sei:dispatch on completion (Phase 1 already does P2).
        // We do NOT loop back into personality here — that would double-count hops. Chain ends.
        return
      }
    } catch (err) {
```
