    import { readFile, mkdir, unlink } from 'node:fs/promises';
    import path from 'node:path';
    import crypto from 'node:crypto';
    import { app } from 'electron';
    import { atomicWrite } from '../bot/brain/storage/atomicWrite.js';
    import { withFileLock } from '../bot/brain/storage/fileLock.js';
    import { paths } from './paths';
    import { getCharacter, saveCharacter } from './characterStore';
    import { DEFAULT_CHARACTERS } from './defaultCharacters';
    import type { Character, Skin, SkinSource } from '../shared/characterSchema';

    /** Resolve the absolute path to the bundled PNG for a default persona. Returns null if the persona isn't a shipped default. */
    export function bundledSkinPath(personaId: string): string | null {
      const isBundled = DEFAULT_CHARACTERS.some(c => c.id === personaId);
      if (!isBundled) return null;
      // app.isPackaged → resources/skins/ ships under app.asar.unpacked (Plan 01 Task 2 added the asarUnpack entry).
      // Dev → resources/ lives at the repo root.
      if (app.isPackaged) {
        return path.join(process.resourcesPath, 'app.asar.unpacked', 'resources', 'skins', `${personaId}.png`);
      }
      return path.join(__dirname, '..', '..', 'resources', 'skins', `${personaId}.png`);
    }

    /** Read the PNG bytes that should be served for this persona, honoring its skin.source descriptor. Returns null when no skin should be served (source === 'none'). */
    export async function resolveSkinPng(character: Character): Promise<Buffer | null> {
      const s = character.skin;
      if (s.source === 'none') return null;
      if (s.source === 'bundled') {
        const p = bundledSkinPath(character.id);
        if (!p) return null;
        try { return await readFile(p); } catch { return null; }
      }
      // 'upload' or 'username' → user-applied PNG under <userData>/skins/<id>.png
      try { return await readFile(paths.skinPngPath(character.id)); } catch { return null; }
    }

    /** Apply a PNG: write atomically, sha256-verify, patch the character's skin descriptor, persist. */
    export async function applyPng(args: {
      personaId: string;
      pngBytes: Buffer;
      source: SkinSource;
      mojangUsername?: string | null;
    }): Promise<Skin> {
      const { personaId, pngBytes, source, mojangUsername = null } = args;
      if (source !== 'upload' && source !== 'username') {
        throw new Error(`applyPng: invalid source '${source}' (must be 'upload' or 'username')`);
      }
      // Validate PNG magic bytes (defense-in-depth — Plan 03 also validates, but main never trusts renderer input).
      if (pngBytes.length < 24 ||
          pngBytes[0] !== 0x89 || pngBytes[1] !== 0x50 ||
          pngBytes[2] !== 0x4E || pngBytes[3] !== 0x47) {
        throw new Error('applyPng: not a PNG (magic-byte mismatch)');
      }
      // Parse IHDR width/height from offset 16-23 (after the 8-byte signature + 8-byte IHDR chunk header).
      const width = pngBytes.readUInt32BE(16);
      const height = pngBytes.readUInt32BE(20);
      if (width !== 64 || height !== 64) {
        throw new Error(`applyPng: PNG must be 64×64 (got ${width}×${height})`);
      }
      await mkdir(paths.skinsDir(), { recursive: true });
      const target = paths.skinPngPath(personaId);
      await withFileLock(target, async () => {
        await atomicWrite(target, pngBytes);
      });
      const sha256 = crypto.createHash('sha256').update(pngBytes).digest('hex');
      const character = await getCharacter(personaId);
      if (!character) throw new Error(`applyPng: character not found '${personaId}'`);
      const newSkin: Skin = {
        source,
        mojang_username: source === 'username' ? mojangUsername : null,
        png_sha256: sha256,
        applied_at: new Date().toISOString(),
      };
      await saveCharacter({ ...character, skin: newSkin });
      return newSkin;
    }

    /** Reset a persona: defaults revert to 'bundled', user-created revert to 'none'. */
    export async function removePng(personaId: string): Promise<Skin> {
      const character = await getCharacter(personaId);
      if (!character) throw new Error(`removePng: character not found '${personaId}'`);
      // Wipe any user-applied PNG (best-effort — file may not exist if user never applied one).
      try { await unlink(paths.skinPngPath(personaId)); } catch { /* ignore ENOENT */ }
      const isBundled = DEFAULT_CHARACTERS.some(c => c.id === personaId);
      const newSkin: Skin = {
        source: isBundled ? 'bundled' : 'none',
        mojang_username: null,
        png_sha256: null,
        applied_at: new Date().toISOString(),
      };
      await saveCharacter({ ...character, skin: newSkin });
      return newSkin;
    }

    /** Pure-read helper used by the HTTP server: takes a username, finds the matching character, calls resolveSkinPng. */
    export async function readSkinPng(args: {
      username: string;
      listCharacters: () => Promise<Character[]>;
    }): Promise<Buffer | null> {
      const all = await args.listCharacters();
      // Match: persona's per-persona `username` field first, then sanitized name fallback so existing
      // bundled-username matching keeps working before the user opens the wizard.
      const sanitize = (s: string) => String(s || '').replace(/[^A-Za-z0-9_]/g, '').slice(0, 16) || 'Sei';
      const match = all.find(c => (c.username && c.username === args.username) || sanitize(c.name) === args.username);
      if (!match) return null;
      return await resolveSkinPng(match);
    }
    ```

    Notes:
    - `withFileLock` + `atomicWrite` are the existing helpers used everywhere else; do NOT re-implement them.
    - PNG IHDR parse uses readUInt32BE at offsets 16/20 — the magic+IHDR layout is rigid per spec.
    - sha256 hashing is on the PNG bytes themselves so a cache-bust query string in the HTTP layer is easy if we need it later (not added in this plan).
    - The username match is intentionally loose so a renderer that hasn't yet wired `character.username` still gets correct bundled-skin behavior on first launch.

    import http from 'node:http';
    import { readSkinPng, bundledSkinPath } from './skinStore';
    import { listCharacters } from './characterStore';
    import { readFile } from 'node:fs/promises';

    export interface SkinServer {
      baseUrl: string;     // e.g. 'http://127.0.0.1:54321'
      port: number;        // bound port
      stop: () => Promise<void>;
    }

    const NOT_FOUND_PNG_TRANSPARENT = (() => {
      // Pre-baked 1×1 fully-transparent PNG so 404 responses still parse client-side. 67 bytes.
      // Generated by the same encoder used in scripts/build-default-skins.mjs; inlined here as
      // a base64 const so we don't depend on filesystem at request time.
      return Buffer.from('iVBORw0KGgoAAAANSUhEUgAAAAEAAAABCAYAAAAfFcSJAAAADUlEQVR42mNk+M9QDwADhgGAWjR9awAAAABJRU5ErkJggg==', 'base64');
    })();

    /**
     * Start a loopback-only HTTP server that serves persona skin PNGs.
     * Binds 127.0.0.1 (no LAN exposure) on an OS-chosen port (port 0).
     * CustomSkinLoader on the host's MC client is configured to fetch from
     * `${baseUrl}/skins/<username>.png` — the wizard writes this URL into
     * the CSL config (Plan 04).
     */
    export async function createSkinServer(args: {
      port?: number;
    } = {}): Promise<SkinServer> {
      const server = http.createServer(async (req, res) => {
        try {
          const url = req.url || '';
          // Accept GET /skins/<username>.png — anything else 404s. No directory listing.
          const m = url.match(/^\/skins\/([A-Za-z0-9_]{1,16})\.png(\?.*)?$/);
          if (req.method !== 'GET' || !m) {
            res.writeHead(404, { 'content-type': 'text/plain' });
            res.end('Not Found');
            return;
          }
          const username = m[1];
          const png = await readSkinPng({ username, listCharacters });
          if (!png) {
            // Per CustomSkinLoader docs: a 404 response is a valid "no skin available" signal,
            // but some CSL builds prefer a transparent 1×1 PNG to short-circuit retries.
            // Return the transparent PNG with content-type image/png so the client doesn't
            // misinterpret the body.
            res.writeHead(404, { 'content-type': 'image/png' });
            res.end(NOT_FOUND_PNG_TRANSPARENT);
            return;
          }
          res.writeHead(200, {
            'content-type': 'image/png',
            'content-length': String(png.length),
            'cache-control': 'no-store',
          });
          res.end(png);
        } catch (err) {
          // eslint-disable-next-line no-console
          console.error('[sei] skinServer request handler error:', err);
          res.writeHead(500, { 'content-type': 'text/plain' });
          res.end('Internal Server Error');
        }
      });

      await new Promise<void>((resolve, reject) => {
        const onError = (err: NodeJS.ErrnoException) => {
          server.off('listening', onListening);
          // EADDRINUSE only fires if we pass an explicit non-zero port. With port=0 the OS picks a free port, so this branch is the "system has zero free ephemeral ports" case — extremely rare but classifiable for the UI.
          const e = new Error(`SKIN_SERVER_PORT_TAKEN: ${err.code ?? 'unknown'} ${err.message}`);
          reject(e);
        };
        const onListening = () => {
          server.off('error', onError);
          resolve();
        };
        server.once('error', onError);
        server.once('listening', onListening);
        // 127.0.0.1 ONLY. NEVER bind 0.0.0.0 — that would expose persona skins to the LAN
        // (and trigger the macOS / Windows firewall prompt the user does not need).
        server.listen(args.port ?? 0, '127.0.0.1');
      });

      const addr = server.address();
      if (!addr || typeof addr === 'string') {
        throw new Error('SKIN_SERVER_PORT_TAKEN: server.address() returned unexpected shape');
      }
      const port = addr.port;
      const baseUrl = `http://127.0.0.1:${port}`;

      return {
        baseUrl,
        port,
        stop: () => new Promise<void>((resolve) => server.close(() => resolve())),
      };
    }
    ```

    Notes:
    - Port 0 = let the OS pick. We expose the bound port via `.port` and `.baseUrl` so the wizard's CustomSkinLoader-config writer (Plan 04) can stamp it into `customskinloader.json`.
    - The 404 path returns a transparent 1×1 PNG to keep CustomSkinLoader retries quiet. The base64 string above MUST be a valid PNG — verify by `Buffer.from(b64,'base64').slice(0,8).toString('hex') === '89504e470d0a1a0a'`.
    - The regex `^/skins/([A-Za-z0-9_]{1,16})\.png(\?.*)?$` is the ONLY accepted shape. Anything else 404s with `text/plain`. Path traversal (`/skins/../etc/passwd`) cannot match.
    - `cache-control: no-store` so the user's MC client refetches after every skin change. No ETag handling needed.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LAN → skin server | Loopback-only bind (127.0.0.1) means LAN traffic cannot reach the server; defense at the socket layer not the application layer |
| renderer → main IPC (`skin:apply`) | PNG bytes cross from renderer's untrusted context into main's privileged context; magic-byte + IHDR-dim validation runs at the IPC handler, then again inside `applyPng` |
| host MC client → skin server | Localhost-only fetch; the only consumer is CustomSkinLoader running inside the user's own Minecraft process — same trust domain as the user themselves |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-S1 | S (Spoofing) | skin server username lookup | accept | Loopback-only; only the local user can reach the server. A second user on the same machine can spoof a request but they could just read `<userData>/skins/` directly anyway. |
| T-09-T1 | T (Tampering) | applyPng PNG validation | mitigate | Magic-byte check + IHDR width/height parse rejects non-PNG and non-64×64 input at the IPC boundary. Atomic write via `withFileLock` + `atomicWrite` prevents partial-write corruption. |
| T-09-I1 | I (Information Disclosure) | path traversal via /skins/... | mitigate | Regex `^/skins/([A-Za-z0-9_]{1,16})\.png(\?.*)?$` rejects any URL containing `..`, `/`, or non-username characters. Username -> persona lookup is via in-memory list (never filesystem path concatenation). Test 3 in verify-skinServer.mjs asserts this. |
| T-09-D1 | D (Denial of Service) | unbounded request rate to skin server | accept | Loopback-only + single-machine + the only client is the user's own MC client. A self-DoS would be the user attacking their own machine. |
| T-09-E1 | E (Elevation of Privilege) | renderer-controlled PNG write to disk | mitigate | Renderer can only write to `<userData>/skins/<personaId>.png` (path constructed from server-side `paths.skinPngPath`). personaId comes from IPC arg but is used as a single filename component — `path.join(userData, 'skins', `${id}.png`)` cannot escape because Node's `path.join` normalizes embedded `..` and the upstream IdSchema permits only kebab-case slugs. |
</threat_model>
