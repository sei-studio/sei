```tsx
import styles from './PercentBar.module.css';

export interface PercentBarProps {
  value: number;        // 0..100
  label?: string;       // optional accessible label
  size?: 'sm' | 'md' | 'lg';
}

export function PercentBar({ value, label, size = 'lg' }: PercentBarProps): React.ReactElement {
  const v = Math.max(0, Math.min(100, value));
  return (
    <div className={`${styles.root} ${styles[size]}`}
         role="progressbar"
         aria-valuenow={v}
         aria-valuemin={0}
         aria-valuemax={100}
         aria-label={label ?? `${v} percent`}>
      <div className={styles.fill} style={{ width: `${v}%` }} />
      <span className={styles.label}>{v}%</span>
    </div>
  );
}
```

```tsx
import { useCreditsStore } from '../lib/stores/useCreditsStore';
import { useUiStore } from '../lib/stores/useUiStore';
import { Button } from '../components/Button';
import { BackIcon } from '../components/icons';
import { PercentBar } from '../components/PercentBar';
import styles from './CreditsScreen.module.css';

const PLAN_LABEL: Record<string, string> = {
  trial: 'Trial',
  pack: 'Pack',
  unlimited: 'Unlimited',
  depleted: 'Depleted',
};

function formatRenewal(iso: string | null): string | null {
  if (!iso) return null;
  try {
    return new Intl.DateTimeFormat('en-US', { month: 'short', day: 'numeric', year: 'numeric' }).format(new Date(iso));
  } catch { return null; }
}

export function CreditsScreen(): React.ReactElement {
  const remainingPct = useCreditsStore((s) => s.remaining_pct);
  const plan = useCreditsStore((s) => s.plan);
  const renewsAt = useCreditsStore((s) => s.renews_at);
  const openCheckout = useCreditsStore((s) => s.openCheckout);
  const cancelSubscription = useCreditsStore((s) => s.cancelSubscription);
  const navigate = useUiStore((s) => s.navigate);
  const renewalText = formatRenewal(renewsAt);
  const isSubscribed = plan === 'unlimited';

  return (
    <div className={styles.root}>
      <div className={styles.backRow}>
        <Button kind="quiet" size="sm" icon={<BackIcon size={14} />} onClick={() => navigate({ kind: 'home' })}>Back</Button>
      </div>
      <h1 className={styles.title}>Credits</h1>

      <section className={styles.section}>
        <div className={styles.sectionTitle}>USAGE</div>
        <PercentBar value={remainingPct} label={`${remainingPct} percent of daily credits remaining`} />
        <p className={styles.planLabel}>{PLAN_LABEL[plan] ?? 'Trial'}</p>
        {isSubscribed && renewalText ? <p className={styles.muted}>Next renewal: {renewalText}</p> : null}
      </section>

      <section className={styles.section}>
        <div className={styles.sectionTitle}>TOP UP</div>
        <div className={styles.tile}>
          <div className={styles.tileTitle}>One-time credit pack</div>
          <p className={styles.tileBody}>Add credits to keep your bot chatting. Powered by Lemon Squeezy.</p>
          <div className={styles.tileActions}>
            <Button kind="accent" onClick={() => void openCheckout('pack')}>Top up</Button>
          </div>
        </div>
      </section>

      <section className={styles.section}>
        <div className={styles.sectionTitle}>UNLIMITED</div>
        <div className={styles.tile}>
          <div className={styles.tileTitle}>Monthly subscription</div>
          {isSubscribed
            ? <p className={styles.tileBody}>You&rsquo;re subscribed. Manage your subscription in the customer portal.</p>
            : <p className={styles.tileBody}>Unlock heavier daily usage with a recurring subscription. Cancellable anytime.</p>}
          <div className={styles.tileActions}>
            {isSubscribed
              ? <Button kind="primary" onClick={() => void cancelSubscription()}>Manage subscription</Button>
              : <Button kind="primary" onClick={() => void openCheckout('subscription')}>Go Unlimited</Button>}
          </div>
        </div>
      </section>
    </div>
  );
}
```
