  <name>Task 1: Add rateLimitGate Hono middleware wrapper</name>
  <files>proxy/src/rateLimit/gate.ts, proxy/src/rateLimit/gate.test.ts</files>
  <behavior>
    - On a request to /v1/messages, the middleware reads c.var.userId, estimates input tokens from the request body (peek without consuming — clone the request), computes the worst-case reservation µ$, then calls checkAllBuckets(userId, estInput, reservationMicro).
    - If allowed → next().
    - If not allowed → sendError(c, {code:'rate_limited', kind, retry_after_seconds}). Includes a `Retry-After: <int>` HTTP header on the response.
    - Body peek: must NOT consume the underlying request stream (downstream forward.ts needs to read it). Use Hono's `c.req.json()` which caches the parsed body on the context per request.
  </behavior>
  <action>
Create gate.ts:

```ts
import type { MiddlewareHandler } from 'hono';
import { checkAllBuckets } from './buckets.js';
import { estimateInputTokens } from '../anthropic/tokenize.js';
import { estimateReservationMicro } from '../anthropic/pricing.js';
import { sendError } from '../middleware/sentinel.js';

export const rateLimitGate: MiddlewareHandler<{ Variables: { userId: string } }> = async (c, next) => {
  const userId = c.get('userId');
  // c.req.json() caches the parsed body on this context — forward.ts will get the SAME parsed body.
  const body = await c.req.json() as { model?: string; max_tokens?: number };
  const model = body.model ?? 'claude-haiku-4-5';
  const maxOutput = body.max_tokens ?? 1024;
  const estInput = await estimateInputTokens(body);
  const reservationMicro = estimateReservationMicro(estInput, maxOutput, model);

  const result = await checkAllBuckets(userId, estInput, reservationMicro);
  if (!result.allowed) {
    c.header('Retry-After', String(result.retry_after_seconds));
    return sendError(c, { code: 'rate_limited', kind: result.kind, retry_after_seconds: result.retry_after_seconds });
  }
  await next();
};
```

```ts
import { Hono } from 'hono';
import { verifyJwt, type AuthVars } from './auth/verifyJwt.js';
import { rateLimitGate } from './rateLimit/gate.js';
import { forwardToAnthropic } from './anthropic/forward.js';
import { preDeduct } from './ledger/preDeduct.js';
import { settle, settleAsRefunded, settleAtReservation } from './ledger/settle.js';
import { remainingPct } from './ledger/balance.js';
import { estimateInputTokens } from './anthropic/tokenize.js';
import { loadEnv } from './env.js';

const app = new Hono<{ Variables: AuthVars }>();

app.get('/health', (c) => c.json({ status: 'ok', version: '1.0.0' }));

const env = loadEnv();

app.post('/v1/messages', verifyJwt, rateLimitGate, (c) => {
  return forwardToAnthropic(c, {
    preDeduct,
    settle,
    settleAsRefunded,
    settleAtReservation,
    remainingPct,
    estimateInputTokens,
    fetchImpl: fetch,
    anthropicApiKey: env.ANTHROPIC_API_KEY,
  });
});

app.notFound((c) => c.json({ error: 'not_found' }, 404));
app.onError((err, c) => {
  console.error('proxy error', { name: err.name, message: err.message });
  return c.json({ error: 'internal_error' }, 500);
});

export default app;
```
