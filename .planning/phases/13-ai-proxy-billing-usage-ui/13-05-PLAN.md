```ts
export type Usage = {
  input_tokens: number;
  cache_creation_input_tokens: number;
  cache_read_input_tokens: number;
  output_tokens: number;
};

// Per-million pricing in $ (verified 2026-05-22)
// Multiplying a token count by these values yields micro-dollars directly
// (since 1 µ$ = 1e-6 $ and $/M-tokens × tokens = µ$).
export const PRICING = {
  'claude-haiku-4-5': { input: 1.0, output: 5.0 },
} as const;

const DEFAULT_MODEL = 'claude-haiku-4-5';

export function computeMicroDollarCost(usage: Usage, model: string): bigint {
  const p = PRICING[model as keyof typeof PRICING] ?? PRICING[DEFAULT_MODEL];
  const usd =
    usage.input_tokens * p.input +
    usage.cache_creation_input_tokens * p.input * 1.25 +
    usage.cache_read_input_tokens * p.input * 0.10 +
    usage.output_tokens * p.output;
  return BigInt(Math.ceil(usd));
}

export function estimateReservationMicro(estInput: number, maxOutput: number, model: string): bigint {
  const p = PRICING[model as keyof typeof PRICING] ?? PRICING[DEFAULT_MODEL];
  const usd = estInput * p.input * 1.25 + maxOutput * p.output;
  return BigInt(Math.ceil(usd));
}
```

```ts
// ~95% accuracy for English per @anthropic-ai/tokenizer 0.0.4. Calibrate the
// 1.25× reservation multiplier from observed usage.input_tokens / estimated ratios
// after enough real traffic (RESEARCH Pitfall 8).
export async function estimateInputTokens(body: unknown): Promise<number> {
  // Anthropic Messages body: { system, messages, tools }
  const b = body as { system?: unknown; messages?: Array<{ content?: unknown }>; tools?: unknown };
  let total = 0;
  const stringify = (v: unknown): string => (typeof v === 'string' ? v : JSON.stringify(v ?? ''));

  if (b.system !== undefined) total += countTokens(stringify(b.system));
  if (Array.isArray(b.messages)) {
    for (const msg of b.messages) total += countTokens(stringify(msg.content));
  }
  if (b.tools !== undefined) total += countTokens(stringify(b.tools));
  return total;
}
```

`proxy/src/ledger/preDeduct.ts`:
```ts
import { getAdminClient } from '../supabase.js';
import { estimateReservationMicro } from '../anthropic/pricing.js';

export type PreDeductResult =
  | { status: 'ok'; reservationId: string; reservationMicro: bigint }
  | { status: 'insufficient' };

export async function preDeduct(
  userId: string,
  estInputTokens: number,
  maxOutputTokens: number,
  model: string,
): Promise<PreDeductResult> {
  const reservationMicro = estimateReservationMicro(estInputTokens, maxOutputTokens, model);
  const admin = getAdminClient();

  // Pass as string — BigInt cannot be JSON-serialized; Postgres bigint accepts numeric text.
  const { data, error } = await admin.rpc('reserve_credits', {
    p_user_id: userId,
    p_reservation_micro: reservationMicro.toString(),
  });
  if (error) throw error;
  if (!Array.isArray(data) || data.length === 0) return { status: 'insufficient' };
  return {
    status: 'ok',
    reservationId: (data[0] as { id: string }).id,
    reservationMicro,
  };
}
```

```ts
import { getAdminClient } from '../supabase.js';

// Trial caps per D-51 (already on Tier 2 — no ramp; trial = $5/day from launch)
const TRIAL_DAILY_CAP_MICRO = 5_000_000n;          // $5 / day
const SUBSCRIBER_DAILY_CAP_MICRO = 20_000_000n;    // $20 / day
const TRIAL_MONTHLY_CAP_MICRO = 5_000_000n;         // trial isn't really monthly; equal to daily for ratio purposes
const SUBSCRIBER_MONTHLY_CAP_MICRO = 600_000_000n;  // $600 / month (30 × $20)

export async function remainingPct(userId: string): Promise<number> {
  const admin = getAdminClient();
  // Balance
  const { data: balRow } = await admin
    .from('ledger_balance').select('balance_micro').eq('user_id', userId).maybeSingle();
  const balance = BigInt(balRow?.balance_micro ?? 0);

  // Plan kind from subscription_status
  const { data: subRow } = await admin
    .from('subscription_status').select('status').eq('user_id', userId).maybeSingle();
  const isSubscriber = subRow?.status === 'active';

  const dailyCap = isSubscriber ? SUBSCRIBER_DAILY_CAP_MICRO : TRIAL_DAILY_CAP_MICRO;
  const monthlyCap = isSubscriber ? SUBSCRIBER_MONTHLY_CAP_MICRO : TRIAL_MONTHLY_CAP_MICRO;

  // Compute the more-restrictive ratio (D-41)
  const dailyRatio = balance >= dailyCap ? 100 : Number((balance * 100n) / dailyCap);
  const monthlyRatio = balance >= monthlyCap ? 100 : Number((balance * 100n) / monthlyCap);
  const raw = Math.min(dailyRatio, monthlyRatio);
  // Round to nearest 5 per D-41
  return Math.max(0, Math.min(100, Math.round(raw / 5) * 5));
}
```
