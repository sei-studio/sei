1. `proxy/src/env.ts` — Zod validates at boot, fail-fast:
```ts
import { z } from 'zod';

const EnvSchema = z.object({
  PORT: z.string().regex(/^\d+$/).default('8080'),
  ANTHROPIC_API_KEY: z.string().min(1),
  SUPABASE_URL: z.string().url(),
  SUPABASE_SERVICE_ROLE_KEY: z.string().min(1),
  SUPABASE_JWKS_URL: z.string().url(),
  SENTRY_DSN: z.string().optional(),
});

export type Env = z.infer<typeof EnvSchema>;

let cached: Env | null = null;

export function loadEnv(): Env {
  if (cached) return cached;
  const parsed = EnvSchema.safeParse(process.env);
  if (!parsed.success) {
    // Surface ALL missing/invalid env vars at once (not just the first)
    const issues = parsed.error.issues.map(i => `  ${i.path.join('.')}: ${i.message}`).join('\n');
    throw new Error(`Invalid proxy env config:\n${issues}`);
  }
  cached = parsed.data;
  return cached;
}
```

2. `proxy/src/index.ts` — Hono shell with /health + /v1/messages stub:
```ts
import { Hono } from 'hono';
import { serve } from '@hono/node-server';  // ADD to package.json: "@hono/node-server": "1.13.0"

import { loadEnv } from './env.js';

// Fail-fast: throws if env is invalid (caught by Fly.io and surfaced as crash)
const env = loadEnv();

const app = new Hono();

// Health check — no auth, returns 200 immediately (used by fly.toml [checks])
app.get('/health', (c) => c.json({ status: 'ok', version: '1.0.0' }));

// Main upstream pass-through — Wave 2 plans (13-04..13-10) implement.
app.post('/v1/messages', (c) => {
  return c.json({ error: 'not_implemented_yet', plan: '13-10' }, 501);
});

// Generic 404
app.notFound((c) => c.json({ error: 'not_found' }, 404));

// Generic error envelope (no leakage of stack traces)
app.onError((err, c) => {
  console.error('proxy error', { name: err.name, message: err.message });
  return c.json({ error: 'internal_error' }, 500);
});

const port = parseInt(env.PORT, 10);
serve({ fetch: app.fetch, port }, (info) => {
  console.log(`sei-proxy listening on port ${info.port}`);
});
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-03-01 | Information Disclosure | Stack traces in onError handler | mitigate | onError logs internally (console.error with name+message only) and returns generic `{error: 'internal_error'}` — never leak `err.stack` to clients |
| T-13-03-02 | Spoofing | /health endpoint exposes app version | accept | Version is `'1.0.0'` static string; not security-sensitive |

| T-13-03-05 | DoS | Open /health endpoint | accept | Fly.io scale-to-zero auto-scales; per-request cost is negligible |
