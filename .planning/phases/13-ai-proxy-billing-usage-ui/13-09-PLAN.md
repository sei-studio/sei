`proxy/src/middleware/sentinel.ts`:
```ts
import type { Context } from 'hono';
import type { ContentfulStatusCode } from 'hono/utils/http-status';

export type ProxyError =
  | { code: 'missing_jwt' }
  | { code: 'invalid_jwt'; detail?: string }
  | { code: 'expired_jwt' }
  | { code: 'payment_required' }
  | { code: 'rate_limited'; kind: 'rpm' | 'itpm' | 'otpm' | 'daily_dollar'; retry_after_seconds: number }
  | { code: 'service_at_capacity'; retry_after_seconds: number }
  | { code: 'upstream_timeout' }
  | { code: 'upstream_error'; detail?: string }
  | { code: 'internal_error' };

export type ErrorEnvelope = { status: ContentfulStatusCode; body: Record<string, unknown> };

export function errorEnvelope(err: ProxyError): ErrorEnvelope {
  switch (err.code) {
    case 'missing_jwt': return { status: 401, body: { error: 'missing_jwt' } };
    case 'invalid_jwt': return { status: 401, body: { error: 'invalid_jwt', ...(err.detail ? { detail: err.detail } : {}) } };
    case 'expired_jwt': return { status: 401, body: { error: 'expired_jwt' } };
    case 'payment_required': return { status: 402, body: { error: 'payment_required' } };
    case 'rate_limited': return { status: 429, body: { error: 'rate_limited', kind: err.kind, retry_after_seconds: err.retry_after_seconds } };
    case 'service_at_capacity': return { status: 503, body: { error: 'service_at_capacity', retry_after_seconds: err.retry_after_seconds } };
    case 'upstream_timeout': return { status: 504, body: { error: 'upstream_timeout' } };
    case 'upstream_error': return { status: 502, body: { error: 'upstream_error', ...(err.detail ? { detail: err.detail } : {}) } };
    case 'internal_error': return { status: 500, body: { error: 'internal_error' } };
    default: {
      // Exhaustiveness check — if a new sentinel is added without a case, TS errors here.
      const _exhaustive: never = err;
      void _exhaustive;
      return { status: 500, body: { error: 'internal_error' } };
    }
  }
}

export function sendError(c: Context, err: ProxyError): Response {
  const { status, body } = errorEnvelope(err);
  return c.json(body, status);
}

import type { Usage } from './pricing.js';

export type ExtractedUsage = {
  input: Usage | null;
  output: { output_tokens: number } | null;
  messageId: string | null;
};

export function extractUsage(rawChunk: string): ExtractedUsage {
  let input: Usage | null = null;
  let output: { output_tokens: number } | null = null;
  let messageId: string | null = null;

  // SSE events separated by \n\n. Each event has lines: `event: name\ndata: json\n...`
  const events = rawChunk.split('\n\n');
  for (const raw of events) {
    const dataLine = raw.split('\n').find(l => l.trim().startsWith('data: '));
    if (!dataLine) continue;
    let parsed: any;
    try { parsed = JSON.parse(dataLine.trim().slice('data: '.length)); }
    catch { continue; }

    if (raw.includes('message_start')) {
      input = parsed.message?.usage ?? null;
      messageId = parsed.message?.id ?? messageId;
    } else if (raw.includes('message_delta')) {
      output = parsed.usage ?? null;
    }
  }
  return { input, output, messageId };
}

export function mergeUsage(input: Usage | null, output: { output_tokens: number } | null): Usage | null {
  if (!input && !output) return null;
  return {
    input_tokens: input?.input_tokens ?? 0,
    cache_creation_input_tokens: input?.cache_creation_input_tokens ?? 0,
    cache_read_input_tokens: input?.cache_read_input_tokens ?? 0,
    output_tokens: output?.output_tokens ?? 0,
  };
}
```
