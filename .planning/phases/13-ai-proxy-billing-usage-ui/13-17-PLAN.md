```tsx
type PricingIconProps = { size?: number; remainingPct?: number };

export function PricingIcon({ size = 24, remainingPct = 0 }: PricingIconProps): React.ReactElement {
  // Clamp to 0..100, render a wallet glyph with the inner fill rect height varying.
  const pct = Math.max(0, Math.min(100, remainingPct));
  const fillHeight = (pct / 100) * 14; // inner shape is 14 tall in a 24 viewport
  const fillY = 19 - fillHeight;        // grow upward from bottom

  return (
    <svg width={size} height={size} viewBox="0 0 24 24" fill="none" aria-hidden="true">
      {/* Outer rounded wallet shape */}
      <rect x="3" y="6" width="18" height="14" rx="2" stroke="currentColor" strokeWidth="1.5" />
      {/* Wallet flap */}
      <path d="M5 6V4.5A1.5 1.5 0 0 1 6.5 3h11A1.5 1.5 0 0 1 19 4.5V6" stroke="currentColor" strokeWidth="1.5" />
      {/* Inner fill — grows from bottom */}
      <rect x="5" y={fillY} width="14" height={fillHeight} fill="currentColor" opacity="0.4"
            style={{ transition: 'height 200ms ease, y 200ms ease' }} />
    </svg>
  );
}
```

The icon uses `currentColor` so it inherits theme palette (CSS vars from tokens.css).
