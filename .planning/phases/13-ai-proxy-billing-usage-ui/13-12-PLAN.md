```ts
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';

const TRIAL_CREDITS_MICRO = 1_000_000n; // $1 (CONTEXT D-42 + §specifics)
const MC_USERNAME_REGEX = /^[A-Za-z0-9_]{1,16}$/;

export async function handler(req: Request): Promise<Response> {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response(JSON.stringify({ error: 'method_not_allowed' }), { status: 405, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  const authHeader = req.headers.get('Authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    return new Response(JSON.stringify({ error: 'missing_jwt' }), { status: 401, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  const SUPABASE_URL = Deno.env.get('SUPABASE_URL');
  const SUPABASE_ANON_KEY = Deno.env.get('SUPABASE_ANON_KEY');
  const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY');
  if (!SUPABASE_URL || !SUPABASE_ANON_KEY || !SUPABASE_SERVICE_ROLE_KEY) {
    return new Response(JSON.stringify({ error: 'server_misconfigured' }), { status: 500, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  const userClient = createClient(SUPABASE_URL, SUPABASE_ANON_KEY, { global: { headers: { Authorization: authHeader } } });
  const { data: userData, error: userErr } = await userClient.auth.getUser();
  if (userErr || !userData?.user) {
    return new Response(JSON.stringify({ error: 'invalid_jwt' }), { status: 401, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }
  const userId = userData.user.id;

  let body: { mc_username?: unknown };
  try { body = await req.json(); }
  catch { return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: { ...corsHeaders, 'content-type': 'application/json' } }); }

  if (typeof body.mc_username !== 'string' || !MC_USERNAME_REGEX.test(body.mc_username)) {
    return new Response(JSON.stringify({ error: 'bad_request', detail: 'invalid_mc_username' }), { status: 400, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }
  const mcUsername = body.mc_username;

  const admin = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);

  // INSERT claim — UNIQUE on mc_username catches re-claim
  const { error: claimErr } = await admin
    .from('trial_claims')
    .insert({ mc_username: mcUsername, sei_user_id: userId });
  if (claimErr) {
    if (claimErr.code === '23505') {
      return new Response(JSON.stringify({ ok: false, code: 'already_claimed' }), { status: 409, headers: { ...corsHeaders, 'content-type': 'application/json' } });
    }
    console.error('trial-claim claim_insert_failed', { code: claimErr.code, message: claimErr.message });
    return new Response(JSON.stringify({ ok: false, code: 'insert_failed' }), { status: 500, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  // Grant trial credits — ledger_grants_trial_per_user_uidx is defense-in-depth idempotency
  const { error: grantErr } = await admin.from('ledger_grants').insert({
    user_id: userId,
    kind: 'trial',
    credits_micro: TRIAL_CREDITS_MICRO.toString(),
    source: 'trial_claim',
    lemon_event_id: null,
  });
  if (grantErr) {
    if (grantErr.code === '23505') {
      // Defense-in-depth: trial_claims accepted us but a prior trial grant exists.
      // Don't compensating-delete (user got their $1 once). Return 409 idempotent.
      return new Response(JSON.stringify({ ok: false, code: 'already_claimed' }), { status: 409, headers: { ...corsHeaders, 'content-type': 'application/json' } });
    }
    // Compensating delete so the user can retry (PATTERNS atomicity gap)
    await admin.from('trial_claims').delete().eq('mc_username', mcUsername).eq('sei_user_id', userId);
    console.error('trial-claim grant_insert_failed', { code: grantErr.code, message: grantErr.message });
    return new Response(JSON.stringify({ ok: false, code: 'grant_failed' }), { status: 500, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  return new Response(JSON.stringify({ ok: true, credits_micro: Number(TRIAL_CREDITS_MICRO) }), { status: 200, headers: { ...corsHeaders, 'content-type': 'application/json' } });
}

if (import.meta.main) Deno.serve(handler);
```

| T-13-12-01 | Spoofing | Caller sets sei_user_id in body | mitigate | Body field IGNORED — userId always from userClient.auth.getUser() (T-12-05-01) |
