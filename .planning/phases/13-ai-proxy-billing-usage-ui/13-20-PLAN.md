Locate the existing ACCOUNT section in SettingsScreen.tsx (per PATTERNS at lines 224-311). Add a new row BEFORE the dangerSeparator (~line 291):

```tsx
// Top of file imports:
import { useCreditsStore } from '../lib/stores/useCreditsStore';

// Inside component:
const aiBackendKind = useCreditsStore((s) => s.ai_backend_kind);
const plan = useCreditsStore((s) => s.plan);
const cancelSubscription = useCreditsStore((s) => s.cancelSubscription);
const handleSwitchToCloud = async (): Promise<void> => {
  await window.sei.proxyConfigure('cloud-proxy');
  // Refresh credits store so the icon rail appears on next tick
  await useCreditsStore.getState().refresh();
};
const handleSwitchToLocal = async (): Promise<void> => {
  await window.sei.proxyConfigure('local');
  await useCreditsStore.getState().refresh();
};

// In the ACCOUNT section, before <hr className={styles.dangerSeparator} />:
<div className={styles.row}>
  <span className={styles.rowLabel}>Cloud AI</span>
  {aiBackendKind === 'local' ? (
    <Button kind="ghost" size="md" onClick={() => void handleSwitchToCloud()}>
      Switch to managed billing
    </Button>
  ) : plan === 'unlimited' ? (
    <Button kind="ghost" size="md" onClick={() => void cancelSubscription()}>
      Manage subscription
    </Button>
  ) : (
    <Button kind="ghost" size="md" onClick={() => void handleSwitchToLocal()}>
      Switch to your own API key
    </Button>
  )}
</div>
<p className={styles.rowHelper}>
  Use Sei&rsquo;s managed cloud — purchase credits, no API key required.
</p>
```

Defensive: the conditional renders ONE of the three button variants based on (aiBackendKind, plan). All three labels are explicitly enumerated to keep the BYOK escape hatch obvious.
