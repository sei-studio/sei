```ts
import { create } from 'zustand';
import type { CreditsStatus } from '../../../../shared/ipc.js';

type HardStopReason = 'depleted' | 'rate_limited';

interface CreditsState {
  remaining_pct: number;
  plan: CreditsStatus['plan'];
  renews_at: string | null;
  trial_claimed: boolean;
  ai_backend_kind: 'local' | 'cloud-proxy';
  hardStopActive: boolean;
  hardStopReason: HardStopReason | null;
  rateLimitedUntil: number | null;
  initialized: boolean;
  loading: boolean;
  pushSeq: number;
  unsubStatus?: () => void;
  unsubHardStop?: () => void;
}

interface CreditsActions {
  init(): Promise<void>;
  refresh(): Promise<void>;
  openCheckout(kind: 'pack' | 'subscription'): Promise<void>;
  cancelSubscription(): Promise<void>;
  acknowledgeHardStop(): void;
  reset(): void;
}

const INITIAL: CreditsState = {
  remaining_pct: 0,
  plan: 'depleted',
  renews_at: null,
  trial_claimed: false,
  ai_backend_kind: 'local',
  hardStopActive: false,
  hardStopReason: null,
  rateLimitedUntil: null,
  initialized: false,
  loading: false,
  pushSeq: 0,
};

export const useCreditsStore = create<CreditsState & CreditsActions>((set, get) => ({
  ...INITIAL,

  async init(): Promise<void> {
    if (get().initialized) return;
    // 1. Subscribe FIRST (useSyncStore race-guard pattern)
    const unsubStatus = window.sei.onCreditsStatusUpdate((status) => {
      set((s) => ({
        ...s,
        remaining_pct: status.remaining_pct,
        plan: status.plan,
        renews_at: status.renews_at,
        trial_claimed: status.trial_claimed,
        ai_backend_kind: status.ai_backend_kind,
        pushSeq: s.pushSeq + 1,
      }));
    });
    const unsubHardStop = window.sei.onCreditsHardStop((info) => {
      set({
        hardStopActive: true,
        hardStopReason: info.reason,
        rateLimitedUntil: info.reason === 'rate_limited' && info.retry_after_seconds
          ? Date.now() + info.retry_after_seconds * 1000
          : null,
      });
    });
    set({ unsubStatus, unsubHardStop });

    // 2. Seed; skip applying if a push has already arrived
    const seqBefore = get().pushSeq;
    set({ loading: true });
    const status = await window.sei.creditsGet();
    if (get().pushSeq === seqBefore) {
      set((s) => ({ ...s, ...status, initialized: true, loading: false }));
    } else {
      set({ initialized: true, loading: false }); // push won; keep its values
    }
  },

  async refresh(): Promise<void> {
    set({ loading: true });
    const status = await window.sei.creditsGet();
    set((s) => ({ ...s, ...status, loading: false }));
  },

  async openCheckout(kind): Promise<void> {
    await window.sei.creditsOpenCheckout(kind);
  },

  async cancelSubscription(): Promise<void> {
    await window.sei.subscriptionCancel();
  },

  acknowledgeHardStop(): void {
    set({ hardStopActive: false, hardStopReason: null });
  },

  reset(): void {
    const { unsubStatus, unsubHardStop } = get();
    unsubStatus?.();
    unsubHardStop?.();
    set({ ...INITIAL });
  },
}));
```
