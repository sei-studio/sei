```ts
import type { Character } from '../../../shared/characterSchema';

// CONTEXT §specifics templates for bundled personas. Bundled persona IDs come from the seed data;
// they're stable strings shipped with the app. For v1.0 we use persona.name lowercased as a fallback key.
const BUNDLED: Record<string, string> = {
  sui:    "Sui's hungry for more conversation — top up to keep her chatting?",
  lyra:   "Lyra would love to keep chatting — top up?",
  clawd:  "Clawd needs more tokens to keep being chaotic — top up?",
};

const GENERIC = "You're out of credits — top up to keep your bot running.";

export function resolveHardStopCopy(persona: Character | null | undefined): string {
  if (!persona) return GENERIC;
  // Bundled persona — match by lowercased name (CONTEXT lists Sui/Lyra/Clawd as the bundled set)
  const key = (persona.name ?? '').toLowerCase();
  if (BUNDLED[key]) return BUNDLED[key];
  // Custom persona — interpolate the persona's name
  if (persona.name) return `${persona.name} needs more credits to keep talking.`;
  return GENERIC;
}
```

NOTE: this file references the bundled-persona NAMES. If the project uses a different match key (e.g., a stable bundled `is_default` flag + persona id), this match function should adapt. Use whichever signal is stable across rebuilds.

```tsx
import { useEffect, useId } from 'react';
import { useCreditsStore } from '../lib/stores/useCreditsStore';
import { useDataStore } from '../lib/stores/useDataStore';
import { Button } from './Button';
import { resolveHardStopCopy } from './hardStopCopy';
import styles from './HardStopModal.module.css';

export function HardStopModal(): React.ReactElement | null {
  const hardStopActive = useCreditsStore((s) => s.hardStopActive);
  const hardStopReason = useCreditsStore((s) => s.hardStopReason);
  const remainingPct = useCreditsStore((s) => s.remaining_pct);
  const openCheckout = useCreditsStore((s) => s.openCheckout);
  const acknowledgeHardStop = useCreditsStore((s) => s.acknowledgeHardStop);
  const activeCharacter = useDataStore((s) => s.activeCharacter);
  const titleId = useId();

  // ESC suppression (mirrors AcceptToSModal:54-62)
  useEffect(() => {
    if (!hardStopActive) return;
    const onKey = (e: KeyboardEvent): void => {
      if (e.key === 'Escape') e.preventDefault();
    };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, [hardStopActive]);

  // Auto-dismiss when balance refills (only for 'depleted', not 'rate_limited')
  useEffect(() => {
    if (hardStopActive && hardStopReason === 'depleted' && remainingPct > 0) {
      acknowledgeHardStop();
    }
  }, [hardStopActive, hardStopReason, remainingPct, acknowledgeHardStop]);

  if (!hardStopActive) return null;

  const copy = resolveHardStopCopy(activeCharacter);
  const handleSwitchToLocal = async (): Promise<void> => {
    await window.sei.proxyConfigure('local');
    acknowledgeHardStop();
  };

  return (
    <div className={styles.scrim} role="dialog" aria-modal="true" aria-labelledby={titleId}>
      <div className={styles.modal}>
        <h2 id={titleId} className={styles.title}>You&rsquo;re out of credits</h2>
        <p className={styles.body}>{copy}</p>
        <div className={styles.footer}>
          <Button kind="accent" onClick={() => void openCheckout('pack')}>Top up</Button>
          <Button kind="primary" onClick={() => void openCheckout('subscription')}>Go Unlimited</Button>
          <Button kind="ghost" className={styles.muted} onClick={() => void handleSwitchToLocal()}>
            Switch to your own API key
          </Button>
        </div>
      </div>
    </div>
  );
}
```

NOTE on the CTA labels: the BUTTON labels intentionally say "Top up" + "Go Unlimited" (NO dollar amounts) for PROXY-05 strict-no-pricing enforcement IN THE MODAL itself. The Lemon Squeezy product pages (opened via shell.openExternal) display the actual dollar amounts.
