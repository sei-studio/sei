2. `src/main/cloud/proxyClient.ts`:
```ts
import { shell } from 'electron';
import { callEdgeFunction } from '../auth/edgeFunctionClient.js';
import { getClient } from '../auth/supabaseClient.js';
import { PROXY_NO_SESSION, PROXY_ALREADY_CLAIMED, PROXY_NETWORK, type ProxyErrorCode } from './proxyErrors.js';
import type { CreditsStatus, SubscriptionStatusInfo } from '../../shared/ipc.js';

const REMAINING_PCT_STEP = 5;

// Tier caps (mirror proxy/src/ledger/balance.ts)
const TRIAL_DAILY_CAP = 5_000_000n;
const SUB_DAILY_CAP = 20_000_000n;

function roundToStep(n: number, step: number): number {
  return Math.max(0, Math.min(100, Math.round(n / step) * step));
}

function buildCheckoutUrl(kind: 'pack' | 'subscription', userId: string): string {
  const variant = kind === 'pack' ? process.env.LEMON_VARIANT_PACK : process.env.LEMON_VARIANT_SUBSCRIPTION;
  if (!variant) throw new Error('LEMON_VARIANT_* env not configured');
  const url = new URL(`https://sei.lemonsqueezy.com/buy/${variant}`);
  url.searchParams.set('checkout[custom][user_id]', userId);
  return url.toString();
}

async function getSessionOrNull(): Promise<{ jwt: string; userId: string } | null> {
  const supabase = getClient();
  const { data } = await supabase.auth.getSession();
  if (!data.session) return null;
  return { jwt: data.session.access_token, userId: data.session.user.id };
}

export async function trialClaim(mcUsername: string): Promise<{ ok: true; credits_micro: number } | { ok: false; code: ProxyErrorCode }> {
  const session = await getSessionOrNull();
  if (!session) return { ok: false, code: PROXY_NO_SESSION };
  const res = await callEdgeFunction('trial-claim', session.jwt, { mc_username: mcUsername });
  if (!res.ok) {
    if (res.status === 409) return { ok: false, code: PROXY_ALREADY_CLAIMED };
    if (res.status === 0) return { ok: false, code: PROXY_NETWORK };
    return { ok: false, code: PROXY_NETWORK };
  }
  const data = res.data as { ok?: boolean; code?: string; credits_micro?: number };
  if (data.ok && typeof data.credits_micro === 'number') {
    return { ok: true, credits_micro: data.credits_micro };
  }
  if (data.code === 'already_claimed') return { ok: false, code: PROXY_ALREADY_CLAIMED };
  return { ok: false, code: PROXY_NETWORK };
}

export async function creditsGet(): Promise<CreditsStatus> {
  const session = await getSessionOrNull();
  if (!session) {
    return { remaining_pct: 0, plan: 'depleted', renews_at: null, trial_claimed: false, ai_backend_kind: 'local' };
  }
  const supabase = getClient();
  const [balanceRow, subRow, trialRow, backendRes] = await Promise.all([
    supabase.from('ledger_balance').select('balance_micro').eq('user_id', session.userId).maybeSingle(),
    supabase.from('subscription_status').select('status,renews_at,ends_at').eq('user_id', session.userId).maybeSingle(),
    supabase.from('trial_claims').select('mc_username').eq('sei_user_id', session.userId).maybeSingle(),
    import('../apiKeyStore.js').then(m => m.getAiBackendKind()),
  ]);
  const balance = BigInt(balanceRow.data?.balance_micro ?? 0);
  const isSubscriber = subRow.data?.status === 'active';
  const dailyCap = isSubscriber ? SUB_DAILY_CAP : TRIAL_DAILY_CAP;
  const pctRaw = balance >= dailyCap ? 100 : Number((balance * 100n) / dailyCap);
  const pct = roundToStep(pctRaw, REMAINING_PCT_STEP);
  const plan: CreditsStatus['plan'] = isSubscriber ? 'unlimited'
    : balance === 0n ? 'depleted'
    : trialRow.data ? 'trial'
    : 'pack';
  return {
    remaining_pct: pct,
    plan,
    renews_at: subRow.data?.renews_at ?? null,
    trial_claimed: !!trialRow.data,
    ai_backend_kind: backendRes,
  };
}

export async function openCheckout(kind: 'pack' | 'subscription'): Promise<{ ok: true } | { ok: false; code: string }> {
  const session = await getSessionOrNull();
  if (!session) return { ok: false, code: PROXY_NO_SESSION };
  try {
    const url = buildCheckoutUrl(kind, session.userId);
    await shell.openExternal(url);
    return { ok: true };
  } catch (err) {
    return { ok: false, code: PROXY_NETWORK };
  }
}

export async function subscriptionStatus(): Promise<SubscriptionStatusInfo> {
  const session = await getSessionOrNull();
  if (!session) return { active: false, status: 'none', renews_at: null, ends_at: null };
  const supabase = getClient();
  const { data } = await supabase.from('subscription_status').select('status,renews_at,ends_at').eq('user_id', session.userId).maybeSingle();
  if (!data) return { active: false, status: 'none', renews_at: null, ends_at: null };
  return {
    active: data.status === 'active',
    status: data.status as SubscriptionStatusInfo['status'],
    renews_at: data.renews_at,
    ends_at: data.ends_at,
  };
}

export async function cancelSubscription(): Promise<{ ok: true; portalUrl: string } | { ok: false; code: string }> {
  // Open-question resolution #5: open LS customer portal externally; Sei does NOT implement cancel endpoint.
  const portalUrl = 'https://sei.lemonsqueezy.com/billing';
  try {
    await shell.openExternal(portalUrl);
    return { ok: true, portalUrl };
  } catch {
    return { ok: false, code: PROXY_NETWORK };
  }
}
```

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
