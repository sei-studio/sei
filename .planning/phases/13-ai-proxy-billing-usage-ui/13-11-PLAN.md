```ts
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';

// Net credit grants per CONTEXT additional + open-question resolution #3
const PACK_CREDITS_MICRO = 4_750_000n;
const SUBSCRIPTION_CREDITS_MICRO = 18_500_000n;

async function hmacSha256Hex(secret: string, rawBody: string): Promise<string> {
  const enc = new TextEncoder();
  const key = await crypto.subtle.importKey('raw', enc.encode(secret), { name: 'HMAC', hash: 'SHA-256' }, false, ['sign']);
  const sig = await crypto.subtle.sign('HMAC', key, enc.encode(rawBody));
  return Array.from(new Uint8Array(sig)).map(b => b.toString(16).padStart(2, '0')).join('');
}

function timingSafeEqual(a: string, b: string): boolean {
  if (a.length !== b.length) return false;
  let diff = 0;
  for (let i = 0; i < a.length; i++) diff |= a.charCodeAt(i) ^ b.charCodeAt(i);
  return diff === 0;
}

async function postDiscordAlert(message: string): Promise<void> {
  const url = Deno.env.get('DISCORD_BILLING_ALERT_WEBHOOK_URL');
  if (!url) return;
  try {
    await fetch(url, {
      method: 'POST',
      headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ content: message }),
      signal: AbortSignal.timeout(15_000),
    });
  } catch (err) {
    console.error('discord_post_failed', err);
  }
}

export async function handler(req: Request): Promise<Response> {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response(JSON.stringify({ error: 'method_not_allowed' }), { status: 405, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  const signature = req.headers.get('X-Signature');
  if (!signature) {
    return new Response(JSON.stringify({ error: 'missing_signature' }), { status: 401, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  // RESEARCH Pitfall 2: read RAW body for HMAC verification; req.json() would consume it.
  const rawBody = await req.text();
  const secret = Deno.env.get('LEMON_SQUEEZY_WEBHOOK_SECRET');
  if (!secret) {
    console.error('LEMON_SQUEEZY_WEBHOOK_SECRET not configured');
    return new Response(JSON.stringify({ error: 'server_misconfigured' }), { status: 500, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }
  const expected = await hmacSha256Hex(secret, rawBody);
  if (!timingSafeEqual(signature, expected)) {
    return new Response(JSON.stringify({ error: 'invalid_signature' }), { status: 401, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  let payload: any;
  try { payload = JSON.parse(rawBody); }
  catch { return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: { ...corsHeaders, 'content-type': 'application/json' } }); }

  const eventName: string = payload?.meta?.event_name ?? '';
  const eventId: string = payload?.meta?.webhook_id ?? `${eventName}-${payload?.data?.id ?? 'unknown'}`;
  const userId: string | undefined = payload?.meta?.custom_data?.user_id;

  if (!userId || !/^[0-9a-f-]{36}$/i.test(userId)) {
    // PATTERNS Pitfall 4: missing/invalid custom_data.user_id — alert and return 202 (don't make LS retry)
    console.warn('lemon-webhook missing_user_id', { eventName, eventId });
    await postDiscordAlert(`⚠️ Lemon webhook missing custom_data.user_id — event=${eventName} id=${eventId}`);
    return new Response(JSON.stringify({ ok: false, code: 'missing_user_id' }), { status: 202, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  const admin = createClient(Deno.env.get('SUPABASE_URL')!, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!);

  try {
    switch (eventName) {
      case 'order_created': {
        const { error } = await admin.from('ledger_grants').insert({
          user_id: userId,
          kind: 'pack',
          credits_micro: PACK_CREDITS_MICRO.toString(),
          source: 'lemon_squeezy',
          lemon_event_id: eventId,
        });
        if (error && error.code !== '23505') throw error;
        break;
      }
      case 'subscription_created':
      case 'subscription_payment_success': {
        const renewsAt = payload?.data?.attributes?.renews_at ?? null;
        const lemonSubId = String(payload?.data?.id ?? '');
        const { error: grantErr } = await admin.from('ledger_grants').insert({
          user_id: userId,
          kind: 'subscription',
          credits_micro: SUBSCRIPTION_CREDITS_MICRO.toString(),
          source: 'lemon_squeezy',
          lemon_event_id: eventId,
        });
        if (grantErr && grantErr.code !== '23505') throw grantErr;
        await admin.from('subscription_status').upsert({
          user_id: userId,
          status: 'active',
          lemon_subscription_id: lemonSubId,
          renews_at: renewsAt,
          ends_at: null,
        });
        break;
      }
      case 'subscription_updated': {
        const status = String(payload?.data?.attributes?.status ?? 'active');
        await admin.from('subscription_status').upsert({
          user_id: userId,
          status: ['active','cancelled','expired','past_due'].includes(status) ? status : 'active',
          lemon_subscription_id: String(payload?.data?.id ?? ''),
          renews_at: payload?.data?.attributes?.renews_at ?? null,
          ends_at: payload?.data?.attributes?.ends_at ?? null,
        });
        break;
      }
      case 'subscription_cancelled':
      case 'subscription_expired': {
        const status = eventName === 'subscription_cancelled' ? 'cancelled' : 'expired';
        await admin.from('subscription_status').upsert({
          user_id: userId,
          status,
          lemon_subscription_id: String(payload?.data?.id ?? ''),
          ends_at: payload?.data?.attributes?.ends_at ?? null,
        });
        break;
      }
      default:
        console.log('lemon-webhook ignored event', { eventName, eventId });
        break;
    }
  } catch (err: any) {
    console.error('lemon-webhook insert_failed', { eventName, eventId, code: err?.code, message: err?.message });
    await postDiscordAlert(`❌ Lemon webhook insert_failed — event=${eventName} id=${eventId} code=${err?.code}`);
    // Always-200 policy (PATTERNS Anti-Pattern §4) — log + alert but don't make LS retry
    return new Response(JSON.stringify({ ok: false, code: 'insert_failed' }), { status: 200, headers: { ...corsHeaders, 'content-type': 'application/json' } });
  }

  return new Response(JSON.stringify({ ok: true }), { status: 200, headers: { ...corsHeaders, 'content-type': 'application/json' } });
}

if (import.meta.main) Deno.serve(handler);
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Lemon Squeezy → webhook | HMAC-SHA256 signature on raw body is the auth model |
| webhook → ledger_grants | service_role insert; UNIQUE(lemon_event_id) idempotency |
| webhook → subscription_status | service_role upsert |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-11-01 | Spoofing | Forged webhook (no signature) | mitigate | Missing X-Signature → 401; signature verified via HMAC-SHA256 over RAW body before JSON.parse |
| T-13-11-02 | Tampering | Modified payload after capture | mitigate | HMAC over rawBody; timingSafeEqual prevents char-by-char inference |
| T-13-11-03 | Replay | Duplicate webhook delivery | mitigate | ledger_grants.lemon_event_id UNIQUE → 23505 on retry → handler returns 200 ok |
| T-13-11-04 | Spoofing | meta.custom_data.user_id forged at checkout | mitigate | user_id round-trips from checkout query param (controlled by Sei main process); webhook validates UUID format; non-UUID → 202 missing_user_id with Discord alert. Defense in depth: would require attacker to control BOTH checkout URL construction AND have a real LS sale on file |
| T-13-11-05 | DoS | Retry-storm on 5xx | mitigate | Always 200/202 on application failures (insert_failed → 200 with Discord alert); only 401/405/400 short-circuit |
| T-13-11-06 | Information Disclosure | Discord webhook URL in logs | mitigate | postDiscordAlert never console.errors the URL; only the message body |
| T-13-11-07 | Tampering | Negative or huge credits_micro | mitigate | Constants PACK_CREDITS_MICRO + SUBSCRIPTION_CREDITS_MICRO are compile-time bigints; payload's `total` field is ignored — we don't trust LS-reported amounts for grant size |
| T-13-11-08 | Spoofing | user_id NOT in auth.users (orphan grant) | accept | Foreign key constraint on ledger_grants.user_id → auth.users(id) will raise on insert; handler returns 200 + Discord alert. Manual reconciliation per PATTERNS Pitfall 4 |
</threat_model>
