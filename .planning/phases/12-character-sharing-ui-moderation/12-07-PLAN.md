```typescript
/**
 * Pre-publish moderation orchestrator (Phase 12 — SHARE-05 / SHARE-06 / SHARE-07).
 *
 * Order of operations (mandatory; tested):
 *   1. Image moderation (Edge Function `moderate-character-images`) —
 *      portraitUrl is DERIVED from `${SUPABASE_URL}/storage/v1/object/public/portraits/${ownerUuid}/${characterId}.png`
 *      so caller cannot point at a different image (T-12-03-01).
 *      Pitfall 12: provider 502 → hard-fail; never publish un-scanned.
 *   2. Prompt moderation hard tier (name + persona_source).
 *   3. Prompt moderation soft tier (persona_expanded) — if flagged, caller
 *      regenerates with mod-steering prompt and re-calls; cap = 2 (Pitfall 6).
 *   4. upsertCharacter with shared=true + moderation_* columns set.
 *
 * Dependency-injected for testability; production uses real cloudCharacterClient
 * + edgeFunctionClient.
 */
import {
  CLOUD_MODERATION_IMAGE_FLAGGED,
  CLOUD_MODERATION_PROMPT_FLAGGED,
  CLOUD_MODERATION_PROVIDER_UNAVAILABLE,
} from './cloudErrors';

const SOFT_RETRY_CAP = 2;  // Pitfall 6
const MODERATION_TIMEOUT_MS = 30_000;  // 30s — SightEngine + OpenAI both ~5-10s

export type PublishResult =
  | { ok: true; moderationProvider: string; textProvider: string }
  | { ok: false; code: typeof CLOUD_MODERATION_IMAGE_FLAGGED | typeof CLOUD_MODERATION_PROMPT_FLAGGED | typeof CLOUD_MODERATION_PROVIDER_UNAVAILABLE; friendlyMessage: string };

export interface PublishDeps {
  callEdgeFunction: <T>(name: string, opts: { jwt: string; body: unknown; timeoutMs?: number }) => Promise<T>;
  upsertCharacter: (character: any, ownerUuid: string) => Promise<void>;
  reExpandPersona: (characterId: string) => Promise<string>;  // soft-regen hook (existing expandAndSaveCharacter)
  getCharacter: (characterId: string) => Promise<{ id: string; owner: string; name: string; persona_source: string; persona_expanded: string }>;
  getJwt: () => Promise<string>;
  supabaseUrl: string;
}

const FRIENDLY_IMAGE = 'Image flagged by automated review — please use a different portrait.';
const FRIENDLY_PROMPT = "We can't publish this character because the persona description hits our content guidelines. Edit the persona and try again, or save it as private.";
const FRIENDLY_UNAVAILABLE = 'Moderation service is temporarily unavailable. Please try again in a few minutes.';

export async function publishWithModeration(
  characterId: string,
  deps: PublishDeps,
): Promise<PublishResult> {
  const char = await deps.getCharacter(characterId);
  const jwt = await deps.getJwt();
  const portraitUrl = `${deps.supabaseUrl}/storage/v1/object/public/portraits/${char.owner}/${characterId}.png`;

  // --- 1. Image moderation (Pitfall 12: hard-fail on provider error) ---
  let imageResult: { status: 'clean' | 'flagged'; provider: string };
  try {
    imageResult = await deps.callEdgeFunction('moderate-character-images', {
      jwt,
      body: { characterId, portraitUrl },
      timeoutMs: MODERATION_TIMEOUT_MS,
    });
  } catch {
    return { ok: false, code: CLOUD_MODERATION_PROVIDER_UNAVAILABLE, friendlyMessage: FRIENDLY_UNAVAILABLE };
  }
  if (imageResult.status === 'flagged') {
    return { ok: false, code: CLOUD_MODERATION_IMAGE_FLAGGED, friendlyMessage: FRIENDLY_IMAGE };
  }

  // --- 2. Prompt moderation (hard + soft with retry cap) ---
  let personaExpanded = char.persona_expanded;
  for (let attempt = 0; attempt <= SOFT_RETRY_CAP; attempt += 1) {
    let promptResult: { verdict: 'clean' | 'block' | 'regenerate'; tier?: string };
    try {
      promptResult = await deps.callEdgeFunction('moderate-character-prompt', {
        jwt,
        body: { name: char.name, persona_source: char.persona_source, persona_expanded: personaExpanded },
        timeoutMs: MODERATION_TIMEOUT_MS,
      });
    } catch {
      return { ok: false, code: CLOUD_MODERATION_PROVIDER_UNAVAILABLE, friendlyMessage: FRIENDLY_UNAVAILABLE };
    }
    if (promptResult.verdict === 'block') {
      return { ok: false, code: CLOUD_MODERATION_PROMPT_FLAGGED, friendlyMessage: FRIENDLY_PROMPT };
    }
    if (promptResult.verdict === 'clean') {
      break;  // success path
    }
    // verdict === 'regenerate' → re-expand and retry (cap = SOFT_RETRY_CAP)
    if (attempt === SOFT_RETRY_CAP) {
      return { ok: false, code: CLOUD_MODERATION_PROMPT_FLAGGED, friendlyMessage: FRIENDLY_PROMPT };
    }
    personaExpanded = await deps.reExpandPersona(characterId);
  }

  // --- 3. All clean → upsert with shared=true ---
  await deps.upsertCharacter(
    {
      ...char,
      persona_expanded: personaExpanded,
      shared: true,
      moderation_status: 'clean',
      moderation_checked_at: new Date().toISOString(),
      moderation_provider: imageResult.provider,
      moderation_text_provider: 'openai-omni-moderation-latest',
      moderation_text_checked_at: new Date().toISOString(),
    },
    char.owner,
  );

  return { ok: true, moderationProvider: imageResult.provider, textProvider: 'openai-omni-moderation-latest' };
}
```
