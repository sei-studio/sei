```typescript
// supabase/functions/_shared/moderationProviders.ts
//
// Shared SightEngine + OpenAI Moderation call surface.
// Used by: backfill-moderate-existing, moderate-character-images,
// moderate-character-prompt.
//
// Pitfall 1 (12-RESEARCH): SightEngine has NO model called 'minor'.
// Correct: models=nudity-2.1,face-attributes; minor signal lives at
// data.faces[i].attributes.minor (0..1 float).
//
// Every external call wrapped in AbortController with 10s timeout
// (CLAUDE.md invariant: every external call has a timeout).

export const PROVIDER_TIMEOUT_MS = 10_000;

export interface SightEngineResult {
  flagged: boolean;
  provider: 'sightengine-v2.1+face-attributes';
  raw: unknown;
}

export interface OpenAIModerationResult {
  blocked: boolean;
  provider: 'openai-omni-moderation-latest';
  flaggedCategories: string[];
  raw: unknown;
}

export async function callSightEngine(portraitUrl: string): Promise<SightEngineResult> {
  const params = new URLSearchParams({
    url: portraitUrl,
    models: 'nudity-2.1,face-attributes',
    api_user: Deno.env.get('SIGHTENGINE_API_USER')!,
    api_secret: Deno.env.get('SIGHTENGINE_API_SECRET')!,
  });
  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), PROVIDER_TIMEOUT_MS);
  try {
    const r = await fetch(`https://api.sightengine.com/1.0/check.json?${params}`, { signal: ctrl.signal });
    if (!r.ok) throw new Error(`sightengine_http_${r.status}`);
    const data = await r.json();
    return { flagged: interpretSightEngineFlag(data), provider: 'sightengine-v2.1+face-attributes', raw: data };
  } finally {
    clearTimeout(timer);
  }
}

export function interpretSightEngineFlag(data: any): boolean {
  // Block rules per Pattern 4 in 12-RESEARCH.md:
  //   minor face (>0.5) + any non-trivial sexualization → hard block
  const hasMinorFace = (data.faces ?? []).some(
    (f: any) => (f?.attributes?.minor ?? 0) > 0.5,
  );
  const sexualSignal =
    (data.nudity?.sexual_activity ?? 0) > 0.3 ||
    (data.nudity?.sexual_display ?? 0) > 0.3 ||
    (data.nudity?.erotica ?? 0) > 0.3 ||
    (data.nudity?.very_suggestive ?? 0) > 0.3 ||
    (data.nudity?.suggestive ?? 0) > 0.5;
  return hasMinorFace && sexualSignal;
}

export async function callOpenAIModeration(text: string): Promise<OpenAIModerationResult> {
  const ctrl = new AbortController();
  const timer = setTimeout(() => ctrl.abort(), PROVIDER_TIMEOUT_MS);
  try {
    const r = await fetch('https://api.openai.com/v1/moderations', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${Deno.env.get('OPENAI_API_KEY')!}`,
      },
      body: JSON.stringify({ model: 'omni-moderation-latest', input: text }),
      signal: ctrl.signal,
    });
    if (!r.ok) throw new Error(`openai_moderation_http_${r.status}`);
    const data = await r.json();
    const result = data.results?.[0] ?? {};
    return {
      blocked: interpretOpenAIModerationFlag(result.category_scores ?? {}),
      provider: 'openai-omni-moderation-latest',
      flaggedCategories: Object.entries(result.categories ?? {}).filter(([, v]) => v).map(([k]) => k),
      raw: data,
    };
  } finally {
    clearTimeout(timer);
  }
}

export function interpretOpenAIModerationFlag(scores: Record<string, number>): boolean {
  // D-33 thresholds: sexual/minors is ZERO-TOLERANCE.
  return (
    (scores['sexual/minors'] ?? 0) > 0 ||
    (scores['violence/graphic'] ?? 0) > 0.85 ||
    (scores['hate/threatening'] ?? 0) > 0.85 ||
    (scores['self-harm/intent'] ?? 0) > 0.85
  );
}
```

```typescript
/**
 * Edge Function: backfill-moderate-existing
 *
 * One-shot retroactive scan closing Phase 11 D-30. Walks
 * `characters WHERE shared = true AND moderation_status IS NULL`
 * in `created_at ASC` order (Pitfall in 12-PATTERNS.md: avoids the
 * `characters_set_updated_at` trigger flooding Browse ordering when
 * many rows complete moderation in a short window) and updates each
 * row's moderation_status to 'clean' | 'flagged'.
 *
 * Idempotent: re-runs only see still-NULL rows.
 * Resumable: caps at 100 rows/invocation; returns nextCursor for
 *   operator to drive a loop until processed === 0.
 *
 * Operator runbook:
 *   while true; do
 *     RESULT=$(curl -X POST -H "Authorization: Bearer $SERVICE_ROLE" \
 *       https://<ref>.supabase.co/functions/v1/backfill-moderate-existing)
 *     PROCESSED=$(echo $RESULT | jq .processed)
 *     [ "$PROCESSED" = "0" ] && break
 *   done
 *   psql -c "select count(*) from characters where shared=true and moderation_status is null"
 *   # MUST return 0 before flipping BROWSE_ENABLED (D-36 gate b).
 */
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';
import { callSightEngine, callOpenAIModeration } from '../_shared/moderationProviders.ts';

const BATCH_SIZE = 100;
const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response('Method not allowed', { status: 405, headers: corsHeaders });
  }

  const authHeader = req.headers.get('Authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    return new Response(JSON.stringify({ error: 'missing_jwt' }), { status: 401, headers: corsHeaders });
  }

  const admin = createClient(SUPABASE_URL, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!);

  // Pitfall in 12-PATTERNS.md: order by created_at ASC so historical
  // characters preserve chronology even though updated_at trigger fires
  // on every UPDATE.
  const { data: rows, error: selErr } = await admin
    .from('characters')
    .select('id, owner, name, persona_source, portrait_image')
    .eq('shared', true)
    .is('moderation_status', null)
    .order('created_at', { ascending: true })
    .limit(BATCH_SIZE);

  if (selErr) {
    return new Response(JSON.stringify({ error: 'select_failed', detail: selErr.message }), {
      status: 500,
      headers: { ...corsHeaders, 'Content-Type': 'application/json' },
    });
  }

  let processed = 0;
  let flagged = 0;
  let errors = 0;

  for (const row of rows ?? []) {
    try {
      // 1. Image scan (portrait_image is a relative Storage path; build the public URL).
      let imageFlagged = false;
      if (row.portrait_image) {
        const portraitUrl =
          `${SUPABASE_URL}/storage/v1/object/public/portraits/${row.portrait_image}`;
        const img = await callSightEngine(portraitUrl);
        imageFlagged = img.flagged;
      }

      // 2. Prompt scan (name + persona_source per D-33b hard tier).
      const prompt = `${row.name}\n\n${row.persona_source ?? ''}`;
      const text = await callOpenAIModeration(prompt);
      const promptFlagged = text.blocked;

      const finalStatus = imageFlagged || promptFlagged ? 'flagged' : 'clean';

      const update: Record<string, unknown> = {
        moderation_status: finalStatus,
        moderation_checked_at: new Date().toISOString(),
        moderation_provider: 'sightengine-v2.1+face-attributes',
        moderation_text_provider: 'openai-omni-moderation-latest',
        moderation_text_checked_at: new Date().toISOString(),
      };
      if (finalStatus === 'flagged') update.shared = false;

      const { error: updErr } = await admin.from('characters').update(update).eq('id', row.id);
      if (updErr) {
        errors += 1;
        continue;
      }

      processed += 1;
      if (finalStatus === 'flagged') flagged += 1;
    } catch (e) {
      errors += 1;
      console.error('backfill row error', row.id, (e as Error).message);
    }
  }

  return new Response(
    JSON.stringify({
      processed,
      flagged,
      errors,
      nextCursor: (rows ?? []).length === BATCH_SIZE ? 'more' : 'done',
    }),
    { headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
  );
});
```
