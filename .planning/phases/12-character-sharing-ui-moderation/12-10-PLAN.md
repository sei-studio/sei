```tsx
// src/renderer/src/screens/CharactersScreen.tsx
import { useEffect, useRef, useState } from 'react';
import { useBrowseStore } from '../lib/stores/useBrowseStore';
import styles from './CharactersScreen.module.css';
// (carry over all imports from HomeScreen.tsx for HomeGrid)
// ... existing HomeScreen imports

type Tab = 'home' | 'browse';

export function CharactersScreen() {
  const [tab, setTab] = useState<Tab>('home');
  const [browseEnabled, setBrowseEnabled] = useState(false);

  useEffect(() => {
    let cancelled = false;
    window.sei.getCapabilities().then((c) => { if (!cancelled) setBrowseEnabled(c.browseEnabled); });
    return () => { cancelled = true; };
  }, []);

  return (
    <div className={styles.screen}>
      {browseEnabled ? (
        <div className={styles.tabBar} role="tablist">
          <button
            role="tab"
            aria-selected={tab === 'home'}
            className={tab === 'home' ? styles.tabActive : styles.tab}
            onClick={() => setTab('home')}
          >Home</button>
          <button
            role="tab"
            aria-selected={tab === 'browse'}
            className={tab === 'browse' ? styles.tabActive : styles.tab}
            onClick={() => setTab('browse')}
          >Browse</button>
        </div>
      ) : null}
      {tab === 'home' || !browseEnabled ? <HomeGrid /> : <BrowseGrid />}
    </div>
  );
}

function HomeGrid() {
  // Lift the entire body of the existing HomeScreen component here verbatim.
  // Keep useEffect that fires sei.charsListMerged(); keep local cloudOnly
  // state; keep the LAN pill + '+ New' header + character cards + sync pill.
  // No behavior change.
  // ...
  return <div>{/* existing HomeScreen JSX */}</div>;
}

function BrowseGrid() {
  const { entries, query, loading, exhausted, error, setQuery, loadMore, refresh } = useBrowseStore();
  const sentinelRef = useRef<HTMLDivElement>(null);

  // Mount-once refresh (store does NOT self-bootstrap per 12-09 contract).
  useEffect(() => { void refresh(); }, []);  // eslint-disable-line react-hooks/exhaustive-deps

  // IntersectionObserver sentinel for infinite scroll.
  useEffect(() => {
    const el = sentinelRef.current;
    if (!el) return;
    const obs = new IntersectionObserver(
      (e) => { if (e[0].isIntersecting) void loadMore(); },
      { rootMargin: '200px' },
    );
    obs.observe(el);
    return () => obs.disconnect();
  }, [loadMore]);

  return (
    <div className={styles.browse}>
      <input
        className={styles.searchField}
        type="search"
        placeholder="Search characters..."
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        aria-label="Search public characters"
      />
      {error ? <div className={styles.error}>Couldn't load Browse. {error}</div> : null}
      {entries.length === 0 && !loading && !error ? (
        <div className={styles.empty}>No public characters yet — be the first to share one.</div>
      ) : null}
      <div className={styles.grid}>
        {entries.map((entry) => (
          // 12-11 replaces this placeholder with <BrowseCard entry={entry} ... />
          <div key={entry.id} className={styles.cardPlaceholder}>{entry.name}</div>
        ))}
      </div>
      {!exhausted ? <div ref={sentinelRef} className={styles.sentinel} aria-hidden /> : null}
      {loading ? <div className={styles.loading}>Loading…</div> : null}
    </div>
  );
}
```
