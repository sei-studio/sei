```typescript
ipcMain.handle(IpcChannel.app.openExternal, async (_event, urlArg: unknown): Promise<void> => {
  const url = z.string().min(1).parse(urlArg);

  // mailto: special-case — Sei opens only the canonical DMCA mailbox.
  if (url.startsWith('mailto:')) {
    const allowedMailto = ['mailto:dmca@sei.app'];
    if (!allowedMailto.includes(url.toLowerCase())) {
      throw new Error(`app:open-external rejected mailto: ${url}`);
    }
    const { shell } = await import('electron');
    await shell.openExternal(url);
    return;
  }

  // https: — host allowlist.
  const u = new URL(url);
  const allowedHosts = ['sei.gg', 'www.sei.gg', 'dmca.copyright.gov'];
  if (u.protocol !== 'https:' || !allowedHosts.includes(u.hostname)) {
    throw new Error(`app:open-external rejected: ${url}`);
  }
  const { shell } = await import('electron');
  await shell.openExternal(url);
});
```

Critical: `z.string().url()` from the original handler REJECTS mailto: URIs (Zod url() is HTTP-only). Switch to `z.string().min(1)` and rely on the explicit scheme branch.
