```tsx
/**
 * ReportModal — SHARE-08 user-facing report submission.
 *
 * Mount point: CharactersScreen BrowseGrid when reportTarget !== null.
 *
 * Phased state machine (mirrors MigrateLocalCharsModal):
 *   idle      — radio + textarea + Cancel + Submit
 *   submitting — disabled controls + spinner
 *   results.success — "Thanks — we'll review." + Close
 *   results.rateLimited — friendly 429 copy + Close (NOT a lockout)
 *   results.error — generic failure + Close
 *
 * Cancellable (unlike AcceptToSModal — reports are NOT legal gates):
 *   - Esc closes
 *   - Scrim click closes
 *   - Cancel button closes
 */
import { useEffect, useState } from 'react';
import type { ReportReason } from '../../../shared/ipc';
import { REPORT_REASONS } from '../../../shared/ipc';
import styles from './ReportModal.module.css';

const LABELS: Record<ReportReason, string> = {
  sexual_content_minors: 'Sexual content involving minors',
  hate_speech_harassment: 'Hate speech or harassment',
  copyright_infringement: 'Copyright infringement',
  other: 'Other',
};

type Phase =
  | { kind: 'idle' }
  | { kind: 'submitting' }
  | { kind: 'success' }
  | { kind: 'rate_limited'; message: string }
  | { kind: 'error'; message: string };

export interface ReportModalProps {
  characterId: string;
  characterName: string;
  onClose: () => void;
}

export function ReportModal({ characterId, characterName, onClose }: ReportModalProps) {
  const [reason, setReason] = useState<ReportReason>('other');
  const [detail, setDetail] = useState('');
  const [phase, setPhase] = useState<Phase>({ kind: 'idle' });
  const titleId = `report-modal-title-${characterId}`;

  useEffect(() => {
    const onKey = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose(); };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, [onClose]);

  const handleSubmit = async () => {
    setPhase({ kind: 'submitting' });
    try {
      const r = await window.sei.browseReport({
        characterId,
        reason,
        detail: detail.trim().length > 0 ? detail.trim() : undefined,
      });
      if (r.ok) {
        setPhase({ kind: 'success' });
      } else if (r.code === 'rate_limited') {
        setPhase({ kind: 'rate_limited', message: r.message });
      } else {
        setPhase({ kind: 'error', message: r.message });
      }
    } catch (e) {
      setPhase({ kind: 'error', message: (e as Error).message ?? 'submit_failed' });
    }
  };

  return (
    <div className={styles.scrim} onClick={onClose} role="dialog" aria-modal="true" aria-labelledby={titleId}>
      <div className={styles.modal} onClick={(e) => e.stopPropagation()}>
        <h2 id={titleId} className={styles.title}>Report "{characterName}"</h2>
        {phase.kind === 'idle' || phase.kind === 'submitting' ? (
          <>
            <fieldset className={styles.fieldset} disabled={phase.kind === 'submitting'}>
              <legend className={styles.legend}>Why are you reporting?</legend>
              {REPORT_REASONS.map((r) => (
                <label key={r} className={styles.radioRow}>
                  <input type="radio" name="reason" value={r} checked={reason === r} onChange={() => setReason(r)} />
                  <span>{LABELS[r]}</span>
                </label>
              ))}
            </fieldset>
            <label className={styles.detailLabel}>
              Optional detail (500 chars max)
              <textarea
                className={styles.detailField}
                value={detail}
                onChange={(e) => setDetail(e.target.value)}
                maxLength={500}
                disabled={phase.kind === 'submitting'}
                rows={4}
              />
              <span className={styles.charCount}>{detail.length}/500</span>
            </label>
            <div className={styles.footer}>
              <button type="button" className={styles.cancelBtn} onClick={onClose} disabled={phase.kind === 'submitting'}>Cancel</button>
              <button type="button" className={styles.submitBtn} onClick={handleSubmit} disabled={phase.kind === 'submitting'}>
                {phase.kind === 'submitting' ? 'Submitting…' : 'Submit report'}
              </button>
            </div>
          </>
        ) : null}
        {phase.kind === 'success' ? (
          <div className={styles.resultsPane}>
            <p>Thanks — we'll review.</p>
            <button className={styles.submitBtn} onClick={onClose}>Close</button>
          </div>
        ) : null}
        {phase.kind === 'rate_limited' ? (
          <div className={styles.resultsPane}>
            <p>{phase.message}</p>
            <button className={styles.submitBtn} onClick={onClose}>Close</button>
          </div>
        ) : null}
        {phase.kind === 'error' ? (
          <div className={styles.resultsPane}>
            <p>Sorry, we couldn't submit your report. {phase.message}</p>
            <button className={styles.cancelBtn} onClick={() => setPhase({ kind: 'idle' })}>Try again</button>
            <button className={styles.submitBtn} onClick={onClose}>Close</button>
          </div>
        ) : null}
      </div>
    </div>
  );
}
```
