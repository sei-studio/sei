```typescript
/**
 * Edge Function: moderate-character-images
 *
 * Synchronous CSAM scan on a portrait image. Called by cloudCharacterClient
 * (Plan 12-07 moderationGate.ts) AFTER the portrait byte upload to Storage
 * but BEFORE characters.shared flips to true (D-32b).
 *
 * Flow (mirrors delete-me/index.ts shape):
 *   1. CORS preflight
 *   2. Method !== POST → 405
 *   3. Missing Bearer → 401
 *   4. userClient.auth.getUser() — invalid JWT → 401
 *   5. Validate body { characterId, portraitUrl }
 *   6. callSightEngine(portraitUrl) — Pitfall 1: models=nudity-2.1,face-attributes
 *   7. adminClient.update(characters) with moderation_status + (shared=false on flag)
 *   8. Return { status: 'clean' | 'flagged', provider, category? }
 *
 * Pitfall 12: SightEngine free-tier 500/day. On HTTP error we HARD FAIL —
 * never publish an un-scanned image. The Edge Function returns 502 and the
 * caller (moderationGate) propagates that to the renderer as a friendly
 * "moderation service temporarily unavailable" message.
 */
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';
import { callSightEngine } from '../_shared/moderationProviders.ts';

const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response('Method not allowed', { status: 405, headers: corsHeaders });
  }

  const authHeader = req.headers.get('Authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    return new Response(JSON.stringify({ error: 'missing_jwt' }), { status: 401, headers: corsHeaders });
  }

  const userClient = createClient(
    SUPABASE_URL,
    Deno.env.get('SUPABASE_ANON_KEY')!,
    { global: { headers: { Authorization: authHeader } } },
  );
  const adminClient = createClient(SUPABASE_URL, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!);

  const { data: userData, error: userErr } = await userClient.auth.getUser();
  if (userErr || !userData.user) {
    return new Response(JSON.stringify({ error: 'invalid_jwt' }), { status: 401, headers: corsHeaders });
  }

  let body: { characterId?: unknown; portraitUrl?: unknown };
  try {
    body = await req.json();
  } catch {
    return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: corsHeaders });
  }

  if (typeof body.characterId !== 'string' || typeof body.portraitUrl !== 'string') {
    return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: corsHeaders });
  }
  const characterId = body.characterId;
  const portraitUrl = body.portraitUrl;

  let sightengine;
  try {
    sightengine = await callSightEngine(portraitUrl);
  } catch (e) {
    return new Response(
      JSON.stringify({ error: 'provider_error', detail: (e as Error).message }),
      { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  const update: Record<string, unknown> = {
    moderation_status: sightengine.flagged ? 'flagged' : 'clean',
    moderation_checked_at: new Date().toISOString(),
    moderation_provider: sightengine.provider,
  };
  if (sightengine.flagged) update.shared = false;

  const { error: updErr } = await adminClient
    .from('characters')
    .update(update)
    .eq('id', characterId);

  if (updErr) {
    // Compensating-write structure (mirrors delete-me): if SightEngine succeeded
    // but DB UPDATE failed, caller MUST retry — never assume clean.
    console.error('moderate-character-images db update failed', updErr);
    return new Response(
      JSON.stringify({ error: 'db_update_failed', detail: updErr.message }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  return new Response(
    JSON.stringify({
      status: sightengine.flagged ? 'flagged' : 'clean',
      provider: sightengine.provider,
      // category surfacing is intentionally minimal — never expose raw SightEngine
      // scores to the renderer (Pitfall: provider abstraction per D-32b).
      ...(sightengine.flagged ? { category: 'minor_or_sexual' } : {}),
    }),
    { headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
  );
});
```
