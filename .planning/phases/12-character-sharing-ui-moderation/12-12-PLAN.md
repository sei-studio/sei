Replace the basic handleAddToMine from 12-11 with:

```tsx
const [toast, setToast] = useState<{ kind: 'success' | 'error'; message: string } | null>(null);

useEffect(() => {
  if (toast === null) return;
  const t = window.setTimeout(() => setToast(null), 3000);
  return () => window.clearTimeout(t);
}, [toast]);

const handleAddToMine = async (entry: BrowseEntry) => {
  if (entry.inMyLibrary) return;  // optimistic guard
  setToast({ kind: 'success', message: `Adding "${entry.name}"…` });
  try {
    const result = await window.sei.charsOpenPrepare(entry.id);
    if (!result || (result as any).ok === false) {
      throw new Error((result as any)?.error ?? 'add_to_mine_failed');
    }
    // Refresh dance:
    //   1. useDataStore.refresh() — re-reads local files so Home tab sees new char
    //   2. useBrowseStore.refresh() — re-fetches Browse so inMyLibrary precomputed by main flips to true
    const { useDataStore } = await import('../lib/stores/useDataStore');
    await useDataStore.getState().refresh?.();  // optional chaining — refresh may be named differently
    await useBrowseStore.getState().refresh();
    setToast({ kind: 'success', message: `Added "${entry.name}" to your library.` });
  } catch (e) {
    setToast({ kind: 'error', message: `Couldn't add "${entry.name}". ${(e as Error).message ?? ''}` });
  }
};
```
