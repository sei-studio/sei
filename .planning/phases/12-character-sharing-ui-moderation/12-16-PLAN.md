```typescript
/**
 * Phase 12 — capabilities module.
 *
 * Resolves boot-time feature flags. Today, only BROWSE_ENABLED (D-36).
 *
 * Resolution order (per CONTEXT D-36 + D-36a):
 *   1. process.env.BROWSE_ENABLED === 'true' (dev/local override) → true
 *   2. config.json browse_enabled === true → true
 *   3. Otherwise → false
 *
 * The user MANUALLY flips browse_enabled in config.json (or via a future
 * admin command) after the D-36 pre-flight checklist passes:
 *   (a) DMCA agent registration confirmed (12-18)
 *   (b) backfill-moderate-existing returns processed=0 + no flagged in human-review
 *   (c) moderate-character-images + moderate-character-prompt + notify-report deployed
 *   (d) reports + characters.moderation_* migrations applied
 *
 * Production builds ship with config defaulting to false — Browse is hidden
 * until manual flip.
 */
import { readFile } from 'node:fs/promises';
import { paths } from './paths';
import { UserConfigSchema } from '../shared/characterSchema';

export interface Capabilities {
  browseEnabled: boolean;
}

export const DEFAULT_CAPABILITIES: Capabilities = { browseEnabled: false };

export async function readCapabilities(): Promise<Capabilities> {
  // Dev override (per D-36a)
  if (process.env.BROWSE_ENABLED === 'true') {
    return { browseEnabled: true };
  }
  // Config file
  try {
    const raw = await readFile(paths.configPath(), 'utf-8');
    const parsed = UserConfigSchema.parse(JSON.parse(raw));
    return { browseEnabled: parsed.browse_enabled === true };
  } catch {
    return DEFAULT_CAPABILITIES;
  }
}
```

Pitfall (per 12-PATTERNS): do NOT expose `process.env` to renderer. Main composes the OR and returns just the resolved boolean. The renderer has no `process.env.BROWSE_ENABLED` access (contextIsolation).
