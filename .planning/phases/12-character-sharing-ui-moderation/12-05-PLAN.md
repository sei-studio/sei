```typescript
/**
 * Edge Function: submit-report
 *
 * THE ONLY PATH to INSERT into reports (Pitfall 4 — no RLS insert-own
 * policy exists). Enforces D-34d rate limit (5 reports per reporter per
 * hour) BEFORE INSERT.
 *
 * On successful INSERT, the Database Webhook (12-01 tg_notify_report_inserted)
 * fans out to notify-report Edge Function (12-06) for email + Discord.
 *
 * Pitfall: reporter_id MUST come from auth.getUser() (JWT) — NEVER from
 * request body. Otherwise a malicious caller could spoof another user's
 * id and bypass per-reporter rate limit.
 */
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';

const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;
const RATE_LIMIT_PER_HOUR = 5;
const REASON_ENUM = ['sexual_content_minors', 'hate_speech_harassment', 'copyright_infringement', 'other'] as const;
type Reason = (typeof REASON_ENUM)[number];

const FRIENDLY_RATE_LIMITED_MESSAGE =
  "You've reported a lot in the last hour. Try again later if you still need to report.";

/** Pure function — counts items created within the last 60 minutes
 *  relative to `now`. Used by rate-limit check + unit-tested. */
export function countReportsInLastHour(
  reports: ReadonlyArray<{ created_at: string }>,
  now: Date,
): number {
  const oneHourAgoMs = now.getTime() - 60 * 60 * 1000;
  return reports.filter((r) => new Date(r.created_at).getTime() > oneHourAgoMs).length;
}

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response('Method not allowed', { status: 405, headers: corsHeaders });
  }

  const authHeader = req.headers.get('Authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    return new Response(JSON.stringify({ error: 'missing_jwt' }), { status: 401, headers: corsHeaders });
  }
  const userClient = createClient(
    SUPABASE_URL,
    Deno.env.get('SUPABASE_ANON_KEY')!,
    { global: { headers: { Authorization: authHeader } } },
  );
  const { data: userData, error: userErr } = await userClient.auth.getUser();
  if (userErr || !userData.user) {
    return new Response(JSON.stringify({ error: 'invalid_jwt' }), { status: 401, headers: corsHeaders });
  }
  const reporterId = userData.user.id;  // Pitfall mitigation: from JWT, never body

  // Validate body
  let body: { characterId?: unknown; reason?: unknown; detail?: unknown };
  try { body = await req.json(); }
  catch { return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: corsHeaders }); }

  if (typeof body.characterId !== 'string' || typeof body.reason !== 'string') {
    return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: corsHeaders });
  }
  if (!(REASON_ENUM as readonly string[]).includes(body.reason)) {
    return new Response(JSON.stringify({ error: 'bad_request', detail: 'invalid_reason' }), { status: 400, headers: corsHeaders });
  }
  const detail = typeof body.detail === 'string' ? body.detail.slice(0, 500) : null;

  const admin = createClient(SUPABASE_URL, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!);

  // Rate-limit check
  const { data: recent } = await admin
    .from('reports')
    .select('created_at')
    .eq('reporter_id', reporterId)
    .gte('created_at', new Date(Date.now() - 60 * 60 * 1000).toISOString())
    .order('created_at', { ascending: false })
    .limit(RATE_LIMIT_PER_HOUR + 1);

  if (recent && countReportsInLastHour(recent, new Date()) >= RATE_LIMIT_PER_HOUR) {
    return new Response(
      JSON.stringify({ ok: false, code: 'rate_limited', message: FRIENDLY_RATE_LIMITED_MESSAGE }),
      { status: 429, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  // INSERT (service_role bypasses RLS — which has NO insert policy)
  const { error: insErr } = await admin.from('reports').insert({
    reporter_id: reporterId,
    character_id: body.characterId,
    reason: body.reason as Reason,
    detail,
  });

  if (insErr) {
    return new Response(
      JSON.stringify({ ok: false, code: 'db_insert_failed', message: insErr.message }),
      { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  return new Response(JSON.stringify({ ok: true }), {
    status: 202,
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
});
```
