```typescript
/**
 * Edge Function: moderate-character-prompt
 *
 * Synchronous two-tier prompt moderation. Called by moderationGate.ts
 * (Plan 12-07) before flipping characters.shared = true.
 *
 * THRESHOLDS (D-33 — encoded as constants, NEVER as magic numbers in conditionals):
 *   sexual/minors      : ANY non-zero score → hard block (zero-tolerance)
 *   violence/graphic   : > 0.85 → hard block
 *   hate/threatening   : > 0.85 → hard block
 *   self-harm/intent   : > 0.85 → hard block
 * (Implemented in _shared/moderationProviders.ts interpretOpenAIModerationFlag.)
 *
 * Two-tier (D-33b):
 *   HARD tier  : input = name + persona_source (concatenated)
 *                flag → verdict='block' → publication rejected
 *   SOFT tier  : input = persona_expanded (only if provided + hard tier clean)
 *                flag → verdict='regenerate' → caller may re-expand with mod-steer
 *                       prompt + retry (Pitfall 6: cap retries at 2 in caller)
 *
 * Friendly message per D-33c (NEVER expose raw OpenAI category names):
 *   "We can't publish this character because the persona description hits
 *    our content guidelines. Edit the persona and try again, or save it
 *    as private."
 */
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';
import { callOpenAIModeration } from '../_shared/moderationProviders.ts';

const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;

const FRIENDLY_BLOCK_MESSAGE =
  "We can't publish this character because the persona description hits our content guidelines. " +
  'Edit the persona and try again, or save it as private.';

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response('Method not allowed', { status: 405, headers: corsHeaders });
  }

  const authHeader = req.headers.get('Authorization');
  if (!authHeader?.startsWith('Bearer ')) {
    return new Response(JSON.stringify({ error: 'missing_jwt' }), { status: 401, headers: corsHeaders });
  }
  const userClient = createClient(
    SUPABASE_URL,
    Deno.env.get('SUPABASE_ANON_KEY')!,
    { global: { headers: { Authorization: authHeader } } },
  );
  const { data: userData, error: userErr } = await userClient.auth.getUser();
  if (userErr || !userData.user) {
    return new Response(JSON.stringify({ error: 'invalid_jwt' }), { status: 401, headers: corsHeaders });
  }

  let body: { name?: unknown; persona_source?: unknown; persona_expanded?: unknown };
  try { body = await req.json(); }
  catch { return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: corsHeaders }); }

  if (typeof body.name !== 'string' || typeof body.persona_source !== 'string') {
    return new Response(JSON.stringify({ error: 'bad_request' }), { status: 400, headers: corsHeaders });
  }
  const personaExpanded = typeof body.persona_expanded === 'string' ? body.persona_expanded : '';

  // HARD tier: name + persona_source (D-33b)
  const hardText = `${body.name}\n\n${body.persona_source}`;
  let hard;
  try { hard = await callOpenAIModeration(hardText); }
  catch (e) {
    return new Response(
      JSON.stringify({ error: 'provider_error', detail: (e as Error).message }),
      { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }
  if (hard.blocked) {
    return new Response(
      JSON.stringify({
        verdict: 'block',
        tier: 'hard',
        friendlyMessage: FRIENDLY_BLOCK_MESSAGE,
        // flaggedCategoriesInternal is for server-side logging only — caller
        // (moderationGate) should NOT surface to renderer.
        flaggedCategoriesInternal: hard.flaggedCategories,
      }),
      { headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
    );
  }

  // SOFT tier: persona_expanded (if provided)
  if (personaExpanded.length > 0) {
    let soft;
    try { soft = await callOpenAIModeration(personaExpanded); }
    catch (e) {
      return new Response(
        JSON.stringify({ error: 'provider_error', detail: (e as Error).message }),
        { status: 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
      );
    }
    if (soft.blocked) {
      return new Response(
        JSON.stringify({
          verdict: 'regenerate',
          tier: 'soft',
          flaggedCategoriesInternal: soft.flaggedCategories,
        }),
        { headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
      );
    }
  }

  return new Response(JSON.stringify({ verdict: 'clean' }), {
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
});
```

Note: this function does NO database writes — `moderationGate.ts` (12-07) is responsible for persisting `moderation_text_provider` + `moderation_text_checked_at` after a clean verdict.
