```tsx
/**
 * DmcaContactModal — D-35a surface (a). Mount: SettingsScreen Legal panel.
 *
 * Cancellable (Esc + scrim close) — info modal not a legal gate.
 *
 * Per CONTEXT D-35: DMCA agent details are PUBLIC by Copyright Office mandate
 * (no private/anon registration option). The template below renders the
 * placeholder text that 12-18 swaps for real values after registration.
 */
import { useEffect } from 'react';
import styles from './DmcaContactModal.module.css';

// IMPORTANT: 12-18 updates these constants after dmca.copyright.gov registration.
// Until then, the modal renders the placeholder text — see 12-18 for the swap.
const AGENT_NAME = '[Designated Agent — pending registration]';
const AGENT_ADDRESS = '[Mailing address pending registration]';
const AGENT_EMAIL = 'dmca@sei.app';
const DIRECTORY_LISTING_URL = 'https://dmca.copyright.gov/list';

export interface DmcaContactModalProps { onClose: () => void }

export function DmcaContactModal({ onClose }: DmcaContactModalProps) {
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose(); };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, [onClose]);

  return (
    <div className={styles.scrim} onClick={onClose} role="dialog" aria-modal="true" aria-labelledby="dmca-title">
      <div className={styles.modal} onClick={(e) => e.stopPropagation()}>
        <h2 id="dmca-title" className={styles.title}>DMCA Designated Agent</h2>
        <p className={styles.lede}>
          To report copyright infringement on a Sei character, send a written DMCA notice to our designated agent:
        </p>
        <dl className={styles.details}>
          <dt>Name</dt><dd>{AGENT_NAME}</dd>
          <dt>Mailing address</dt><dd className={styles.address}>{AGENT_ADDRESS}</dd>
          <dt>Email</dt><dd><a href={`mailto:${AGENT_EMAIL}`} onClick={(e) => { e.preventDefault(); window.sei.openExternal(`mailto:${AGENT_EMAIL}`); }}>{AGENT_EMAIL}</a></dd>
        </dl>
        <p className={styles.note}>
          Designated Agent listings are public per US Copyright Office rules. You can view our registration in the public Directory.
        </p>
        <div className={styles.footer}>
          <button className={styles.linkBtn} onClick={() => window.sei.openExternal(DIRECTORY_LISTING_URL)}>Open Directory listing</button>
          <button className={styles.linkBtn} onClick={() => window.sei.openExternal(`mailto:${AGENT_EMAIL}`)}>Email DMCA agent</button>
          <button className={styles.closeBtn} onClick={onClose}>Close</button>
        </div>
      </div>
    </div>
  );
}
```
