/**
 * Typed Edge Function client for the moderation domain.
 * Mirrors src/main/cloud/cloudCharacterClient.ts shape.
 * Used by: src/main/ipc.ts browse:report handler + operator backfill IPC.
 */
import { callEdgeFunction } from '../auth/edgeFunctionClient';

const MODERATION_TIMEOUT_MS = 30_000;

export interface SubmitReportArgs {
  characterId: string;
  reason: 'sexual_content_minors' | 'hate_speech_harassment' | 'copyright_infringement' | 'other';
  detail?: string;
}

export type SubmitReportResult =
  | { ok: true }
  | { ok: false; code: 'rate_limited' | 'network' | 'unauthenticated' | 'bad_request'; message: string };

export async function callSubmitReport(jwt: string, args: SubmitReportArgs): Promise<SubmitReportResult> {
  try {
    const response = await callEdgeFunction<{ ok?: boolean; code?: string; message?: string }>(
      'submit-report',
      { jwt, body: args, timeoutMs: MODERATION_TIMEOUT_MS },
    );
    if (response?.ok === true) return { ok: true };
    if (response?.code === 'rate_limited') {
      return { ok: false, code: 'rate_limited', message: response.message ?? "You've reported a lot recently — try again later." };
    }
    return { ok: false, code: 'network', message: response?.message ?? 'submit_report_failed' };
  } catch (e) {
    const msg = (e as Error).message ?? 'submit_report_failed';
    if (msg.includes('401') || msg.includes('missing_jwt') || msg.includes('invalid_jwt')) {
      return { ok: false, code: 'unauthenticated', message: 'Please sign in to report.' };
    }
    if (msg.includes('429')) {
      return { ok: false, code: 'rate_limited', message: "You've reported a lot recently — try again later." };
    }
    return { ok: false, code: 'network', message: msg };
  }
}

export interface BackfillResult { processed: number; flagged: number; errors: number; nextCursor: 'more' | 'done' }

export async function callBackfillModerate(jwt: string): Promise<BackfillResult> {
  return callEdgeFunction<BackfillResult>('backfill-moderate-existing', {
    jwt, body: {}, timeoutMs: 60_000,  // batch of 100 may take a minute
  });
}

// --- Phase 12 — Browse listing (signed-in OR signed-out — LIB-04) ---
const BrowseListSchema = z.object({
  query: z.string().max(200),
  limit: z.number().int().min(1).max(50),
  offset: z.number().int().min(0),
});
ipcMain.handle(IpcChannel.browse.list, async (_event, args: unknown): Promise<{ entries: BrowseEntry[]; hasMore: boolean }> => {
  const parsed = BrowseListSchema.parse(args);
  const { getClient } = await import('./cloud/supabaseClient');
  const { getStoragePublicUrl } = await import('./cloud/cloudCharacterClient');
  const { data: rows, error } = await getClient().rpc('search_public_characters', {
    search_query: parsed.query,
    page_limit: parsed.limit,
    page_offset: parsed.offset,
  });
  if (error) {
    console.error('browse:list rpc failed', error.message);
    return { entries: [], hasMore: false };
  }

  // Compute inMyLibrary set: per Researcher correction #4, predicate is
  // "is this in useDataStore.characters" (local file list). Main reads
  // the characters directory and produces the set.
  const { listLocalCharacters } = await import('./characterStore');
  const localIds = new Set((await listLocalCharacters()).map((c) => c.id));

  const entries: BrowseEntry[] = (rows ?? []).map((row: any) => ({
    id: row.id,
    name: row.name,
    personaSnippet: typeof row.persona_source === 'string'
      ? (row.persona_source.length > 120 ? row.persona_source.slice(0, 120).trimEnd() + '…' : row.persona_source)
      : '',
    creatorLabel: row.owner ? `by user-${String(row.owner).slice(0, 4)}` : 'by anonymous',
    portraitUrl: row.portrait_image ? getStoragePublicUrl('portraits', row.portrait_image) : null,
    skinUrl: row.skin_png_sha256 ? getStoragePublicUrl('skins', `${row.owner}/${row.id}.png`) : null,
    updatedAt: row.updated_at,
    inMyLibrary: localIds.has(row.id),
  }));
  return { entries, hasMore: entries.length === parsed.limit };
});

// --- Phase 12 — Submit report (gated by isCloudWriteAllowed) ---
const BrowseReportSchema = z.object({
  characterId: IdSchema,
  reason: z.enum(['sexual_content_minors', 'hate_speech_harassment', 'copyright_infringement', 'other']),
  detail: z.string().max(500).optional(),
});
ipcMain.handle(IpcChannel.browse.report, async (_event, args: unknown) => {
  const parsed = BrowseReportSchema.parse(args);
  const { isCloudWriteAllowed } = await import('./auth/authState');
  if (!(await isCloudWriteAllowed())) {
    return { ok: false as const, code: 'unauthenticated' as const, message: 'Please sign in to report.' };
  }
  const { getClient } = await import('./cloud/supabaseClient');
  const { data: { session } } = await getClient().auth.getSession();
  if (!session?.access_token) {
    return { ok: false as const, code: 'unauthenticated' as const, message: 'Please sign in to report.' };
  }
  const { callSubmitReport } = await import('./cloud/moderationEdgeClient');
  return callSubmitReport(session.access_token, parsed);
});

// --- Phase 12 — Publish-with-moderation (gated) ---
ipcMain.handle(IpcChannel.browse.publishWithModeration, async (_event, characterId: unknown) => {
  const parsed = IdSchema.parse(characterId);
  const { isCloudWriteAllowed } = await import('./auth/authState');
  if (!(await isCloudWriteAllowed())) {
    return { ok: false as const, code: 'CLOUD_MODERATION_PROVIDER_UNAVAILABLE' as const, message: 'Please sign in and accept the Terms of Service.' };
  }
  const { publishWithModeration } = await import('./cloud/moderationGate');
  const { callEdgeFunction } = await import('./auth/edgeFunctionClient');
  const { upsertCharacter, getStoragePublicUrl: _u } = await import('./cloud/cloudCharacterClient');
  const { loadCharacterRaw } = await import('./characterStore');
  const { getClient } = await import('./cloud/supabaseClient');
  const { data: { session } } = await getClient().auth.getSession();
  if (!session?.access_token) {
    return { ok: false as const, code: 'CLOUD_MODERATION_PROVIDER_UNAVAILABLE' as const, message: 'Please sign in.' };
  }
  const SUPABASE_URL = process.env.SUPABASE_URL ?? '';
  const result = await publishWithModeration(parsed, {
    callEdgeFunction: (name, opts) => callEdgeFunction(name, opts),
    upsertCharacter,
    reExpandPersona: async (id) => {
      const { expandAndSaveCharacter } = await import('./expandAndSaveCharacter');
      const expanded = await expandAndSaveCharacter(id, { moderationRetry: true });
      return expanded;
    },
    getCharacter: async (id) => loadCharacterRaw(id),
    getJwt: async () => session.access_token,
    supabaseUrl: SUPABASE_URL,
  });
  if (result.ok) return { ok: true as const };
  return { ok: false as const, code: result.code, message: result.friendlyMessage };
});

// --- Phase 12 — Capabilities (BROWSE_ENABLED) ---
ipcMain.handle(IpcChannel.capabilities.get, async (): Promise<{ browseEnabled: boolean }> => {
  const { readCapabilities } = await import('./capabilities');
  return readCapabilities();
});
