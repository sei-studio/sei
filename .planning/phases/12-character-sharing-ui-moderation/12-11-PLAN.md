```tsx
/**
 * BrowseCard — Phase 12 public character card.
 *
 * Renders a public BrowseEntry with:
 *   - PixelPortrait (top, gradient background, seeded by id+name)
 *   - Name overlay
 *   - Creator label (e.g., "by user-a1b2")
 *   - Persona snippet (~120 chars + ellipsis)
 *   - Top-LEFT: Report button (small ghost icon)
 *   - Hover overlay: Add-to-Mine button OR "Already in My Library" pill
 *
 * Per CONTEXT D-31c + Pitfall 7:
 *   - Report button calls e.stopPropagation() on click
 *   - inMyLibrary is precomputed by main; component does NOT subscribe to
 *     useAuthStore / useSyncStore / useCloudCharactersStore
 */
import type { BrowseEntry } from '../../../shared/ipc';
import { PixelPortrait } from './PixelPortrait';
import { pickPalette } from '../lib/pickPalette';
import characterStyles from './CharacterCard.module.css';
import styles from './BrowseCard.module.css';

export interface BrowseCardProps {
  entry: BrowseEntry;
  onOpen: () => void;
  onAddToMine: () => void;
  onReport: () => void;
  theme: 'light' | 'dark';
}

export function BrowseCard({ entry, onOpen, onAddToMine, onReport, theme }: BrowseCardProps) {
  const palette = pickPalette(entry.id + entry.name, theme);
  return (
    <div
      className={`${characterStyles.card} ${styles.cardExtras}`}
      onClick={onOpen}
      role="button"
      tabIndex={0}
      onKeyDown={(e) => { if (e.key === 'Enter' || e.key === ' ') onOpen(); }}
      aria-label={`Open ${entry.name}`}
    >
      <button
        className={styles.reportBtn}
        aria-label={`Report ${entry.name}`}
        onClick={(e) => { e.stopPropagation(); onReport(); }}
      >
        ⚐
      </button>
      <div className={characterStyles.portraitWrap}>
        <div className={characterStyles.gradient} style={{ background: palette.gradient }} />
        <PixelPortrait
          src={entry.portraitUrl ?? undefined}
          alt={entry.name}
          theme={theme}
        />
        <div className={characterStyles.nameOverlay}>{entry.name}</div>
        <div className={characterStyles.hoverOverlay}>
          {entry.inMyLibrary ? (
            <span className={styles.alreadyPill}>Already in My Library</span>
          ) : (
            <button
              className={styles.addBtn}
              onClick={(e) => { e.stopPropagation(); onAddToMine(); }}
            >
              + Add to Mine
            </button>
          )}
        </div>
      </div>
      <div className={styles.metaRow}>
        <span className={styles.creatorMeta}>{entry.creatorLabel}</span>
      </div>
      <div className={styles.personaSnippet}>{entry.personaSnippet}</div>
    </div>
  );
}
```

```tsx
function BrowseGrid() {
  const { entries, query, loading, exhausted, error, setQuery, loadMore, refresh } = useBrowseStore();
  const sentinelRef = useRef<HTMLDivElement>(null);
  const [reportTarget, setReportTarget] = useState<BrowseEntry | null>(null);
  const theme = useThemeMode(); // existing hook

  useEffect(() => { void refresh(); }, []);
  useEffect(() => {
    const el = sentinelRef.current; if (!el) return;
    const obs = new IntersectionObserver((e) => { if (e[0].isIntersecting) void loadMore(); }, { rootMargin: '200px' });
    obs.observe(el);
    return () => obs.disconnect();
  }, [loadMore]);

  const handleOpen = (entry: BrowseEntry) => {
    // v1.0: navigate to character page (CharacterPage.tsx renders cloud-only chars too per 11-19).
    window.location.hash = `#/character/${entry.id}`;
  };
  const handleAddToMine = async (entry: BrowseEntry) => {
    try {
      await window.sei.charsOpenPrepare(entry.id);  // Phase 11 cacheOnDemand reuse
      void refresh();
    } catch (e) {
      console.error('add to mine failed', e);
    }
  };
  const handleReport = (entry: BrowseEntry) => {
    setReportTarget(entry);
    // 12-13 ships ReportModal; for now this opens nothing — leave the state hook in place.
  };

  return (
    <div className={styles.browse}>
      <input className={styles.searchField} type="search" value={query} onChange={(e) => setQuery(e.target.value)} placeholder="Search characters..." aria-label="Search public characters" />
      {error ? <div className={styles.error}>Couldn't load Browse. {error}</div> : null}
      {entries.length === 0 && !loading && !error ? (
        <div className={styles.empty}>No public characters yet — be the first to share one.</div>
      ) : null}
      <div className={styles.grid}>
        {entries.map((entry) => (
          <BrowseCard
            key={entry.id}
            entry={entry}
            onOpen={() => handleOpen(entry)}
            onAddToMine={() => handleAddToMine(entry)}
            onReport={() => handleReport(entry)}
            theme={theme}
          />
        ))}
      </div>
      {!exhausted ? <div ref={sentinelRef} className={styles.sentinel} aria-hidden /> : null}
      {loading ? <div className={styles.loading}>Loading…</div> : null}
      {/* 12-13 will replace this with <ReportModal characterId={reportTarget.id} ... onClose={() => setReportTarget(null)} /> */}
    </div>
  );
}
```
