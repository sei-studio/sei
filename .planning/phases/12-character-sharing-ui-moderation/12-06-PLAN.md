```typescript
/**
 * Edge Function: notify-report
 *
 * Invoked by Database Webhook (12-01 tg_notify_report_inserted) AFTER a
 * row INSERTs into public.reports. Trigger ordering (alphabetical):
 * reports_auto_hide_trigger fires BEFORE reports_after_insert_webhook,
 * so by the time this function runs, characters.shared has already been
 * flipped to false if the 3-distinct-reporter threshold was crossed.
 *
 * Outbound fan-out:
 *   1. Discord webhook (DISCORD_REPORT_WEBHOOK_URL) — always
 *   2. Email to dmca@sei.app via Resend (RESEND_API_KEY) — always
 *   3. If auto-hidden: email the creator via Resend (admin.auth.admin.getUserById)
 *
 * Security (Pitfall in 12-PATTERNS.md):
 *   - Database Webhooks are NOT HMAC-signed on Supabase free tier.
 *     Defense: verify Authorization header carries our service_role_key
 *     (set in both the DB webhook config and Deno.env). Any direct call
 *     without that exact bearer is rejected 401.
 *   - Discord webhook URL is SECRET; never log it (even on error).
 *   - SMTP failures DO NOT roll back the report INSERT — log + 200 so
 *     webhook doesn't retry-storm.
 */
import { createClient } from '@supabase/supabase-js';
import { corsHeaders } from '../_shared/cors.ts';

const SUPABASE_URL = Deno.env.get('SUPABASE_URL')!;
const NOTIFY_TIMEOUT_MS = 15_000;

Deno.serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders });
  if (req.method !== 'POST') {
    return new Response('Method not allowed', { status: 405, headers: corsHeaders });
  }

  // Verify the call comes from our DB webhook (service_role bearer match).
  const auth = req.headers.get('Authorization');
  const expected = `Bearer ${Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!}`;
  if (auth !== expected) {
    return new Response(JSON.stringify({ error: 'unauthorized' }), { status: 401, headers: corsHeaders });
  }

  let body: { report_id?: unknown; character_id?: unknown };
  try { body = await req.json(); }
  catch { return new Response('bad_request', { status: 400, headers: corsHeaders }); }
  if (typeof body.report_id !== 'string' || typeof body.character_id !== 'string') {
    return new Response('bad_request', { status: 400, headers: corsHeaders });
  }

  const admin = createClient(SUPABASE_URL, Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!);
  const [{ data: report }, { data: char }] = await Promise.all([
    admin.from('reports').select('id, reporter_id, character_id, reason, detail, created_at').eq('id', body.report_id).single(),
    admin.from('characters').select('id, name, owner, shared').eq('id', body.character_id).single(),
  ]);

  if (!report || !char) {
    return new Response(JSON.stringify({ error: 'not_found' }), { status: 404, headers: corsHeaders });
  }

  const autoHidden = char.shared === false;
  const tag = autoHidden ? '🚨 AUTO-HIDDEN' : '⚠️ New report';

  // --- 1. Discord webhook ---
  try {
    const discordUrl = Deno.env.get('DISCORD_REPORT_WEBHOOK_URL');
    if (discordUrl) {
      const ctrl = new AbortController();
      const t = setTimeout(() => ctrl.abort(), NOTIFY_TIMEOUT_MS);
      try {
        await fetch(discordUrl, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            content:
              `${tag} for character \`${char.name}\` (\`${char.id}\`)\n` +
              `Reason: **${report.reason}**\n` +
              `Detail: ${report.detail ?? '(none)'}\n` +
              `Reporter: \`${report.reporter_id}\``,
          }),
          signal: ctrl.signal,
        });
      } finally { clearTimeout(t); }
    }
  } catch (e) {
    console.error('discord_post_failed', (e as Error).message);  // never log URL
  }

  // --- 2. Email to dmca@sei.app via Resend ---
  try {
    const resendKey = Deno.env.get('RESEND_API_KEY');
    if (resendKey) {
      const ctrl = new AbortController();
      const t = setTimeout(() => ctrl.abort(), NOTIFY_TIMEOUT_MS);
      try {
        await fetch('https://api.resend.com/emails', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${resendKey}` },
          body: JSON.stringify({
            from: 'reports@sei.app',
            to: ['dmca@sei.app'],
            subject: `${tag}: ${char.name} (${report.reason})`,
            text:
              `Character: ${char.name} (${char.id})\n` +
              `Owner: ${char.owner}\n` +
              `Reason: ${report.reason}\n` +
              `Detail: ${report.detail ?? '(none)'}\n` +
              `Reporter: ${report.reporter_id}\n` +
              `Report ID: ${report.id}\n` +
              `Auto-hidden: ${autoHidden}\n`,
          }),
          signal: ctrl.signal,
        });
      } finally { clearTimeout(t); }
    }
  } catch (e) {
    console.error('resend_post_failed', (e as Error).message);
  }

  // --- 3. If auto-hidden, email the creator too ---
  if (autoHidden) {
    try {
      const { data: ownerUser } = await admin.auth.admin.getUserById(char.owner);
      const ownerEmail = ownerUser?.user?.email;
      const resendKey = Deno.env.get('RESEND_API_KEY');
      if (ownerEmail && resendKey) {
        const ctrl = new AbortController();
        const t = setTimeout(() => ctrl.abort(), NOTIFY_TIMEOUT_MS);
        try {
          await fetch('https://api.resend.com/emails', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${resendKey}` },
            body: JSON.stringify({
              from: 'reports@sei.app',
              to: [ownerEmail],
              subject: `Your character "${char.name}" was auto-hidden pending review`,
              text:
                `Hi,\n\nYour public character "${char.name}" was auto-hidden ` +
                `because it received multiple reports in a 24-hour window. ` +
                `A human will review the reports and either restore the ` +
                `character or contact you with next steps.\n\n` +
                `If you believe this was in error, reply to this email ` +
                `(dmca@sei.app) with the character name and we'll take a look.\n\n` +
                `— Sei moderation`,
            }),
            signal: ctrl.signal,
          });
        } finally { clearTimeout(t); }
      }
    } catch (e) {
      console.error('creator_email_failed', (e as Error).message);
    }
  }

  // Always 200 — webhook MUST NOT retry-storm even if downstream fails.
  return new Response(JSON.stringify({ ok: true, auto_hidden: autoHidden }), {
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
});
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| DB Webhook → Edge Function | Bearer service_role crosses |
| Edge Function → Discord/Resend | Outbound HTTPS with secrets |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-06-01 | Spoofing | Direct caller without DB webhook context | mitigate | Authorization Bearer compared verbatim to env-stored service_role_key |
| T-12-06-02 | Information Disclosure | Discord webhook URL leaked via logs | mitigate | `console.error` calls log only failure message, never URL/key; explicit comment in code |
| T-12-06-03 | DoS | Attacker triggers many fake reports to flood SMTP | mitigate | Upstream rate limit (12-05 RATE_LIMIT_PER_HOUR=5) caps per-reporter inflow |
| T-12-06-04 | Tampering | Modified payload sets character_id to wrong row | accept | DB webhook is internal; we trust report_id and character_id come from same INSERT row (FK enforces consistency) |
| T-12-06-05 | Repudiation | SMTP delivery fails silently | mitigate | console.error logged + Supabase dashboard logs retained 7d; webhook does NOT retry (200 returned) per spec |
| T-12-06-06 | Information Disclosure | Creator email includes report detail | accept | Creator already sees their own character was reported; reason category disclosed but reporter_id is NOT disclosed to creator |
</threat_model>
