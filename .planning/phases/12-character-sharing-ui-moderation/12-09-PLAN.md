```typescript
/**
 * useBrowseStore — Phase 12 Browse tab state.
 *
 * Lifecycle:
 *   - CharactersScreen's BrowseTab useEffect calls refresh() on first mount
 *     (analog: useSyncStore.init pattern — store does NOT self-bootstrap).
 *   - setQuery cancels any pending debounce + schedules a fresh refresh
 *     after 250ms (D-31a; debounce lives HERE not in the screen so multiple
 *     consumers stay consistent).
 *   - loadMore appends next page (offset += pageSize) unless loading or
 *     exhausted (Pitfall 8 in-flight guard).
 *
 * Pitfall in 12-PATTERNS.md: debounce timer is `window.setTimeout` so the
 * TS type is `number` (browser), NOT `NodeJS.Timeout`. Mirrors useDataStore.
 */
import { create } from 'zustand';
import type { BrowseEntry } from '../../../../shared/ipc';

const PAGE_SIZE = 24;
const DEBOUNCE_MS = 250;

interface BrowseState {
  entries: BrowseEntry[];
  query: string;
  loading: boolean;
  exhausted: boolean;
  offset: number;
  error: string | null;
  _debounceHandle: number | null;
}

interface BrowseActions {
  setQuery: (q: string) => void;
  loadMore: () => Promise<void>;
  refresh: () => Promise<void>;
  reset: () => void;
}

const initial: BrowseState = {
  entries: [],
  query: '',
  loading: false,
  exhausted: false,
  offset: 0,
  error: null,
  _debounceHandle: null,
};

export const useBrowseStore = create<BrowseState & BrowseActions>((set, get) => ({
  ...initial,

  setQuery: (q) => {
    const prev = get()._debounceHandle;
    if (prev !== null) window.clearTimeout(prev);
    set({ query: q });
    const handle = window.setTimeout(() => {
      set({ _debounceHandle: null });
      void get().refresh();
    }, DEBOUNCE_MS);
    set({ _debounceHandle: handle });
  },

  loadMore: async () => {
    const s = get();
    if (s.loading || s.exhausted) return;  // Pitfall 8 guard
    set({ loading: true, error: null });
    try {
      const page = await window.sei.browseList({ query: s.query, limit: PAGE_SIZE, offset: s.offset });
      set({
        entries: [...s.entries, ...page.entries],
        offset: s.offset + page.entries.length,
        exhausted: !page.hasMore,
        loading: false,
      });
    } catch (e) {
      set({ loading: false, error: (e as Error).message ?? 'browse_load_failed' });
    }
  },

  refresh: async () => {
    set({ loading: true, entries: [], offset: 0, exhausted: false, error: null });
    try {
      const s = get();
      const page = await window.sei.browseList({ query: s.query, limit: PAGE_SIZE, offset: 0 });
      set({
        entries: page.entries,
        offset: page.entries.length,
        exhausted: !page.hasMore,
        loading: false,
      });
    } catch (e) {
      set({ loading: false, error: (e as Error).message ?? 'browse_load_failed' });
    }
  },

  reset: () => {
    const prev = get()._debounceHandle;
    if (prev !== null) window.clearTimeout(prev);
    set({ ...initial });
  },
}));
```
